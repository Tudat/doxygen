27 {
28  // Locally store the atmosphere table file name.
29  atmosphereTableFile_ = atmosphereTableFile;

31  Eigen::MatrixXd containerOfAtmosphereTableFileData
32  = input_output::readMatrixFromFile( atmosphereTableFile_, " \t", "%" );
33 
34  // Check whether data is present in the file.
35  if ( containerOfAtmosphereTableFileData.rows( ) < 1
36  || containerOfAtmosphereTableFileData.cols( ) < 1 )
37  {
38  std::string errorMessage = "The atmosphere table file " + atmosphereTableFile_ + " is empty";
39  throw std::runtime_error( errorMessage );
40  }
41 
42  // Initialize vectors.
43  altitudeData_.resize( containerOfAtmosphereTableFileData.rows( ) );
44  densityData_.resize( containerOfAtmosphereTableFileData.rows( ) );
45  pressureData_.resize( containerOfAtmosphereTableFileData.rows( ) );
46  temperatureData_.resize( containerOfAtmosphereTableFileData.rows( ) );
47 
48  // Loop through all the strings stored in the container and store the data
49  // in the right Eigen::VectorXd.
50  for ( int i = 0; i < containerOfAtmosphereTableFileData.rows( ); i++ )
51  {
52  altitudeData_[ i ] = containerOfAtmosphereTableFileData( i, 0 );
53  densityData_[ i ] = containerOfAtmosphereTableFileData( i, 1 );
54  pressureData_[ i ] = containerOfAtmosphereTableFileData( i, 2 );
55  temperatureData_[ i ] = containerOfAtmosphereTableFileData( i, 3 );
56  }
