53 
54  // Iterate over link ends
55  for( typename std::map< observation_models::LinkEnds, boost::shared_ptr< observation_models::ObservationModel<
56  ObservationSize, ObservationScalarType, TimeType> > >::const_iterator

58  observationModelIterator != observationModels.end( ); observationModelIterator++ )

66  }

69  std::vector< boost::shared_ptr< observation_models::LightTimeCorrection > > singleObservableCorrectionList;
70

150  }
