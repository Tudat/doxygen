31 boost::shared_ptr< aerodynamics::FlightConditions > createFlightConditions(
32  const boost::shared_ptr< Body > bodyWithFlightConditions,
33  const boost::shared_ptr< Body > centralBody,
34  const std::string& nameOfBodyUndergoingAcceleration,
35  const std::string& nameOfBodyExertingAcceleration,
36  const boost::function< double( ) > angleOfAttackFunction,
37  const boost::function< double( ) > angleOfSideslipFunction,
38  const boost::function< double( ) > bankAngleFunction,
39  const boost::function< void( const double ) > angleUpdateFunction )
40 {
41  // Check whether all required environment models are set.
42  if( centralBody->getAtmosphereModel( ) == NULL )
43  {
44  throw std::runtime_error(
45  "Error when making flight conditions, body " + nameOfBodyExertingAcceleration +
46  " has no atmosphere model." );
47  }
48 
49  if( centralBody->getShapeModel( ) == NULL )
50  {
51  throw std::runtime_error(
52  "Error when making flight conditions, body " + nameOfBodyExertingAcceleration +
53  " has no shape model." );
54  }
55 
56  if( centralBody->getRotationalEphemeris( ) == NULL )
57  {
58  throw std::runtime_error(
59  "Error when making flight conditions, body " + nameOfBodyExertingAcceleration +
60  " has no rotation model." );
61  }
62 
63  if( bodyWithFlightConditions->getAerodynamicCoefficientInterface( ) == NULL )
64  {
65  throw std::runtime_error(
66  "Error when making flight conditions, body " + nameOfBodyUndergoingAcceleration +
67  " has no aerodynamic coefficients." );
68  }
69 
70 
71  // Create function to rotate state from intertial to body-fixed frame.
72  boost::function< Eigen::Quaterniond( ) > rotationToFrameFunction =
73  boost::bind( &Body::getCurrentRotationToLocalFrame, centralBody );
74  boost::function< Eigen::Matrix3d( ) > rotationMatrixToFrameDerivativeFunction =
75  boost::bind( &Body::getCurrentRotationMatrixDerivativeToLocalFrame, centralBody );
76 
77  boost::function< Eigen::Matrix< double, 6, 1 >( ) > bodyStateFunction = boost::bind( &Body::getState, bodyWithFlightConditions );
78  boost::function< Eigen::Matrix< double, 6, 1 >( ) > centralBodyStateFunction = boost::bind( &Body::getState, centralBody );
79 
80  boost::function< Eigen::Matrix< double, 6, 1 >( ) > relativeBodyFixedStateFunction =
81  boost::bind( &ephemerides::transformRelativeStateToFrame< double >,
82  bodyStateFunction, centralBodyStateFunction,
83  rotationToFrameFunction,
84  rotationMatrixToFrameDerivativeFunction );
85 
86  // Create aerodynamic angles calculator and set in flight conditions.
87  boost::shared_ptr< reference_frames::AerodynamicAngleCalculator > aerodynamicAngleCalculator =
88  boost::make_shared< reference_frames::AerodynamicAngleCalculator >(
89  relativeBodyFixedStateFunction,
90  boost::bind( &simulation_setup::Body::getCurrentRotationToGlobalFrame, centralBody ),
91  nameOfBodyExertingAcceleration, 1,
92  angleOfAttackFunction, angleOfSideslipFunction, bankAngleFunction, angleUpdateFunction );
93 
94  // Add wind model if present
95  if( centralBody->getAtmosphereModel( )->getWindModel( ) != NULL )
96  {
97  if( centralBody->getShapeModel( ) == NULL )
98  {
99  std::cerr << "Warnning, body " << nameOfBodyExertingAcceleration << " has wind model, but no shape model, cannot compute wind as function of altitude " << std::endl;
100  }
101  else
102  {
103  aerodynamicAngleCalculator->setWindModel(
104  centralBody->getAtmosphereModel( )->getWindModel( ), centralBody->getShapeModel( ) );
105  }
106  }
107 
108 
109  // Create flight conditions.
110  boost::function< double( const std::string& )> controlSurfaceDeflectionFunction;
111  if( bodyWithFlightConditions->getVehicleSystems( ) != NULL )
112  {
113  controlSurfaceDeflectionFunction = boost::bind(
114  &system_models::VehicleSystems::getCurrentControlSurfaceDeflection,
115  bodyWithFlightConditions->getVehicleSystems( ), _1 );
116  }
117  boost::shared_ptr< aerodynamics::FlightConditions > flightConditions =
118  boost::make_shared< aerodynamics::FlightConditions >(
119  centralBody->getAtmosphereModel( ), centralBody->getShapeModel( ),
120  bodyWithFlightConditions->getAerodynamicCoefficientInterface( ), aerodynamicAngleCalculator,
121  controlSurfaceDeflectionFunction );
122 
123  return flightConditions;
