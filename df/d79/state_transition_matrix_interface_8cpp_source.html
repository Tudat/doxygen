99 Eigen::MatrixXd MultiArcCombinedStateTransitionAndSensitivityMatrixInterface::getCombinedStateTransitionAndSensitivityMatrix( const double evaluationTime )
100 {
101  Eigen::MatrixXd combinedStateTransitionMatrix = Eigen::MatrixXd::Zero(
102  stateTransitionMatrixSize_, stateTransitionMatrixSize_ + sensitivityMatrixSize_ );
103 
104  int currentArc = lookUpscheme_->findNearestLowerNeighbour( evaluationTime );
105 
106  // Set Phi and S matrices.
107 
108  combinedStateTransitionMatrix.block( 0, 0, stateTransitionMatrixSize_, stateTransitionMatrixSize_ ) =
109  stateTransitionMatrixInterpolators_.at( currentArc )->interpolate( evaluationTime );
110  combinedStateTransitionMatrix.block( 0, stateTransitionMatrixSize_, stateTransitionMatrixSize_, sensitivityMatrixSize_ ) =
111  sensitivityMatrixInterpolators_.at( currentArc )->interpolate( evaluationTime );
