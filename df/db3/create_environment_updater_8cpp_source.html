23 void checkValidityOfRequiredEnvironmentUpdates(
24  const std::map< propagators::EnvironmentModelsToUpdate, std::vector< std::string > >&
25  requestedUpdates, const simulation_setup::NamedBodyMap& bodyMap )
26 {
27  using namespace propagators;
28 
29  // Iterate over all environment update types.
30  for( std::map< propagators::EnvironmentModelsToUpdate,
31  std::vector< std::string > >::const_iterator
32  updateIterator = requestedUpdates.begin( );
33  updateIterator != requestedUpdates.end( ); updateIterator++ )
34  {
35  // Iterate over all updated bodies.
36  for( unsigned int i = 0; i < updateIterator->second.size( ); i++ )
37  {
38  // Ignore global required updates.
39  if( updateIterator->second.at( i ) != "" )
40  {
41  // Check if body exists.
42  if( bodyMap.count( updateIterator->second.at( i ) ) == 0 )
43  {
44  throw std::runtime_error(
45  "Error when making environment model update settings, could not find body "
46  + boost::lexical_cast< std::string>( updateIterator->second.at( i ) ) );
47  }
48 
49  // Check if requested environment model exists.
50  switch( updateIterator->first )
51  {
52  case body_transational_state_update:
53  {
54  if( bodyMap.at( updateIterator->second.at( i ) )->getEphemeris( ) == NULL )
55  {
56  throw std::runtime_error(
57  "Error when making environment model update settings, could not find ephemeris of body "
58  + boost::lexical_cast< std::string>( updateIterator->second.at( i ) ) );
59  }
60  break;
61  }
62  case body_rotational_state_update:
63  {
64  if( ( bodyMap.at( updateIterator->second.at( i ) )->
65  getRotationalEphemeris( ) == NULL ) &&
66  ( bodyMap.at( updateIterator->second.at( i ) )->getDependentOrientationCalculator( ) == NULL ) )
67  {
68  throw std::runtime_error(
69  "Error when making environment model update settings, could not find rotational ephemeris or dependent orientation calculator of body "
70  + boost::lexical_cast< std::string>( updateIterator->second.at( i ) ) );
71  }
72  break;
73  }
74  case spherical_harmonic_gravity_field_update:
75  {
76  boost::shared_ptr< gravitation::SphericalHarmonicsGravityField >
77  gravityFieldModel =
78  boost::dynamic_pointer_cast< gravitation::SphericalHarmonicsGravityField >(
79  bodyMap.at( updateIterator->second.at( i ) )->getGravityFieldModel( ) );
80  if( gravityFieldModel == NULL )
81  {
82  throw std::runtime_error(
83  "Error when making environment model update settings, could not find spherical harmonic gravity field of body "
84  + boost::lexical_cast< std::string>( updateIterator->second.at( i ) ) );
85  }
86  break;
87  }
88  case vehicle_flight_conditions_update:
89  {
90  boost::shared_ptr< aerodynamics::FlightConditions > flightConditions = bodyMap.at(
91  updateIterator->second.at( i ) )->getFlightConditions( );
92  if( flightConditions == NULL )
93  {
94  throw std::runtime_error(
95  "Error when making environment model update settings, could not find flight conditions of body "
96  + boost::lexical_cast< std::string>( updateIterator->second.at( i ) ) );
97  }
98  break;
99  }
100  case radiation_pressure_interface_update:
101  {
102  std::map< std::string, boost::shared_ptr< electro_magnetism::RadiationPressureInterface > >
103  radiationPressureInterfaces = bodyMap.at(
104  updateIterator->second.at( i ) )->getRadiationPressureInterfaces( );
105  if( radiationPressureInterfaces.size( ) == 0 )
106  {
107  throw std::runtime_error(
108  "Error when making environment model update settings, could not find radiation pressure interface of body "
109  + boost::lexical_cast< std::string>( updateIterator->second.at( i ) ) );
110  }
111  break;
112  }
113  case body_mass_update:
114  if( bodyMap.at( updateIterator->second.at( i ) )->getBodyMassFunction( ) == NULL )
115  {
116  throw std::runtime_error(
117  "Error when making environment model update settings, no body mass function of body "
118  + boost::lexical_cast< std::string>( updateIterator->second.at( i ) ) );
119  }
120 
121  break;
122  }
123  }
124  }

137  std::map< propagators::EnvironmentModelsToUpdate, std::vector< std::string > > singleTorqueUpdateNeeds;
138 
139  // Iterate over all bodies on which torques are being exerting
140  for( TorqueModelMap::const_iterator acceleratedBodyIterator = torqueModels.begin( );
141  acceleratedBodyIterator != torqueModels.end( ); acceleratedBodyIterator++ )
142  {
143  // Iterate over all bodies exerting on current body
144  for( SingleBodyTorqueModelMap::const_iterator torqueModelIterator = acceleratedBodyIterator->second.begin( );
145  torqueModelIterator != acceleratedBodyIterator->second.end( ); torqueModelIterator++ )
146  {
147  singleTorqueUpdateNeeds.clear( );
148  for( unsigned int i = 0; i < torqueModelIterator->second.size( ); i++ )
149  {
150  AvailableTorque currentTorqueModelType =
151  getTorqueModelType( torqueModelIterator->second.at( i ) );
152 
153  switch( currentTorqueModelType )
154  {
155  case second_order_gravitational_torque:
156  break;
157  case aerodynamic_torque:
158  singleTorqueUpdateNeeds[ body_rotational_state_update ].push_back(
159  torqueModelIterator->first );
160  singleTorqueUpdateNeeds[ vehicle_flight_conditions_update ].push_back(
161  acceleratedBodyIterator->first );
162  break;
163  default:
164  std::cerr<<"Error, update information not found for torque model "<<currentTorqueModelType<<std::endl;
165  break;
166  }
167  }
168 
169  checkValidityOfRequiredEnvironmentUpdates( singleTorqueUpdateNeeds, bodyMap );
170  addEnvironmentUpdates( environmentModelsToUpdate, singleTorqueUpdateNeeds );
171  }
172  }
173

377  // Add requested updates of current acceleration model to

379  addEnvironmentUpdates( environmentModelsToUpdate, singleAccelerationUpdateNeeds );
380  }
381  }

383  return environmentModelsToUpdate;

437 std::map< propagators::EnvironmentModelsToUpdate,
438 std::vector< std::string > > createFullEnvironmentUpdaterSettings(
439  const simulation_setup::NamedBodyMap& bodyMap )
440 {
441  using namespace basic_astrodynamics;
442  using namespace electro_magnetism;
443  using namespace gravitation;
444  using namespace propagators;
445 
446  std::map< propagators::EnvironmentModelsToUpdate,
447  std::vector< std::string > > environmentModelsToUpdate;
448  std::map< propagators::EnvironmentModelsToUpdate,
449  std::vector< std::string > > singleAccelerationUpdateNeeds;
450 
451  // Iterate over all bodies.
452  for( simulation_setup::NamedBodyMap::const_iterator
453  bodyIterator = bodyMap.begin( ); bodyIterator != bodyMap.end( ); bodyIterator++ )
454  {
455  singleAccelerationUpdateNeeds.clear( );
456 
457  // Check if current body is a vehicle.
458 
459  // Check if current body has flight conditions set.
460  if( bodyIterator->second ->getFlightConditions( ) != NULL )
461  {
462  // If vehicle has flight conditions, add flight conditions update function to update list.
463  singleAccelerationUpdateNeeds[ vehicle_flight_conditions_update ].
464  push_back( bodyIterator->first );
465  }
466 
467  // Get body radiation pressure interface(s) (one per source)
468  std::map< std::string,
469  boost::shared_ptr< RadiationPressureInterface > > radiationPressureInterfaces
470  = bodyIterator->second->getRadiationPressureInterfaces( );
471 
472  // Add each interface update function to update list.
473  for( std::map< std::string, boost::shared_ptr< RadiationPressureInterface > >::iterator
474  iterator = radiationPressureInterfaces.begin( );
475  iterator != radiationPressureInterfaces.end( ); iterator++ )
476  {
477  singleAccelerationUpdateNeeds[ radiation_pressure_interface_update ].
478  push_back( bodyIterator->first );
479  }
480 
481  // If body has rotation model, update rotational state in each time step.;
482  if( ( bodyIterator->second->getRotationalEphemeris( ) != NULL ) ||
483  ( bodyIterator->second->getDependentOrientationCalculator( ) != NULL ) )
484  {
485  singleAccelerationUpdateNeeds[ body_rotational_state_update ].
486  push_back( bodyIterator->first );
487  }
488 
489 
490  boost::shared_ptr< TimeDependentSphericalHarmonicsGravityField > gravityField =
491  boost::dynamic_pointer_cast< TimeDependentSphericalHarmonicsGravityField >
492  ( bodyIterator->second->getGravityFieldModel( ) );
493  if( gravityField != NULL )
494  {
495  singleAccelerationUpdateNeeds[ spherical_harmonic_gravity_field_update ].
496  push_back( bodyIterator->first );
497  }
498 
499  singleAccelerationUpdateNeeds[ body_mass_update ].push_back( bodyIterator->first );
500 
501  // Check whether requested updates are possible.
502  checkValidityOfRequiredEnvironmentUpdates( singleAccelerationUpdateNeeds, bodyMap );
503 
504  // Add requested updates of current acceleration model to full list of environment updates.
505  addEnvironmentUpdates( environmentModelsToUpdate, singleAccelerationUpdateNeeds );
506  }
507  return environmentModelsToUpdate;
508 }
