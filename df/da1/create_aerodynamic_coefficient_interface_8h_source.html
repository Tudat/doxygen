838 template< unsigned int NumberOfIndependentVariables >
839 boost::shared_ptr< AerodynamicCoefficientSettings >
840 readGivenSizeTabulatedAerodynamicCoefficientsFromFiles(
841  const std::map< int, std::string > forceCoefficientFiles,
842  const std::map< int, std::string > momentCoefficientFiles,
843  const double referenceLength,
844  const double referenceArea,
845  const double lateralReferenceLength,
846  const Eigen::Vector3d& momentReferencePoint,
847  const std::vector< aerodynamics::AerodynamicCoefficientsIndependentVariables > independentVariableNames,
848  const bool areCoefficientsInAerodynamicFrame = 1,
849  const bool areCoefficientsInNegativeAxisDirection = 1 )
850 {
851  std::pair< boost::multi_array< Eigen::Vector3d, static_cast< size_t >( NumberOfIndependentVariables ) >,
852  std::vector< std::vector< double > > >
853  aerodynamicForceCoefficients =
854  input_output::readAerodynamicCoefficients< NumberOfIndependentVariables >( forceCoefficientFiles );
855  std::pair< boost::multi_array< Eigen::Vector3d, static_cast< size_t >( NumberOfIndependentVariables ) >,
856  std::vector< std::vector< double > > >
857  aerodynamicMomentCoefficients =
858  input_output::readAerodynamicCoefficients< NumberOfIndependentVariables >( momentCoefficientFiles );
859 
860  if( !input_output::compareIndependentVariables(
861  aerodynamicForceCoefficients.second, aerodynamicMomentCoefficients.second ) )
862  {
863  throw std::runtime_error( "Error when creating aerodynamic coefficient settings from file, force and moment independent variables are inconsistent" );
864  }
865 
866  if( independentVariableNames.size( ) != NumberOfIndependentVariables )
867  {
868  throw std::runtime_error( "Error when creating aerodynamic coefficient settings from file, input sizes are inconsistent" );
869  }
870 
871  boost::shared_ptr< TabulatedAerodynamicCoefficientSettings< NumberOfIndependentVariables > > tabulatedCoefficients =
872  boost::make_shared< TabulatedAerodynamicCoefficientSettings< NumberOfIndependentVariables > >(
873  aerodynamicForceCoefficients.second, aerodynamicForceCoefficients.first, aerodynamicMomentCoefficients.first,
874  referenceLength, referenceArea, lateralReferenceLength, momentReferencePoint, independentVariableNames,
875  areCoefficientsInAerodynamicFrame, areCoefficientsInNegativeAxisDirection );
876  tabulatedCoefficients->setForceCoefficientsFiles( forceCoefficientFiles );
877  tabulatedCoefficients->setMomentCoefficientsFiles( momentCoefficientFiles );
878  return tabulatedCoefficients;
