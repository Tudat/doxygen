20 boost::shared_ptr< aerodynamics::AerodynamicTorque > createAerodynamicTorqueModel(
21  const boost::shared_ptr< simulation_setup::Body > bodyUndergoingTorque,
22  const boost::shared_ptr< simulation_setup::Body > bodyExertingTorque,
23  const std::string& nameOfBodyUndergoingTorque,
24  const std::string& nameOfBodyExertingTorque )
25 {
26  // Check existence of required environment models
27  if( bodyUndergoingTorque->getAerodynamicCoefficientInterface( ) == NULL )
28  {
29  throw std::runtime_error( "Error when making aerodynamic torque, body " +
30  nameOfBodyUndergoingTorque +
31  "has no aerodynamic coefficients." );
32  }
33 
34  if( bodyExertingTorque->getAtmosphereModel( ) == NULL )
35  {
36  throw std::runtime_error( "Error when making aerodynamic torque, central body " +
37  nameOfBodyExertingTorque + " has no atmosphere model.");
38  }
39 
40  if( bodyExertingTorque->getShapeModel( ) == NULL )
41  {
42  throw std::runtime_error( "Error when making aerodynamic torque, central body " +
43  nameOfBodyExertingTorque + " has no shape model." );
44  }
45 
46  // Retrieve flight conditions; create object if not yet extant.
47  boost::shared_ptr< aerodynamics::FlightConditions > bodyFlightConditions =
48  bodyUndergoingTorque->getFlightConditions( );
49 
50  if( bodyFlightConditions == NULL )
51  {
52  bodyUndergoingTorque->setFlightConditions(
53  createFlightConditions( bodyUndergoingTorque,
54  bodyExertingTorque,
55  nameOfBodyUndergoingTorque,
56  nameOfBodyExertingTorque ) );
57  bodyFlightConditions = bodyUndergoingTorque->getFlightConditions( );
58  }
59 
60  // Retrieve frame in which aerodynamic coefficients are defined.
61  boost::shared_ptr< aerodynamics::AerodynamicCoefficientInterface > aerodynamicCoefficients =
62  bodyUndergoingTorque->getAerodynamicCoefficientInterface( );
63  reference_frames::AerodynamicsReferenceFrames torqueFrame;
64  if( aerodynamicCoefficients->getAreCoefficientsInAerodynamicFrame( ) )
65  {
66  torqueFrame = reference_frames::aerodynamic_frame;
67  }
68  else
69  {
70  torqueFrame = reference_frames::body_frame;
71  }
72 
73  // Create function to transform from frame of aerodynamic coefficienrs to that of propagation.
74  boost::function< Eigen::Vector3d( const Eigen::Vector3d& ) > toPropagationFrameTransformation;
75  toPropagationFrameTransformation =
76  reference_frames::getAerodynamicForceTransformationFunction(
77  bodyFlightConditions->getAerodynamicAngleCalculator( ),
78  torqueFrame,
79  boost::bind( &Body::getCurrentRotationToGlobalFrame, bodyExertingTorque ),
80  reference_frames::body_frame );
81 
82 
83  boost::function< Eigen::Vector3d( ) > coefficientFunction =
84  boost::bind( &aerodynamics::AerodynamicCoefficientInterface::getCurrentMomentCoefficients,
85  aerodynamicCoefficients );
86  boost::function< Eigen::Vector3d( ) > coefficientInPropagationFrameFunction =
87  boost::bind( &reference_frames::transformVectorFunctionFromVectorFunctions,
88  coefficientFunction, toPropagationFrameTransformation );
89 
90  // Create torque model.
91  return boost::make_shared< aerodynamics::AerodynamicTorque >(
92  coefficientInPropagationFrameFunction,
93  boost::bind( &aerodynamics::FlightConditions::getCurrentDensity, bodyFlightConditions ),
94  boost::bind( &aerodynamics::FlightConditions::getCurrentAirspeed, bodyFlightConditions ),
95  boost::bind( &aerodynamics::AerodynamicCoefficientInterface::getReferenceArea, aerodynamicCoefficients ),
96  boost::bind( &aerodynamics::AerodynamicCoefficientInterface::getReferenceLengths, aerodynamicCoefficients ),
97  aerodynamicCoefficients->getAreCoefficientsInNegativeAxisDirection( ) );
98 }

138 boost::shared_ptr< basic_astrodynamics::TorqueModel > createTorqueModel(
139  const boost::shared_ptr< simulation_setup::Body > bodyUndergoingTorque,
140  const boost::shared_ptr< simulation_setup::Body > bodyExertingTorque,
141  const boost::shared_ptr< TorqueSettings > torqueSettings,
142  const std::string& nameOfBodyUndergoingTorque,
143  const std::string& nameOfBodyExertingTorque )
144 {
145  boost::shared_ptr< basic_astrodynamics::TorqueModel > torqueModel;
146 
147  switch( torqueSettings->torqueType_ )
148  {
149  case basic_astrodynamics::second_order_gravitational_torque:
150  {
151  torqueModel = createSecondDegreeGravitationalTorqueModel(
152  bodyUndergoingTorque, bodyExertingTorque, nameOfBodyUndergoingTorque, nameOfBodyExertingTorque );
153  break;
154  }
155  case basic_astrodynamics::aerodynamic_torque:
156  {
157  torqueModel = createAerodynamicTorqueModel(
158  bodyUndergoingTorque, bodyExertingTorque, nameOfBodyUndergoingTorque, nameOfBodyExertingTorque );
159  break;
160  }
161  default:
162  throw std::runtime_error(
163  "Error, did not recognize type " + boost::lexical_cast< std::string >( torqueSettings->torqueType_ ) +
164  " when making torque model" );
165  }
166 
167  return torqueModel;
168 }
