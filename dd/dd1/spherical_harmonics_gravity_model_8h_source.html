306  void updateMembers( const double currentTime = TUDAT_NAN )
307  {
308  if( !( this->currentTime_ == currentTime ) )
309  {
310  cosineHarmonicCoefficients = getCosineHarmonicsCoefficients( );
311  sineHarmonicCoefficients = getSineHarmonicsCoefficients( );
312  rotationToIntegrationFrame_ = rotationFromBodyFixedToIntegrationFrameFunction_( );
313  this->updateBaseMembers( );
314  currentAcceleration_ = rotationToIntegrationFrame_ *
315  computeGeodesyNormalizedGravitationalAccelerationSum(
316  rotationToIntegrationFrame_.inverse( ) * (
317  this->positionOfBodySubjectToAcceleration - this->positionOfBodyExertingAcceleration ),
318  gravitationalParameter,
319  equatorialRadius,
320  cosineHarmonicCoefficients,
321  sineHarmonicCoefficients, sphericalHarmonicsCache_ );
322  }
