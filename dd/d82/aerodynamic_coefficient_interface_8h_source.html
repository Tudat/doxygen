140  void updateCurrentControlSurfaceCoefficientsCoefficients(
141  const std::string& currentControlSurface,
142  std::vector< double > controlSurfaceIndependentVariables )
143  {
144  if( controlSurfaceIncrementInterfaces_.count( currentControlSurface ) == 0 )
145  {
146  throw std::runtime_error( "Error when updating coefficients, could not fid control surface " + currentControlSurface );
147  }
148  controlSurfaceIncrementInterfaces_.at( currentControlSurface )->updateCurrentCoefficients(
149  controlSurfaceIndependentVariables );
150  currentForceCoefficients_ +=
151  controlSurfaceIncrementInterfaces_.at( currentControlSurface )->getCurrentForceCoefficients( );
152  currentMomentCoefficients_ +=
153  controlSurfaceIncrementInterfaces_.at( currentControlSurface )->getCurrentMomentCoefficients( );
154  }
