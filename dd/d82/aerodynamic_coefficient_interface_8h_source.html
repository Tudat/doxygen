144  void updateCurrentControlSurfaceCoefficientsCoefficients(
145  const std::string& currentControlSurface,
146  std::vector< double > controlSurfaceIndependentVariables )
147  {
148  if( controlSurfaceIncrementInterfaces_.count( currentControlSurface ) == 0 )
149  {
150  throw std::runtime_error( "Error when updating coefficients, could not fid control surface " + currentControlSurface );
151  }
152  controlSurfaceIncrementInterfaces_.at( currentControlSurface )->updateCurrentCoefficients(
153  controlSurfaceIndependentVariables );
154  currentForceCoefficients_ +=
155  controlSurfaceIncrementInterfaces_.at( currentControlSurface )->getCurrentForceCoefficients( );
156  currentMomentCoefficients_ +=
157  controlSurfaceIncrementInterfaces_.at( currentControlSurface )->getCurrentMomentCoefficients( );
158  }

168  void updateFullCurrentCoefficients(
169  const std::vector< double >& independentVariables,
170  const std::map< std::string, std::vector< double > >& controlSurfaceIndependentVariables =
171  std::map< std::string, std::vector< double > > ( ) )
172  {
173  updateCurrentCoefficients( independentVariables );
174 
175  for( std::map< std::string, std::vector< double > >::const_iterator controlSurfaceIterator =
176  controlSurfaceIndependentVariables.begin( ); controlSurfaceIterator != controlSurfaceIndependentVariables.end( );
177  controlSurfaceIterator++ )
178  {
179  updateCurrentControlSurfaceCoefficientsCoefficients(
180  controlSurfaceIterator->first, controlSurfaceIterator->second );
181  }
182  }
