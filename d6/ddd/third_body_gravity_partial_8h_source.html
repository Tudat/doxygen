35 template< typename DirectGravityPartial >
36 basic_astrodynamics::AvailableAcceleration getAccelerationTypeOfThirdBodyGravity(
37  const boost::shared_ptr< DirectGravityPartial > directGravityPartial )
38 {
39  using namespace basic_astrodynamics;
40  AvailableAcceleration accelerationType;
41 
42  // Check type of direct partial derivative.
43  if( boost::dynamic_pointer_cast< CentralGravitationPartial >( directGravityPartial ) != NULL )
44  {
45  accelerationType = third_body_central_gravity;
46  }
47  else if( boost::dynamic_pointer_cast< SphericalHarmonicsGravityPartial >( directGravityPartial ) != NULL )
48  {
49  accelerationType = third_body_spherical_harmonic_gravity;
50  }
51  else if( boost::dynamic_pointer_cast< MutualSphericalHarmonicsGravityPartial >( directGravityPartial ) != NULL )
52  {
53  accelerationType = third_body_mutual_spherical_harmonic_gravity;
54  }
55 
56  else
57  {
58  throw std::runtime_error( "Error when getting third body partial type, type not identified" );
59  }
60  return accelerationType;
61 }

361  boost::shared_ptr< estimatable_parameters::EstimatableParameter< double > > parameter )
362  {
363  // Check parameter dependency of direct acceleration.
364  std::pair< boost::function< void( Eigen::MatrixXd& ) >, int > partialFunctionFromDirectGravity =
365  partialOfDirectGravityOnBodyUndergoingAcceleration_->getParameterPartialFunction( parameter );
366  if( partialFunctionFromDirectGravity.second > 0 )
367  {
368  partialOfDirectGravityOnBodyUndergoingAcceleration_->setParameterPartialUpdateFunction( parameter );
369  }
370 
371  // Check parameter dependency of indirect acceleration.
372  std::pair< boost::function< void( Eigen::MatrixXd& ) >, int > partialFunctionFromCentralGravity =
373  partialOfDirectGravityOnCentralBody_->getParameterPartialFunction( parameter );
374  if( partialFunctionFromCentralGravity.second > 0 )
375  {
376  partialOfDirectGravityOnCentralBody_->setParameterPartialUpdateFunction( parameter );
377  }
378 
379  // If either dependency exists, create combined partial.
380  if( partialFunctionFromCentralGravity.second > 0 || partialFunctionFromDirectGravity.second > 0 )
381  {
382  parameterDoublePartialFunctions_[ parameter ] =
383  getCombinedCurrentDoubleParameterFunction(
384  partialOfDirectGravityOnBodyUndergoingAcceleration_,
385  partialOfDirectGravityOnCentralBody_,
386  parameter, partialFunctionFromDirectGravity.second, partialFunctionFromCentralGravity.second, 1 );
387  isCurrentDoubleParameterPartialSet_[ parameter ] = 0;
388  currentDoubleParameterPartials_[ parameter ] = Eigen::MatrixXd( accelerationSize_, 1 );
389  }
390  return std::max( partialFunctionFromDirectGravity.second, partialFunctionFromCentralGravity.second );
