169 std::pair< double, double > getTabulatedEphemerisSafeInterval( const boost::shared_ptr< Ephemeris > ephemeris )
170 {
171  // Initialize return pair
172  std::pair< double, double > safeInterval = std::make_pair( TUDAT_NAN, TUDAT_NAN );
173 
174  // Check input consistency
175  if( !isTabulatedEphemeris( ephemeris ) )
176  {
177  throw std::runtime_error( "Error wgen getting tabulated ephemeris safe interval, input is not a tabulated ephemeris" );
178  }
179  // Identify type of tabulated ephemeris, and call associated safe interval function
180  else if( boost::dynamic_pointer_cast< TabulatedCartesianEphemeris< double, double > >( ephemeris ) != NULL )
181  {
182  safeInterval = boost::dynamic_pointer_cast< TabulatedCartesianEphemeris< double, double > >(
183  ephemeris )->getSafeInterpolationInterval( );
184  }
185  else if( boost::dynamic_pointer_cast< TabulatedCartesianEphemeris< long double, double > >( ephemeris ) != NULL )
186  {
187  safeInterval = boost::dynamic_pointer_cast< TabulatedCartesianEphemeris< long double, double > >(
188  ephemeris )->getSafeInterpolationInterval( );
189  }
190  else if ( boost::dynamic_pointer_cast< TabulatedCartesianEphemeris< long double, Time > >( ephemeris ) != NULL )
191  {
192  safeInterval = boost::dynamic_pointer_cast< TabulatedCartesianEphemeris< long double, Time > >(
193  ephemeris )->getSafeInterpolationInterval( );
194  }
195  else if( boost::dynamic_pointer_cast< TabulatedCartesianEphemeris< double, Time > >( ephemeris ) != NULL )
196  {
197  safeInterval = boost::dynamic_pointer_cast< TabulatedCartesianEphemeris< double, Time > >(
198  ephemeris )->getSafeInterpolationInterval( );
199  }
200  return safeInterval;
201 }
