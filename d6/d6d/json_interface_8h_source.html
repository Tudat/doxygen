130  {
131  // Check if any keys in jsonObject_ haven't been used
132  checkUnusedKeys( jsonObject_, applicationOptions_->unusedKey_ );
133 
134  if ( profiling )

139  }
140

142  if ( ! applicationOptions_->fullSettingsFile_.empty( ) )
143  {
144  exportAsJson( applicationOptions_->fullSettingsFile_ );
145  }
146 
147  // Print message on propagation start if requested
148  if ( applicationOptions_->notifyOnPropagationStart_ )
149  {
150  std::cout << "Propagation of file " << inputFilePath_ << " started." << std::endl;
151  }
152 
153  // Run simulation
154  dynamicsSimulator_->integrateEquationsOfMotion( propagatorSettings_->getInitialStates( ) );
155 
156  // Print message on propagation termination if requested
157  if ( applicationOptions_->notifyOnPropagationTermination_ )
158  {

160  {
161  std::cout << "SUCCESS: propagation of file " << inputFilePath_ << " terminated with no errors."
162  << std::endl;
163  }
164  else

166  std::cout << "FAILURE: propagation of file " << inputFilePath_ << " terminated with errors."
167  << std::endl;
168  }
169  }
170

172  {
173  std::cout << "run: " << std::chrono::duration_cast< std::chrono::milliseconds >(

597 template< typename TimeType, typename StateScalarType >
599 void to_json( nlohmann::json& jsonObject,
600  const std::shared_ptr< JsonSimulationManager< TimeType, StateScalarType > >& jsonSimulationManager )
601 {
602  if ( ! jsonSimulationManager )
603  {
604  return;
605  }
606 
607  // integrator
608  jsonObject[ Keys::integrator ] = jsonSimulationManager->getIntegratorSettings( );
609 
610  // spice
611  assignIfNotnullptr( jsonObject, Keys::spice, jsonSimulationManager->getSpiceSettings( ) );
612 
613  // bodies
614  jsonObject[ Keys::globalFrameOrigin ] = jsonSimulationManager->getGlobalFrameOrigin( );
615  jsonObject[ Keys::globalFrameOrientation ] = jsonSimulationManager->getGlobalFrameOrientation( );
616  jsonObject[ Keys::bodies ] = jsonSimulationManager->getBodySettingsMap( );
617 
618  // export
619  assignIfNotEmpty( jsonObject, Keys::xport, jsonSimulationManager->getExportSettingsVector( ) );
620 
621  // options
622  jsonObject[ Keys::options ] = jsonSimulationManager->getApplicationOptions( );
623 
624  // propagation + termination + options.printInterval
625  propagators::to_json( jsonObject, jsonSimulationManager->getPropagatorSettings( ) );
626 }
