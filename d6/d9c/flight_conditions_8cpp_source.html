76 {
77  if( ( independentVariable == mach_number_dependent ) ||

89  customCoefficientDependencies_[ independentVariable ] = coefficientDependency;

94 void FlightConditions::updateConditions( const double currentTime )
95 {
96  if( !( currentTime == currentTime_ ) )
97  {
98  currentTime_ = currentTime;
99 
100  // Update aerodynamic angles (but not angles w.r.t. body-fixed frame).
101  if( aerodynamicAngleCalculator_!= NULL )
102  {
103  aerodynamicAngleCalculator_->update( currentTime, false );
104  }
105 
106  // Calculate state of vehicle in global frame and corotating frame.
107  currentBodyCenteredAirspeedBasedBodyFixedState_ = bodyCenteredPseudoBodyFixedStateFunction_( );
108 
109  updateAerodynamicCoefficientInput( );
110 
111  // Update angles from aerodynamic to body-fixed frame (if relevant).
112  if( aerodynamicAngleCalculator_!= NULL )
113  {
114  aerodynamicAngleCalculator_->update( currentTime, true );
115  updateAerodynamicCoefficientInput( );
116  }
117 
118  // Update aerodynamic coefficients.
119  aerodynamicCoefficientInterface_->updateFullCurrentCoefficients(
120  aerodynamicCoefficientIndependentVariables_, controlSurfaceAerodynamicCoefficientIndependentVariables_ );
121  }

125 double FlightConditions::getAerodynamicCoefficientIndependentVariable(
126  const AerodynamicCoefficientsIndependentVariables independentVariableType,
127  const std::string& secondaryIdentifier )
128 {
129  double currentIndependentVariable;
130  switch( independentVariableType )
131  {
132  //Calculate Mach number if needed.
133  case mach_number_dependent:
134  currentIndependentVariable = getCurrentMachNumber( );
135  break;
136  //Get angle of attack if needed.
137  case angle_of_attack_dependent:
138 
139  if( aerodynamicAngleCalculator_== NULL )
140  {
141  throw std::runtime_error( "Error, aerodynamic angle calculator is null, but require angle of attack" );
142  }
143  currentIndependentVariable = aerodynamicAngleCalculator_->getAerodynamicAngle(
144  reference_frames::angle_of_attack );
145  break;
146  //Get angle of sideslip if needed.
147  case angle_of_sideslip_dependent:
148  if( aerodynamicAngleCalculator_== NULL )
149  {
150  throw std::runtime_error( "Error, aerodynamic angle calculator is null, but require angle of sideslip" );
151  }
152  currentIndependentVariable = aerodynamicAngleCalculator_->getAerodynamicAngle(
153  reference_frames::angle_of_sideslip );
154  break;
155  case altitude_dependent:
156  if( aerodynamicAngleCalculator_== NULL )
157  {
158  throw std::runtime_error( "Error, aerodynamic angle calculator is null, but require angle of sideslip" );
159  }
160  currentIndependentVariable = getCurrentAltitude( );
161  break;
162  case control_surface_deflection_dependent:
163  {
164  try
165  {
166  currentIndependentVariable = controlSurfaceDeflectionFunction_( secondaryIdentifier );
167  }
168  catch( std::runtime_error )
169  {
170  throw std::runtime_error( "Error, control surface " + secondaryIdentifier + "not recognized when updating coefficients" );
171  }
172 
173  break;
174  }
175  default:
176  if( customCoefficientDependencies_.count( independentVariableType ) == 0 )
177  {
178  throw std::runtime_error( "Error, did not recognize aerodynamic coefficient dependency "
179  + std::to_string( independentVariableType ) );
180  }
181  else
182  {
183  currentIndependentVariable = customCoefficientDependencies_.at( independentVariableType )( );
184  }
185  }
186 
187  return currentIndependentVariable;
188 }

191 void FlightConditions::updateAerodynamicCoefficientInput( )
192 {
193  aerodynamicCoefficientIndependentVariables_.clear( );
194  // Calculate independent variables for aerodynamic coefficients.
195  for( unsigned int i = 0; i < aerodynamicCoefficientInterface_->getNumberOfIndependentVariables( ); i++ )
196  {
197  aerodynamicCoefficientIndependentVariables_.push_back(
198  getAerodynamicCoefficientIndependentVariable(
199  aerodynamicCoefficientInterface_->getIndependentVariableName( i ) ) );
200  }
201 
202  controlSurfaceAerodynamicCoefficientIndependentVariables_.clear( );
203  for( unsigned int i = 0; i < aerodynamicCoefficientInterface_->getNumberOfControlSurfaces( ); i++ )
204  {
205  std::string currentControlSurface = aerodynamicCoefficientInterface_->getControlSurfaceName( i );
206  for( unsigned int j = 0; j < aerodynamicCoefficientInterface_->getNumberOfControlSurfaceIndependentVariables( currentControlSurface ); j++ )
207  {
208  controlSurfaceAerodynamicCoefficientIndependentVariables_[ currentControlSurface ].push_back(
209  getAerodynamicCoefficientIndependentVariable(
210  aerodynamicCoefficientInterface_->getControlSurfaceIndependentVariableName(
211  currentControlSurface, j ), currentControlSurface ) );
212  }
213  }
