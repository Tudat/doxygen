104  RelativisticAccelerationPartial(
105  const boost::shared_ptr< relativity::RelativisticAccelerationCorrection > accelerationModel,
106  const std::string& acceleratedBody,
107  const std::string& acceleratingBody ):
108  AccelerationPartial( acceleratedBody, acceleratingBody, basic_astrodynamics::relativistic_correction_acceleration )
109  {
110  // Only Schwarzschild correction is implemented so far.
111  if( accelerationModel->getCalculateDeSitterCorrection( ) || accelerationModel->getCalculateLenseThirringCorrection( ) )
112  {
113  throw std::runtime_error( "Error when creating relativistic acceleration correction partial, only Schwarzschild term implemented" );
114  }

116  acceleratedBodyState_ = accelerationModel->getStateFunctionOfAcceleratedBody( );
117

119  ppnBetaParameterFunction_ = accelerationModel->getPpnParameterBetaFunction_( );
120  centralBodyGravitationalParameterFunction_ = accelerationModel->getGravitationalParameterFunctionOfCentralBody( );
