24 void getOccultingBodiesInformation(
25  const NamedBodyMap& bodyMap, const std::vector< std::string >& occultingBodies,
26  std::vector< std::function< Eigen::Vector3d( ) > >& occultingBodyPositions,
27  std::vector< double >& occultingBodyRadii )
28 {
29  // Iterate over occulring bodies and retrieve radius and position function.
30  for( unsigned int i = 0; i < occultingBodies.size( ); i++ )
31  {
32  if( bodyMap.count( occultingBodies[ i ] ) == 0 )
33  {
34  throw std::runtime_error( "Error, could not find body " + occultingBodies[ i ] +
35  " in body map when making occulting body settings" );
36  }
37  else
38  {
39  std::shared_ptr< basic_astrodynamics::BodyShapeModel > shapeModel =
40  bodyMap.at( occultingBodies.at( i ) )->getShapeModel( );
41  if( shapeModel == nullptr )
42  {
43  throw std::runtime_error( "Error, no shape model for " + occultingBodies[ i ] +
44  " when making occulting body settings" );
45  }
46  occultingBodyPositions.push_back(
47  std::bind( &Body::getPosition, bodyMap.at( occultingBodies[ i ] ) ) );
48  occultingBodyRadii.push_back( shapeModel->getAverageRadius( ) );
49  }
50  }

54 std::shared_ptr< electro_magnetism::RadiationPressureInterface > createRadiationPressureInterface(
55  const std::shared_ptr< RadiationPressureInterfaceSettings > radiationPressureInterfaceSettings,
56  const std::string& bodyName, const NamedBodyMap& bodyMap )
57 {
58  std::shared_ptr< electro_magnetism::RadiationPressureInterface > radiationPressureInterface;
59 
60  // Check type of radiation pressure interface
61  switch( radiationPressureInterfaceSettings->getRadiationPressureType( ) )
62  {
63  case cannon_ball:
64  {
65  // Check type consistency.
66  std::shared_ptr< CannonBallRadiationPressureInterfaceSettings > cannonBallSettings =
67  std::dynamic_pointer_cast< CannonBallRadiationPressureInterfaceSettings >(
68  radiationPressureInterfaceSettings );
69  if( cannonBallSettings == nullptr )
70  {
71  throw std::runtime_error( "Error when making cannon ball radiation interface, type does not match object" );
72  }
73 
74  // Retrieve source body and check consistency.
75  if( bodyMap.count( radiationPressureInterfaceSettings->getSourceBody( ) ) == 0 )
76  {
77  throw std::runtime_error( "Error when making cannon ball radiation interface, source not found.");
78  }
79 
80  std::shared_ptr< Body > sourceBody =
81  bodyMap.at( radiationPressureInterfaceSettings->getSourceBody( ) );
82 
83  // Get reqruied data for occulting bodies.
84  std::vector< std::string > occultingBodies = cannonBallSettings->getOccultingBodies( );
85  std::vector< std::function< Eigen::Vector3d( ) > > occultingBodyPositions;
86  std::vector< double > occultingBodyRadii;
87  getOccultingBodiesInformation(
88  bodyMap, occultingBodies, occultingBodyPositions, occultingBodyRadii );
89 
90  // Retrive radius of source if occultations are used.
91  double sourceRadius;
92  if( occultingBodyPositions.size( ) > 0 )
93  {
94  std::shared_ptr< basic_astrodynamics::BodyShapeModel > sourceShapeModel =
95  sourceBody->getShapeModel( );
96 
97  if( sourceShapeModel == nullptr )
98  {
99  throw std::runtime_error( "Error when making occulted body, source body " +
100  radiationPressureInterfaceSettings->getSourceBody( ) +
101  " does not have a shape" );
102  }
103  else
104  {
105  sourceRadius = sourceShapeModel->getAverageRadius( );
106  }
107  }
108  else
109  {
110  sourceRadius = 0.0;
111  }
112 
113  // Create function returning radiated power.
114  std::function< double( ) > radiatedPowerFunction;
115  if( defaultRadiatedPowerValues.count(
116  radiationPressureInterfaceSettings->getSourceBody( ) ) == 0 )
117  {
118  throw std::runtime_error( "Error, no radiated power found for " +
119  radiationPressureInterfaceSettings->getSourceBody( ) );
120  }
121  else
122  {
123  radiatedPowerFunction = [ = ]( ){ return
124  defaultRadiatedPowerValues.at(
125  radiationPressureInterfaceSettings->getSourceBody( ) ); };
126  }
127 
128  // Create radiation pressure interface.
129  radiationPressureInterface =
130  std::make_shared< electro_magnetism::RadiationPressureInterface >(
131  radiatedPowerFunction,
132  std::bind( &Body::getPosition, sourceBody ),
133  std::bind( &Body::getPosition, bodyMap.at( bodyName ) ),
134  cannonBallSettings->getRadiationPressureCoefficient( ),
135  cannonBallSettings->getArea( ), occultingBodyPositions, occultingBodyRadii,
136  sourceRadius );
137  break;
138 
139  }
140  default:
141  throw std::runtime_error(
142  "Error, radiation pressure type" + std::to_string(
143  radiationPressureInterfaceSettings->getRadiationPressureType( ) ) +
144  "not recognized for body" + bodyName );
145  }
146 
147  return radiationPressureInterface;
148 }
