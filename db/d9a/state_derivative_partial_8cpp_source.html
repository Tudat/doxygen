57 std::pair< boost::function< void( Eigen::MatrixXd& ) >, int > createMergedParameterPartialFunction(
58  const std::pair< boost::function< void( Eigen::MatrixXd& ) >, int >& partialFunctionOfAccelerationToAdd,
59  const std::pair< boost::function< void( Eigen::MatrixXd& ) >, int >& partialFunctionOfAccelerationToSubtract )
60 {
61  std::pair< boost::function< void( Eigen::MatrixXd& ) >, int > parameterPartialFunction;
62 
63  // Check partial size and act accordingly.
64  if( ( partialFunctionOfAccelerationToAdd.second == 0 ) &&
65  ( partialFunctionOfAccelerationToSubtract.second == 0 ) )
66  {
67  parameterPartialFunction = std::make_pair( boost::function< void( Eigen::MatrixXd& ) >( ), 0 );
68  }
69  else if( partialFunctionOfAccelerationToSubtract.second == 0 )
70  {
71  parameterPartialFunction = partialFunctionOfAccelerationToAdd;
72  }
73  else if( partialFunctionOfAccelerationToAdd.second == 0 )
74  {
75  parameterPartialFunction = std::make_pair( boost::bind(
76  &evaluateNegativeParameterPartialFunction,
77  partialFunctionOfAccelerationToSubtract.first, _1 ),
78  partialFunctionOfAccelerationToSubtract.second );
79  }
80  // Partial size must be equal if both non-zero
81  else if( partialFunctionOfAccelerationToSubtract.second !=
82  partialFunctionOfAccelerationToAdd.second )
83  {
84  throw std::runtime_error( "Error when making merged parameter partial function, separate functions are incompatible" );
85  }
86  else
87  {
88  parameterPartialFunction = std::make_pair( boost::bind( &evaluateSubtractedParameterPartialFunction,
89  partialFunctionOfAccelerationToAdd.first,
90  partialFunctionOfAccelerationToSubtract.first, _1 ),
91  partialFunctionOfAccelerationToSubtract.second );
92  }
93  return parameterPartialFunction;
94 }
