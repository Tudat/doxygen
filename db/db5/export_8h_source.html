82 template< typename TimeType = double, typename StateScalarType = double >
83 void exportResultsOfDynamicsSimulator(
84  const std::shared_ptr< propagators::SingleArcDynamicsSimulator< StateScalarType, TimeType > >& singleArcDynamicsSimulator,
85  const std::vector< std::shared_ptr< ExportSettings > >& exportSettingsVector )
86 {
87  using namespace propagators;
88  using namespace input_output;
89 
90  std::map< TimeType, Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > > statesHistory =
91  singleArcDynamicsSimulator->getEquationsOfMotionNumericalSolution( );
92  std::map< TimeType, Eigen::VectorXd > dependentVariables =
93  singleArcDynamicsSimulator->getDependentVariableHistory( );
94  std::map< TimeType, double > cpuTimes =
95  singleArcDynamicsSimulator->getCumulativeComputationTimeHistory( );
96 
97  for ( std::shared_ptr< ExportSettings > exportSettings : exportSettingsVector )
98  {
99  std::vector< std::shared_ptr< VariableSettings > > variables;
100  std::vector< unsigned int > variableSizes;
101  std::vector< unsigned int > variableIndices;
102 
103  // Determine number of columns (not including first column = epoch).
104  unsigned int cols = 0;
105 
106  for ( std::shared_ptr< VariableSettings > variable : exportSettings->variables_ )
107  {
108  unsigned int variableSize = 0;
109  unsigned int variableIndex = 0;
110  switch ( variable->variableType_ )
111  {
112  case independentVariable:
113  case cpuTimeVariable:
114  {
115  variableSize = 1;
116  break;
117  }
118  case stateVariable:
119  {
120  variableSize = statesHistory.begin( )->second.rows( );
121  break;
122  }
123  case dependentVariable:
124  {
125  const std::shared_ptr< SingleDependentVariableSaveSettings > dependentVar =
126  std::dynamic_pointer_cast< SingleDependentVariableSaveSettings >( variable );
127  assertNonnullptrPointer( dependentVar );
128  try
129  {
130  const std::string variableID = getDependentVariableId( dependentVar );
131  try
132  {
133  variableIndex = getKeyWithValue(
134  singleArcDynamicsSimulator->getDependentVariableIds( ), variableID );
135  try
136  {
137  variableSize = getDependentVariableSaveSize( dependentVar );
138  }
139  catch ( ... )
140  {
141  std::cerr << "Could not export the results for variable \"" << variableID << "\" "
142  << "because its size is not known." << std::endl;
143  }
144  }
145  catch ( ... )
146  {
147  std::cerr << "Could not export the results for variable \"" << variableID << "\" "
148  << "because the main propagator was not configured to compute this variable."
149  << std::endl;
150  }
151  }
152  catch ( ... )
153  {
154  std::cerr << "Could not export results for dependent variable of type "
155  << dependentVar->dependentVariableType_
156  << " because its ID is not known." << std::endl;
157  }
158  break;
159  }
160  default:
161  {
162  std::cerr << "Could not export results for variable of unsupported type "
163  << variable->variableType_ << "." << std::endl;
164  break;
165  }
166  }
167 
168  if ( variableSize > 0 )
169  {
170  variables.push_back( variable );
171  variableSizes.push_back( variableSize );
172  variableIndices.push_back( variableIndex );
173 
174  cols += variableSize;
175  }
176  }
177 
178  // Concatenate requested results
179  std::map< TimeType, Eigen::VectorXd > results;
180  for ( auto it = statesHistory.begin( ); it != statesHistory.end( ); ++it )
181  {
182  if ( ( it == statesHistory.begin( ) && ! exportSettings->onlyInitialStep_ &&
183  exportSettings->onlyFinalStep_ ) ||
184  ( it == --statesHistory.end( ) && ! exportSettings->onlyFinalStep_ &&
185  exportSettings->onlyInitialStep_ )
186  || ( ( it != statesHistory.begin( ) && it != --statesHistory.end( ) ) &&
187  ( exportSettings->onlyInitialStep_ || exportSettings->onlyFinalStep_ ) ) )
188  {
189  continue;
190  }
191  unsigned int currentIndex = 0;
192 
193  const TimeType epoch = it->first;
194  Eigen::VectorXd result = Eigen::VectorXd::Zero( cols );
195  for ( unsigned int i = 0; i < variables.size( ); ++i )
196  {
197  const std::shared_ptr< VariableSettings > variable = variables.at( i );
198  const unsigned int variableSize = variableSizes.at( i );
199 
200  switch ( variable->variableType_ )
201  {
202  case independentVariable:
203  {
204  result.segment( currentIndex, variableSize ) =
205  ( Eigen::VectorXd( 1 ) << static_cast< double >( epoch ) ).finished( );
206  break;
207  }
208  case cpuTimeVariable:
209  {
210  result.segment( currentIndex, variableSize ) =
211  ( Eigen::VectorXd( 1 ) << cpuTimes.at( epoch ) ).finished( );
212  break;
213  }
214  case stateVariable:
215  {
216  result.segment( currentIndex, variableSize ) = it->second.template cast< double >( );
217  break;
218  }
219  case dependentVariable:
220  {
221  result.segment( currentIndex, variableSize ) =
222  dependentVariables.at( epoch ).segment( variableIndices.at( i ), variableSize );
223  break;
224  }
225  default:
226  break;
227  }
228  currentIndex += variableSize;
229  }
230  results[ epoch ] = result;
231  }
232 
233  if ( exportSettings->epochsInFirstColumn_ )
234  {
235  // Write results map to file.
236  writeDataMapToTextFile( results,
237  exportSettings->outputFile_,
238  exportSettings->header_,
239  exportSettings->numericalPrecision_ );
240  }
241  else
242  {
243  // Write results matrix to file.
244  Eigen::MatrixXd resultsMatrix( results.size( ), cols );
245  int currentRow = 0;
246  for ( auto entry : results )
247  {
248  resultsMatrix.row( currentRow++ ) = entry.second.transpose( );
249  }
250  writeMatrixToFile( resultsMatrix,
251  exportSettings->outputFile_.filename( ).string( ),
252  exportSettings->numericalPrecision_,
253  exportSettings->outputFile_.parent_path( ),
254  " ", exportSettings->header_ );
255  }
256  }
257 }
