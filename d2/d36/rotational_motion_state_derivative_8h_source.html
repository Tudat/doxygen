209  void calculateSystemStateDerivative(
210  const TimeType time,
211  const Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 >& stateOfSystemToBeIntegrated,
212  Eigen::Block< Eigen::Matrix< StateScalarType, Eigen::Dynamic, Eigen::Dynamic > > stateDerivative )
213  {
214  stateDerivative = Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 >::Zero( stateOfSystemToBeIntegrated.rows( ), 1 );
215  std::vector< Eigen::Vector3d > torquesActingOnBodies = sumTorquesPerBody( );
216 
217  for( unsigned int i = 0; i < torquesActingOnBodies.size( ); i++ )
218  {
219  Eigen::Matrix< StateScalarType, 4, 1 > currentQuaternion = ( stateOfSystemToBeIntegrated.block( 7 * i, 0, 4, 1 ) ).normalized( );
220  Eigen::Matrix< StateScalarType, 3, 1 > currentBodyFixedRotationRate = stateOfSystemToBeIntegrated.block( 7 * i + 4, 0, 3, 1 );
221 
222  stateDerivative.block( 7 * i, 0, 4, 1 ) = calculateQuaternionDerivative(
223  currentQuaternion.template cast< double >( ), currentBodyFixedRotationRate.template cast< double >( ) ).
224  template cast< StateScalarType >( );
225  stateDerivative.block( 7 * i + 4, 0, 3, 1 ) = evaluateRotationalEquationsOfMotion(
226  bodyInertiaTensorFunctions_.at( i )( ), torquesActingOnBodies.at( i ),
227  currentBodyFixedRotationRate.template cast< double >( ),
228  bodyInertiaTensorTimeDerivativeFunctions_.at( i )( ) ).template cast< StateScalarType >( );

283  Eigen::Vector3d getTotalTorqueForBody(
284  const std::string& bodyName )
285  {
286  // Check if body is propagated.
287  Eigen::Vector3d totalTorque = Eigen::Vector3d::Zero( );
288  if( std::find( bodiesToPropagate_.begin( ),
289  bodiesToPropagate_.end( ),
290  bodyName ) == bodiesToPropagate_.end( ) )
291  {
292  std::string errorMessage = "Error when getting total torque for body " + bodyName +
293  ", no such torque is found";
294  throw std::runtime_error( errorMessage );
295  }
296  else
297  {
298  if( torqueModelsPerBody_.count( bodyName ) != 0 )
299  {
300  basic_astrodynamics::SingleBodyTorqueModelMap torquesOnBody =
301  torqueModelsPerBody_.at( bodyName );
302 
303  // Iterate over all torques acting on body
304  for( innerTorqueIterator = torquesOnBody.begin( );
305  innerTorqueIterator != torquesOnBody.end( );
306  innerTorqueIterator++ )
307  {
308  for( unsigned int j = 0; j < innerTorqueIterator->second.size( ); j++ )
309  {
310  // Calculate torque and add to state derivative.
311  totalTorque += innerTorqueIterator->second[ j ]->getTorque( );
312  }
313  }
314  }
315  }
316  return totalTorque;
317  }
