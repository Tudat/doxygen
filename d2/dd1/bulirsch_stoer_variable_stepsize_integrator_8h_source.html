225  virtual StateType performIntegrationStep( const TimeStepType stepSize )
226  {
227  StateType stateAtFirstPoint_;
228  StateType stateAtCenterPoint_;
229  StateType stateAtLastPoint_;
230 
231  bool stepSuccessful = 0;
232 
233  // Compute sub steps to take.
234  for ( unsigned int p = 0; p < subSteps_.size( ); p++ )
235  {
236  subSteps_.at( p ) = stepSize / static_cast< double >(
237  sequence_.at( p ) );
238  }
239 
240  double errorScaleTerm = TUDAT_NAN;
241  for ( unsigned int i = 0; i <= maximumStepIndex_; i++ )
242  {
243  // Compute Euler step and set as state at center point for use with mid-point method.
244  stateAtCenterPoint_ = currentState_ + subSteps_.at( i )
245  * this->stateDerivativeFunction_( currentIndependentVariable_, currentState_ );
246 
247  // Apply modified mid-point rule.
248  stateAtFirstPoint_ = currentState_;
249  IndependentVariableType independentVariableAtFirstPoint_ = currentIndependentVariable_;
250  for ( unsigned int j = 0; j < sequence_.at( i ) - 1; j++ )
251  {
252  stateAtLastPoint_ = executeMidPointMethod(
253  stateAtFirstPoint_, stateAtCenterPoint_,
254  independentVariableAtFirstPoint_, subSteps_.at( i ) );
255 
256  if ( j < sequence_.at( i ) - 2 )
257  {
258  stateAtFirstPoint_ = stateAtCenterPoint_;
259  stateAtCenterPoint_ = stateAtLastPoint_;
260  independentVariableAtFirstPoint_ += subSteps_.at( i );
261  }
262  }
263 
264  // Apply end-point correction.
265  integratedStates_[ i ][ 0 ]
266  = 0.5 * ( stateAtLastPoint_ + stateAtCenterPoint_+ subSteps_.at( i ) * this->stateDerivativeFunction_(
267  currentIndependentVariable_ + stepSize, stateAtLastPoint_ ) );
268 
269  for ( unsigned int k = 1; k < i + 1; k++ )
270  {
271  integratedStates_[ i ][ k ] =
272  integratedStates_[ i ][ k - 1 ] + 1.0 /
273  ( pow( subSteps_.at( i - k ), 2.0 ) / std::pow( subSteps_.at( i ), 2.0 ) - 1.0 )
274  * ( integratedStates_[ i ][ k - 1 ] - integratedStates_[ i - 1 ][ k - 1 ] );
275  }
276 
277  if( i == maximumStepIndex_ )
278  {
279  const StateType errorTolerance_ =
280  ( integratedStates_.at( i ).at( i ).array( ).abs( ) * relativeErrorTolerance_.array( ) ).matrix( )
281  + absoluteErrorTolerance_;
282 
283  double maximumAllowableErrorValue = errorTolerance_.array( ).maxCoeff( );
284  double maximumErrorValue = ( integratedStates_.at( i ).at( i ) - integratedStates_.at( i ).at( i - 1 ) ).array( ).abs( ).maxCoeff( );
285 
286  errorScaleTerm = safetyFactorForNextStepSize_ * std::pow( maximumAllowableErrorValue / maximumErrorValue,
287  ( 1.0 / static_cast< double >( 2 * i - 1 ) ) );
288 
289  if( maximumErrorValue < maximumAllowableErrorValue )
290  {
291  // Accept the current step.
292  lastIndependentVariable_ = currentIndependentVariable_;
293  lastState_ = currentState_;
294  currentIndependentVariable_ += stepSize;
295  currentState_ = integratedStates_[ i ][ i ];
296  stepSize_ = stepSize;
297  stepSuccessful = true;
298  }
299  else
300  {
301  stepSuccessful = false;
302  }
303  }
304  }
305  if( !stepSuccessful )
306  {
307  if( safetyFactorForNextStepSize_ * errorScaleTerm < minimumFactorDecreaseForNextStepSize_ )
308  {
309  this->stepSize_ = stepSize * minimumFactorDecreaseForNextStepSize_ * errorScaleTerm;
310  }
311  else
312  {
313  this->stepSize_ = stepSize * safetyFactorForNextStepSize_ * errorScaleTerm;
314  }
315 
316  if( std::fabs( stepSize_ ) < std::fabs( minimumStepSize_ ) )
317  {
318  isMinimumStepSizeViolated_ = true;
319  throw std::runtime_error( "Error in BS integrator, minimum step size exceeded" );
320  }
321  performIntegrationStep( stepSize_ );
322  }
323  else
324  {
325  if( errorScaleTerm > maximumFactorIncreaseForNextStepSize_ )
326  {
327  this->stepSize_ = stepSize * maximumFactorIncreaseForNextStepSize_;
328  if( std::fabs( this->stepSize_ ) >= std::fabs( maximumStepSize_ ) )
329  {
330  this->stepSize_ = stepSize / ( std::fabs( stepSize ) ) * maximumStepSize_ ;
331  }
332  }
333  else
334  {
335  this->stepSize_ = stepSize * errorScaleTerm;
336  }
337  }
338 
339 
340  return currentState_;
341  }
