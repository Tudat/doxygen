211 template< typename ObservationScalarType = double, typename TimeType = double >
212 std::map< ObservableType, std::map< LinkEnds, std::pair< Eigen::Matrix< ObservationScalarType, Eigen::Dynamic, 1 >,
213 std::pair< std::vector< TimeType >, LinkEndType > > > >
214 simulateObservations(
215  const std::map< ObservableType, std::map< LinkEnds,
216  std::shared_ptr< ObservationSimulationTimeSettings< TimeType > > > >& observationsToSimulate,
217  const std::map< ObservableType,
218  std::shared_ptr< ObservationSimulatorBase< ObservationScalarType, TimeType > > >& observationSimulators,
219  const PerObservableObservationViabilityCalculatorList viabilityCalculatorList =
220  PerObservableObservationViabilityCalculatorList( ) )
221 {
222  // Declare return map.
223  std::map< ObservableType, std::map< LinkEnds, std::pair< Eigen::Matrix< ObservationScalarType, Eigen::Dynamic, 1 >,
224  std::pair< std::vector< TimeType >, LinkEndType > > > > observations;
225 
226  // Iterate over all observables.
227  for( typename std::map< ObservableType, std::map< LinkEnds,
228  std::shared_ptr< ObservationSimulationTimeSettings< TimeType > > > >::const_iterator observationIterator =
229  observationsToSimulate.begin( ); observationIterator != observationsToSimulate.end( ); observationIterator++ )
230  {
231 
232  PerLinkEndsObservationViabilityCalculatorList perLinkViabilityCalculators;
233  if( viabilityCalculatorList.count( observationIterator->first ) > 0 )
234  {
235  perLinkViabilityCalculators = viabilityCalculatorList.at( observationIterator->first );
236  }
237  // Iterate over all link ends for current observable.
238  for( typename std::map< LinkEnds,
239  std::shared_ptr< ObservationSimulationTimeSettings< TimeType > > >::const_iterator linkEndIterator =
240  observationIterator->second.begin( ); linkEndIterator != observationIterator->second.end( ); linkEndIterator++ )
241  {
242 
243  std::vector< std::shared_ptr< ObservationViabilityCalculator > > currentObservationViabilityCalculators;
244  if( perLinkViabilityCalculators.count( linkEndIterator->first ) > 0 )
245  {
246  currentObservationViabilityCalculators = perLinkViabilityCalculators.at( linkEndIterator->first );
247  }
248 
249  int observationSize = observationSimulators.at( observationIterator->first )->getObservationSize(
250  linkEndIterator->first );
251 
252  switch( observationSize )
253  {
254  case 1:
255  {
256  std::shared_ptr< ObservationSimulator< 1, ObservationScalarType, TimeType > > derivedObservationSimulator =
257  std::dynamic_pointer_cast< ObservationSimulator< 1, ObservationScalarType, TimeType > >(
258  observationSimulators.at( observationIterator->first ) );
259 
260  if( derivedObservationSimulator == nullptr )
261  {
262  throw std::runtime_error( "Error when simulating observation: dynamic case to size 1 is nullptr" );
263  }
264 
265  // Simulate observations for current observable and link ends set.
266  observations[ observationIterator->first ][ linkEndIterator->first ] =
267  simulateSingleObservationSet< ObservationScalarType, TimeType, 1 >(
268  linkEndIterator->second, derivedObservationSimulator,
269  linkEndIterator->first, currentObservationViabilityCalculators );
270  break;
271  }
272  case 2:
273  {
274  std::shared_ptr< ObservationSimulator< 2, ObservationScalarType, TimeType > > derivedObservationSimulator =
275  std::dynamic_pointer_cast< ObservationSimulator< 2, ObservationScalarType, TimeType > >(
276  observationSimulators.at( observationIterator->first ) );
277 
278  if( derivedObservationSimulator == nullptr )
279  {
280  throw std::runtime_error( "Error when simulating observation: dynamic case to size 2 is nullptr" );
281  }
282 
283  // Simulate observations for current observable and link ends set.
284  observations[ observationIterator->first ][ linkEndIterator->first ] =
285  simulateSingleObservationSet< ObservationScalarType, TimeType, 2 >(
286  linkEndIterator->second, derivedObservationSimulator,
287  linkEndIterator->first, currentObservationViabilityCalculators );
288  break;
289  }
290  case 3:
291  {
292  std::shared_ptr< ObservationSimulator< 3, ObservationScalarType, TimeType > > derivedObservationSimulator =
293  std::dynamic_pointer_cast< ObservationSimulator< 3, ObservationScalarType, TimeType > >(
294  observationSimulators.at( observationIterator->first ) );
295 
296  if( derivedObservationSimulator == nullptr )
297  {
298  throw std::runtime_error( "Error when simulating observation: dynamic case to size 3 is nullptr" );
299  }
300 
301  // Simulate observations for current observable and link ends set.
302  observations[ observationIterator->first ][ linkEndIterator->first ] =
303  simulateSingleObservationSet< ObservationScalarType, TimeType, 3 >(
304  linkEndIterator->second, derivedObservationSimulator,
305  linkEndIterator->first, currentObservationViabilityCalculators );
306  break;
307  }
308  default:
309  throw std::runtime_error( "Error, simulation of observations not yet implemented for size " +
310  std::to_string( observationSize ) );
311 
312  }
313  }
314  }
315  return observations;
316 }

331 template< typename ObservationScalarType = double, typename TimeType = double >
332 std::map< ObservableType, std::map< LinkEnds, std::pair< Eigen::Matrix< ObservationScalarType, Eigen::Dynamic, 1 >,
333 std::pair< std::vector< TimeType >, LinkEndType > > > >
334 simulateObservationsWithNoise(
335  const std::map< ObservableType, std::map< LinkEnds,
336  std::shared_ptr< ObservationSimulationTimeSettings< TimeType > > > >& observationsToSimulate,
337  const std::map< ObservableType,
338  std::shared_ptr< ObservationSimulatorBase< ObservationScalarType, TimeType > > >& observationSimulators,
339  const std::map< ObservableType, std::map< LinkEnds, std::function< Eigen::VectorXd( const double ) > > >& noiseFunctions,
340  const PerObservableObservationViabilityCalculatorList viabilityCalculatorList =
341  PerObservableObservationViabilityCalculatorList( ) )
342 {
343  typedef std::map< LinkEnds, std::pair< Eigen::Matrix< ObservationScalarType, Eigen::Dynamic, 1 >,
344  std::pair< std::vector< TimeType >, LinkEndType > > > SingelTypeObservationsMap;
345  typedef std::map< ObservableType, std::map< LinkEnds, std::pair< Eigen::Matrix< ObservationScalarType, Eigen::Dynamic, 1 >,
346  std::pair< std::vector< TimeType >, LinkEndType > > > > ObservationsMap;
347 
348  // Simulate noise-free observations
349  ObservationsMap noiseFreeObservationsList = simulateObservations(
350  observationsToSimulate, observationSimulators, viabilityCalculatorList );
351 
352  // Declare return map with noisy observations.
353  ObservationsMap noisyObservationsList;
354 
355  // Iterate over all observable types
356  for( typename ObservationsMap::iterator observationIterator = noiseFreeObservationsList.begin( );
357  observationIterator != noiseFreeObservationsList.end( ); observationIterator++ )
358  {
359  int currentObservableSize = getObservableSize( observationIterator->first );
360 
361  // Iterate over all link ends of current observable
362  SingelTypeObservationsMap singleObservableObservationsWithNoise;
363  for( typename SingelTypeObservationsMap::iterator linkIterator = observationIterator->second.begin( );
364  linkIterator != observationIterator->second.end( ); linkIterator++ )
365  {
366  // Retrieve noise-free observations/times
367  std::pair< Eigen::Matrix< ObservationScalarType, Eigen::Dynamic, 1 >,
368  std::pair< std::vector< TimeType >, LinkEndType > > nominalObservationsAndTimes =
369  linkIterator->second;
370  Eigen::Matrix< ObservationScalarType, Eigen::Dynamic, 1 > noisyObservations = nominalObservationsAndTimes.first;
371  std::vector< TimeType > nominalTimes = nominalObservationsAndTimes.second.first;
372 
373  // Check data consistency
374  if( static_cast< int >( nominalTimes.size( ) ) != noisyObservations.rows( ) / currentObservableSize )
375  {
376  throw std::runtime_error( "Error when adding noise to observations, input data is inconsistent" );
377  }
378 
379  // Retrieve noise function
380  std::function< Eigen::VectorXd( const double ) > currentNoiseFunction =
381  noiseFunctions.at( observationIterator->first ).at( linkIterator->first );
382 
383  // Simulate noise for all observations and add to calculated values
384  for( unsigned int i = 0; i < nominalTimes.size( ); i++ )
385  {
386  // Check noise function consistency
387  if( i == 0 )
388  {
389  if( currentNoiseFunction( nominalTimes.at( i ) ).rows( ) != currentObservableSize )
390  {
391  throw std::runtime_error( "Error when adding noise to observations, noise size is inconsistent" );
392  }
393  }
394 
395  // Add noise to observation
396  noisyObservations.segment( i * currentObservableSize, currentObservableSize ) +=
397  currentNoiseFunction( nominalTimes.at( i ) );
398  }
399 
400  singleObservableObservationsWithNoise[ linkIterator->first ] =
401  std::make_pair( noisyObservations, nominalObservationsAndTimes.second );
402  }
403  noisyObservationsList[ observationIterator->first ] = singleObservableObservationsWithNoise;
404  }
405  return noisyObservationsList;
406 }
