157 Eigen::Matrix3d getVelocityBasedLvlhToInertialRotation(
158  const Eigen::Vector6d& vehicleState,
159  const Eigen::Vector6d& centralBodyState,
160  const bool doesNaxisPointAwayFromCentralBody )
161 {
162  Eigen::Vector3d vehicleVelocity, vehicleRadius;
163  vehicleRadius = vehicleState.segment( 0, 3 ) - centralBodyState.segment( 0, 3 );
164  vehicleVelocity = vehicleState.segment( 3, 3 ) - centralBodyState.segment( 3, 3 );
165 
166  Eigen::Vector3d unitT = vehicleVelocity.normalized( );// / vehicleVelocity.norm( );
167  if ( vehicleRadius.cross( vehicleVelocity ).norm( ) == 0.0 )
168  {
169  std::string errorMessage = "Division by zero: radius and velocity are in the same direction.";
170  throw std::runtime_error( errorMessage );
171  }
172 
173  Eigen::Vector3d unitW = ( ( ( doesNaxisPointAwayFromCentralBody == true ) ? -1.0 : 1.0 ) *
174  ( vehicleRadius.cross( vehicleVelocity ) ).normalized( ) );
175 
176  Eigen::Vector3d unitN = ( unitW.cross( unitT ) ).normalized( );
177 
178  Eigen::Matrix3d transformationMatrix;
179  transformationMatrix << unitT( 0 ), unitN( 0 ), unitW( 0 ),
180  unitT( 1 ), unitN( 1 ), unitW( 1 ),
181  unitT( 2 ), unitN( 2 ), unitW( 2 );
182 
183  return transformationMatrix;
184 }

197 Eigen::Quaterniond getVelocityBasedLvlhToPlanetocentricRotationKeplerian(
198  const Eigen::Matrix< double, 6, 1 > spacecraftKeplerianState )
199 {
200 
201  double eccentricity = spacecraftKeplerianState( 1 );
202  double inclination = spacecraftKeplerianState( 2 );
203  double argumentOfPeriapsis = spacecraftKeplerianState( 3 );
204  double rightAscensionOfAscendingNode = spacecraftKeplerianState( 4 );
205  double trueAnomaly = spacecraftKeplerianState( 5 );
206 
207  double flightPathAngle = std::atan( ( eccentricity * std::sin( trueAnomaly ) ) /
208  ( 1.0 + eccentricity * std::cos( trueAnomaly ) ) );
209 
210  // Compute first rotation around Z axis.
211  Eigen::AngleAxisd firstRotationAroundZaxis(
212  -( -mathematical_constants::PI * 0.5 + flightPathAngle - ( trueAnomaly + argumentOfPeriapsis ) ),
213  Eigen::Vector3d::UnitZ( ) );
214 
215  // Compute rotation around X axis.
216  Eigen::AngleAxisd rotationAroundXaxis( inclination, Eigen::Vector3d::UnitX( ) );
217 
218  // Compute second rotation around Z axis.
219  Eigen::AngleAxisd secondRotationAroundZaxis( rightAscensionOfAscendingNode, Eigen::Vector3d::UnitZ( ) );
220 
221  Eigen::Quaterniond frameTransformationQuaternion = Eigen::Quaterniond(
222  ( secondRotationAroundZaxis * rotationAroundXaxis * firstRotationAroundZaxis) );
223 
224  // Return transformation quaternion.
225  return frameTransformationQuaternion;
226 }
