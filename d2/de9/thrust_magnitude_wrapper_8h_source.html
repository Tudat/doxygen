131  void update( const double time )
132  {
133  if( !( currentTime_ == time ) )
134  {
135  // If engine is one, update engine.
136  if( isEngineOnFunction_( time ) )
137  {
138  currentThrustMagnitude_ = thrustMagnitudeFunction_( time );
139  currentSpecificImpulse_ = specificImpulseFunction_( time );
140  }
141  // If engine is off, set thrust to zero.
142  else

144  currentThrustMagnitude_ = 0.0;

146  }

267  {
268  currentThrust_ += engineModels_.at( i )->getCurrentThrust( );
269  currentMassRate_ += engineModels_.at( i )->getCurrentMassRate( );
270 
271  }

358  const std::vector< std::function< double( ) > > thrustInputVariableFunctions,
359  const std::vector< std::function< double( ) > > specificImpulseInputVariableFunctions,
360  const std::vector< propulsion::ThrustIndependentVariables > thrustIndependentVariables,
361  const std::vector< propulsion::ThrustIndependentVariables > specificImpulseDependentVariables,
362  const std::function< void( const double) > inputUpdateFunction =
363  std::function< void( const double) >( ) ):
364  thrustMagnitudeFunction_( thrustMagnitudeFunction ),
365  specificImpulseFunction_( specificImpulseFunction ),
366  thrustInputVariableFunctions_( thrustInputVariableFunctions ),
367  specificImpulseInputVariableFunctions_( specificImpulseInputVariableFunctions ),

370  inputUpdateFunction_( inputUpdateFunction ),
371  currentThrustMagnitude_( TUDAT_NAN ),
372  currentSpecificImpulse_( TUDAT_NAN )
373  {
374  if( thrustInputVariableFunctions_.size( ) != thrustIndependentVariables_.size( ) )
375  {
376  throw std::runtime_error( "Error in parameterized thrust, inconsistent number of user-defined input variables for thrust" );
377  }
378 
379  if( specificImpulseInputVariableFunctions_.size( ) != specificImpulseDependentVariables_.size( ) )
380  {
381  throw std::runtime_error( "Error in parameterized thrust, inconsistent number of user-defined input variables for Isp" );
382  }
383 
384  currentThrustInputVariables_.resize( thrustInputVariableFunctions.size( ) );
385  currentSpecificImpulseInputVariables_.resize( specificImpulseInputVariableFunctions.size( ) );
386  }

396  void update( const double time )
397  {
398  if( !( currentTime_ == time ) )
399  {
400  if( !( inputUpdateFunction_ == nullptr ) )
401  {
402  inputUpdateFunction_( time );
403  }
404  // Retrieve thrust independent variables
405  for( unsigned int i = 0; i < thrustInputVariableFunctions_.size( ); i++ )
406  {
407  currentThrustInputVariables_[ i ] = thrustInputVariableFunctions_.at( i )( );
408  }
409 
410  // Compute thrust
411  currentThrustMagnitude_ = thrustMagnitudeFunction_( currentThrustInputVariables_ );
412 
413  // Retrieve specific impulse independent variables
414  for( unsigned int i = 0; i < specificImpulseInputVariableFunctions_.size( ); i++ )
415  {
416  currentSpecificImpulseInputVariables_[ i ] = specificImpulseInputVariableFunctions_.at( i )( );
417  }
418 
419  // Compute specific impulse
420  currentSpecificImpulse_ = specificImpulseFunction_( currentSpecificImpulseInputVariables_ );
421  }
422  }
