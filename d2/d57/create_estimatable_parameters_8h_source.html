36 template< typename InitialStateParameterType = double >
37 std::shared_ptr< estimatable_parameters::EstimatableParameter< Eigen::Matrix
38 < InitialStateParameterType, Eigen::Dynamic, 1 > > > createInitialDynamicalStateParameterToEstimate(
39  const NamedBodyMap& bodyMap,
40  const std::shared_ptr< estimatable_parameters::EstimatableParameterSettings >& parameterSettings )
41 {
42  using namespace tudat::estimatable_parameters;
43 
44  std::shared_ptr< EstimatableParameter< Eigen::Matrix< InitialStateParameterType, Eigen::Dynamic, 1 > > >
45  initialStateParameterToEstimate;
46 
47  // Check consistency of input.
48  if( !isParameterDynamicalPropertyInitialState( parameterSettings->parameterType_.first ) )
49  {
50  std::string errorMessage = "Error when requesting to make initial state parameter " +
51  std::to_string( parameterSettings->parameterType_.first ) + " of " +
52  parameterSettings->parameterType_.second.first +
53  ", parameter is not an initial state parameter ";
54  throw std::runtime_error( errorMessage );
55  }
56  else
57  {
58  // Identify state that is to be estimation
59  switch( parameterSettings->parameterType_.first )
60  {
61  case initial_body_state:
62 
63  // Check consistency of input.
64  if( std::dynamic_pointer_cast<
65  InitialTranslationalStateEstimatableParameterSettings< InitialStateParameterType > >(
66  parameterSettings ) == nullptr )
67  {
68  throw std::runtime_error( "Error when making body initial state parameter, settings type is incompatible" );
69  }
70  else
71  {
72  std::shared_ptr< InitialTranslationalStateEstimatableParameterSettings< InitialStateParameterType > >
73  initialStateSettings =
74  std::dynamic_pointer_cast<
75  InitialTranslationalStateEstimatableParameterSettings< InitialStateParameterType > >(
76  parameterSettings );
77 
78  Eigen::Matrix< InitialStateParameterType, Eigen::Dynamic, 1 > initialTranslationalState;
79 
80  // If initial time is not defined, use preset initial state
81  if( ! ( initialStateSettings->initialTime_ == initialStateSettings->initialTime_ ) )
82  {
83  initialTranslationalState = initialStateSettings->initialStateValue_;
84 
85 
86  }
87  // Compute initial state from environment
88  else
89  {
90  initialTranslationalState = propagators::getInitialStateOfBody
91  < double, InitialStateParameterType >(
92  initialStateSettings->parameterType_.second.first, initialStateSettings->centralBody_,
93  bodyMap, initialStateSettings->initialTime_ );
94 
95  }
96 
97  // Create translational state estimation interface object
98  initialStateParameterToEstimate =
99  std::make_shared< InitialTranslationalStateParameter< InitialStateParameterType > >(
100  initialStateSettings->parameterType_.second.first, initialTranslationalState,
101  initialStateSettings->centralBody_,
102  initialStateSettings->frameOrientation_ );
103  }
104  break;
105  case arc_wise_initial_body_state:
106  if( std::dynamic_pointer_cast< ArcWiseInitialTranslationalStateEstimatableParameterSettings<
107  InitialStateParameterType > >( parameterSettings ) == nullptr )
108  {
109  throw std::runtime_error(
110  "Error when making body initial state parameter, settings type is incompatible" );
111  }
112  else
113  {
114  std::shared_ptr< ArcWiseInitialTranslationalStateEstimatableParameterSettings< InitialStateParameterType > >
115  initialStateSettings = std::dynamic_pointer_cast<
116  ArcWiseInitialTranslationalStateEstimatableParameterSettings< InitialStateParameterType > >(
117  parameterSettings );
118 
119  if( initialStateSettings->isStateSet_ )
120  {
121  initialStateParameterToEstimate = std::make_shared< ArcWiseInitialTranslationalStateParameter<
122  InitialStateParameterType > >(
123  initialStateSettings->parameterType_.second.first,
124  initialStateSettings->arcStartTimes_,
125  initialStateSettings->initialStateValue_,
126  initialStateSettings->centralBody_,
127  initialStateSettings->frameOrientation_ );
128  }
129  else
130  {
131  initialStateParameterToEstimate = std::make_shared< ArcWiseInitialTranslationalStateParameter<
132  InitialStateParameterType > >(
133  initialStateSettings->parameterType_.second.first, initialStateSettings->arcStartTimes_,
134  propagators::getInitialArcWiseStateOfBody< double, InitialStateParameterType >(
135  initialStateSettings->parameterType_.second.first,
136  initialStateSettings->centralBody_, bodyMap,
137  initialStateSettings->arcStartTimes_ ),
138  initialStateSettings->centralBody_, initialStateSettings->frameOrientation_ );
139  }
140  }
141  break;
142  default:
143  std::string errorMessage = "Error, could not create parameter for initial state of type " +
144  std::to_string( parameterSettings->parameterType_.first );
145  throw std::runtime_error( errorMessage );
146  }
147  }
148 
149  return initialStateParameterToEstimate;
150 }

191 template< typename InitialStateParameterType = double >
192 std::shared_ptr< estimatable_parameters::EstimatableParameterSet< InitialStateParameterType > > createParametersToEstimate(
193  const std::vector< std::shared_ptr< estimatable_parameters::EstimatableParameterSettings > >& parameterNames,
194  const NamedBodyMap& bodyMap,
195  const basic_astrodynamics::AccelerationMap& accelerationModelMap = basic_astrodynamics::AccelerationMap( ) )
196 
197 {
198  using namespace tudat::estimatable_parameters;
199 
200  std::vector< std::shared_ptr< EstimatableParameter< Eigen::Matrix< InitialStateParameterType, Eigen::Dynamic, 1 > > > >
201  initialDynamicalParametersToEstimate;
202  std::vector< std::shared_ptr< EstimatableParameter< double > > > doubleParametersToEstimate;
203  std::vector< std::shared_ptr< EstimatableParameter< Eigen::VectorXd > > > vectorParametersToEstimate;
204 
205  // Iterate over all parameters.
206  for( unsigned int i = 0; i < parameterNames.size( ); i++ )
207  {
208  // Create initial dynamical parameters.
209  if( isParameterDynamicalPropertyInitialState( parameterNames.at( i )->parameterType_.first ) )
210  {
211  initialDynamicalParametersToEstimate.push_back(
212  createInitialDynamicalStateParameterToEstimate< InitialStateParameterType >(
213  bodyMap, parameterNames.at( i ) ) );
214  }
215  // Create parameters defined by single double value
216  else if( isDoubleParameter( parameterNames[ i ]->parameterType_.first ) == true )
217  {
218  doubleParametersToEstimate.push_back( createDoubleParameterToEstimate(
219  parameterNames[ i ], bodyMap, accelerationModelMap ) );
220  }
221  // Create parameters defined by list of double values
222  else if( isDoubleParameter( parameterNames[ i ]->parameterType_.first ) == false )
223  {
224  vectorParametersToEstimate.push_back( createVectorParameterToEstimate(
225  parameterNames[ i ], bodyMap, accelerationModelMap ) );
226  }
227  else
228  {
229  std::string errorMessage = "Error, parameter type of " +
230  std::string( parameterNames[ i ]->parameterType_.second.first ) + "of " +
231  std::to_string( parameterNames[ i ]->parameterType_.first ) +
232  "not recognized when making estimatable parameter set.";
233 
234  throw std::runtime_error( errorMessage );
235  }
236  }
237 
238  return std::make_shared< EstimatableParameterSet< InitialStateParameterType > >(
239  doubleParametersToEstimate, vectorParametersToEstimate, initialDynamicalParametersToEstimate );
240 }
