26 Eigen::Matrix3d calculateAccelerationWrtStatePartials(
27  std::function< void( Eigen::Vector6d ) > setBodyState,
28  std::shared_ptr< basic_astrodynamics::AccelerationModel< Eigen::Vector3d > > accelerationModel,
29  Eigen::Vector6d originalState,
30  Eigen::Vector3d statePerturbation,
31  int startIndex,
32  std::function< void( ) > updateFunction,
33  const double evaluationTime )
34 {
35  Eigen::Matrix3d upAccelerations = Eigen::Matrix3d::Zero( );
36  Eigen::Matrix3d downAccelerations = Eigen::Matrix3d::Zero( );
37 
38  Eigen::Vector6d perturbedState = originalState;
39 
40  accelerationModel->resetTime( TUDAT_NAN );
41 
42  // Calculate perturbed accelerations for up-perturbed state entries.
43  for( int i = 0; i < 3; i++ )
44  {
45  perturbedState( i + startIndex ) += statePerturbation( i );
46  setBodyState( perturbedState );
47  updateFunction( );
48  upAccelerations.block( 0, i, 3, 1 ) = basic_astrodynamics::updateAndGetAcceleration< Eigen::Vector3d >(
49  accelerationModel, evaluationTime );
50  accelerationModel->resetTime( TUDAT_NAN );
51  perturbedState = originalState;
52  }
53 
54  // Calculate perturbed accelerations for down-perturbed state entries.
55  for( int i = 0; i < 3; i++ )
56  {
57  perturbedState( i + startIndex ) -= statePerturbation( i );
58  setBodyState( perturbedState );
59  updateFunction( );
60  downAccelerations.block( 0, i, 3, 1 ) = basic_astrodynamics::updateAndGetAcceleration< Eigen::Vector3d >(
61  accelerationModel, evaluationTime );
62  accelerationModel->resetTime( TUDAT_NAN );
63  perturbedState = originalState;
64  }
65 
66 
67  // Reset state/environment to original state.
68  setBodyState( perturbedState );
69  updateFunction( );
70 
71  basic_astrodynamics::updateAndGetAcceleration< Eigen::Vector3d >(
72  accelerationModel, evaluationTime );
73 
74  // Numerically compute partial derivatives.
75  Eigen::Matrix3d accelerationPartials = upAccelerations - downAccelerations;
76  for( int i = 0; i < 3; i++ )
77  {
78  accelerationPartials.block( 0, i, 3, 1 ) /= ( 2.0 * statePerturbation( i ) );
79  }
80 
81  return accelerationPartials;
82 }

86  std::shared_ptr< estimatable_parameters::EstimatableParameter< double > > parameter,
87  std::shared_ptr< basic_astrodynamics::AccelerationModel< Eigen::Vector3d > > accelerationModel,
88  double parameterPerturbation,
89  std::function< void( ) > updateDependentVariables,

91  std::function< void( const double ) > timeDependentUpdateDependentVariables )
92 {

95

104  accelerationModel, currentTime );
105  accelerationModel->resetTime( TUDAT_NAN );
106  // Calculate down-perturbation.
107  parameter->setParameterValue(
108  unperturbedParameterValue - parameterPerturbation );
109  updateDependentVariables( );
110  timeDependentUpdateDependentVariables( currentTime );
111 
112  Eigen::Vector3d downPerturbedAcceleration = basic_astrodynamics::updateAndGetAcceleration< Eigen::Vector3d >(
113  accelerationModel, currentTime );
114 
115  accelerationModel->resetTime( TUDAT_NAN );
116 
117  // Reset to original value.
118  parameter->setParameterValue(
119  unperturbedParameterValue ) ;
120  updateDependentVariables( );
121  timeDependentUpdateDependentVariables( currentTime );
122 
123  accelerationModel->resetTime( TUDAT_NAN );
124  basic_astrodynamics::updateAndGetAcceleration< Eigen::Vector3d >(

135  std::shared_ptr< basic_astrodynamics::AccelerationModel< Eigen::Vector3d > > accelerationModel,

145

157  {
158  perturbedParameterValue = unperturbedParameterValue;
159  perturbedParameterValue( i ) += parameterPerturbation( i );
160 
161  // Calculate up-perturbation
162  parameter->setParameterValue( perturbedParameterValue );
163  updateDependentVariables( );
164  timeDependentUpdateDependentVariables( currentTime );
165  Eigen::Vector3d upPerturbedAcceleration = basic_astrodynamics::updateAndGetAcceleration< Eigen::Vector3d >(
166  accelerationModel, currentTime );
167  accelerationModel->resetTime( TUDAT_NAN );
168 
169  // Calculate down-perturbation.
170  perturbedParameterValue = unperturbedParameterValue;
171  perturbedParameterValue( i ) -= parameterPerturbation( i );
172  parameter->setParameterValue( perturbedParameterValue );
173  updateDependentVariables( );
174  timeDependentUpdateDependentVariables( currentTime );
175  Eigen::Vector3d downPerturbedAcceleration = basic_astrodynamics::updateAndGetAcceleration< Eigen::Vector3d >(
176  accelerationModel, currentTime );
177  accelerationModel->resetTime( TUDAT_NAN );
178 
179  // Compute partial entry.
180  partialMatrix.block( 0, i, 3, 1 ) =
181  ( upPerturbedAcceleration - downPerturbedAcceleration ) / ( 2.0 * parameterPerturbation( i ) );
182 
183  }
184

186  parameter->setParameterValue(

190  accelerationModel->resetTime( TUDAT_NAN );
191 
192  basic_astrodynamics::updateAndGetAcceleration< Eigen::Vector3d >(
193  accelerationModel, currentTime );
194 
195  // Calculate partial using central difference.
196  return partialMatrix;
