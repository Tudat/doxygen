252  ObservationScalarType calculateLightTimeWithLinkEndsStates(
253  StateType& receiverStateOutput,
254  StateType& transmitterStateOutput,
255  const TimeType time,
256  const bool isTimeAtReception = 1,
257  const ObservationScalarType tolerance =
258  ( getDefaultLightTimeTolerance< ObservationScalarType >( ) ) )
259  {
260  using physical_constants::SPEED_OF_LIGHT;
261  using std::fabs;
262 
263  // Initialize reception and transmission times and states to initial guess (zero light time)
264  TimeType receptionTime = time;
265  TimeType transmissionTime = time;
266  StateType receiverState = stateFunctionOfReceivingBody_( receptionTime );
267  StateType transmitterState =
268  stateFunctionOfTransmittingBody_( transmissionTime );
269 
270  // Set initial light-time correction.
271  setTotalLightTimeCorrection(
272  transmitterState, receiverState, transmissionTime, receptionTime );
273 
274  // Calculate light-time solution assuming infinte speed of signal as initial estimate.
275  ObservationScalarType previousLightTimeCalculation =
276  calculateNewLightTimeEstime( receiverState, transmitterState );
277 
278  // Set variables for iteration
279  ObservationScalarType newLightTimeCalculation = 0.0;
280  bool isToleranceReached = false;
281 
282  // Recalculate light-time solution until tolerance is reached.
283  int counter = 0;
284 
285  // Set variable determining whether to update the light time each iteration.
286  bool updateLightTimeCorrections = false;
287  if( iterateCorrections_ )
288  {
289  updateLightTimeCorrections = true;
290  }
291 
292  // Iterate until tolerance reached.
293  while( !isToleranceReached )
294  {
295  // Update light-time corrections, if necessary.
296  if( updateLightTimeCorrections )
297  {
298  setTotalLightTimeCorrection(
299  transmitterState, receiverState, transmissionTime, receptionTime );
300  }
301 
302  // Update light-time estimate for this iteration.
303  if( isTimeAtReception )
304  {
305  receptionTime = time;
306  transmissionTime = time - previousLightTimeCalculation;
307  transmitterState = ( stateFunctionOfTransmittingBody_( transmissionTime ) );
308  }
309  else
310  {
311  receptionTime = time + previousLightTimeCalculation;
312  transmissionTime = time;
313  receiverState = ( stateFunctionOfReceivingBody_( receptionTime ) );
314  }
315  newLightTimeCalculation = calculateNewLightTimeEstime( receiverState, transmitterState );
316 
317  // Check for convergence.
318  if( fabs( newLightTimeCalculation - previousLightTimeCalculation ) < tolerance )
319  {
320  // If convergence reached, but light-time corrections not iterated,
321  // perform 1 more iteration to check for change in correction.
322  if( !updateLightTimeCorrections )
323  {
324  updateLightTimeCorrections = true;
325  }
326  else
327  {
328  isToleranceReached = true;
329  }
330  }
331  else
332  {
333  // Get out of infinite loop (for instance due to low accuracy state functions,
334  // to stringent tolerance or limit case for trop. corrections).
335  if( counter == 50 )
336  {
337  isToleranceReached = true;
338  std::string errorMessage =
339  "Warning, light time unconverged at level " +
340  boost::lexical_cast< std::string >(
341  fabs( newLightTimeCalculation - previousLightTimeCalculation ) ) +
342  "; current light-time corrections are: " +
343  boost::lexical_cast< std::string >( currentCorrection_ ) + " and input time was " +
344  boost::lexical_cast< std::string >( time );
345  std::cerr << errorMessage <<std::endl;
346  }
347 
348  // Update light time for new iteration.
349  previousLightTimeCalculation = newLightTimeCalculation;
350  }
351 
352  counter++;
353  }
354 
355  // Set output variables and return the light time.
356  receiverStateOutput = receiverState;
357  transmitterStateOutput = transmitterState;
358 
359  return newLightTimeCalculation;
360  }
