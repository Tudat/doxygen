53 double computeCosineOfAngleBetweenVectors( const Eigen::VectorXd& vector0,
54  const Eigen::VectorXd& vector1 )
55 {
56  if( !( vector0.size( ) == vector1.size( ) ) )
57  {
58  throw std::runtime_error( "Error when computing angle between vectors; size is incompatible" );
59  }
60 
61  // Get the cosine of the angle by dotting the normalized vectors.
62  double dotProductOfNormalizedVectors = vector0.normalized( ).dot( vector1.normalized( ) );
63 
64  // Explicitly define the extreme cases, which can give problems with the acos function.
65  if ( dotProductOfNormalizedVectors >= 1.0 )
66  {
67  return 1.0;
68  }
69 
70  else if ( dotProductOfNormalizedVectors <= -1.0 )
71  {
72  return -1.0;
73  }
74  // Determine the actual angle.
75  else
76  {
77  return dotProductOfNormalizedVectors;
78  }
79 }
