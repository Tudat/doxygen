36 template< typename IndependentVariableType, typename DependentVariableType >
37 DependentVariableType performTrapezoidalQuadrature(
38  const std::vector< IndependentVariableType >& independentVariables,
39  const std::vector< DependentVariableType >& dependentVariables )
40 {
41  DependentVariableType integral = dependentVariables.at( 0 ) - dependentVariables.at( 0 );
42  IndependentVariableType timeStep;
43  for( unsigned int i = 0 ; i < independentVariables.size( ) - 1 ; i++ )
44  {
45  timeStep = independentVariables[ i + 1 ] - independentVariables[ i ];
46  integral += timeStep * ( dependentVariables[ i + 1 ] + dependentVariables[ i ] ) / 2.0 ;
47  }
48  return integral;
49 }
