120 template< typename ParameterType >
121 RotationMatrixPartialNamedList createRotationMatrixPartials(
122  const boost::shared_ptr< estimatable_parameters::EstimatableParameterSet< ParameterType > > parametersToEstimate,
123  const std::string& bodyName, const simulation_setup::NamedBodyMap& bodyMap )
124 
125 {
126  // Declare map to return.
127  std::map< std::pair< estimatable_parameters::EstimatebleParametersEnum, std::string >,
128  boost::shared_ptr< RotationMatrixPartial > >
129  rotationMatrixPartials;
130 
131  // Retrieve double and vector parameters from total set of parameters.
132  std::map< int, boost::shared_ptr< estimatable_parameters::EstimatableParameter< double > > > doubleParameters =
133  parametersToEstimate->getDoubleParameters( );
134  std::map< int, boost::shared_ptr< estimatable_parameters::EstimatableParameter< Eigen::VectorXd > > > vectorParameters =
135  parametersToEstimate->getVectorParameters( );
136 
137  // Iterate over double parameters.
138  for( std::map< int, boost::shared_ptr< estimatable_parameters::EstimatableParameter< double > > >::iterator
139  parameterIterator = doubleParameters.begin( ); parameterIterator != doubleParameters.end( ); parameterIterator++ )
140  {
141  // Check parameter is rotational property of requested body.
142  if( ( parameterIterator->second->getParameterName( ).second.first == bodyName ) &&
143  ( estimatable_parameters::isParameterRotationMatrixProperty(
144  parameterIterator->second->getParameterName( ).first ) ) )
145  {
146  // Create partial object.
147  rotationMatrixPartials[ std::make_pair(
148  parameterIterator->second->getParameterName( ).first,
149  parameterIterator->second->getSecondaryIdentifier( ) ) ] =
150  createRotationMatrixPartialsWrtParameter( bodyMap, parameterIterator->second );
151  }
152  }
153 
154  // Iterate over vector parameters.
155  for( std::map< int, boost::shared_ptr< estimatable_parameters::EstimatableParameter< Eigen::VectorXd > > >::iterator
156  parameterIterator = vectorParameters.begin( ); parameterIterator != vectorParameters.end( ); parameterIterator++ )
157  {
158  // Check parameter is rotational property of requested body.
159  if( ( parameterIterator->second->getParameterName( ).second.first == bodyName ) &&
160  ( estimatable_parameters::isParameterRotationMatrixProperty(
161  parameterIterator->second->getParameterName( ).first ) ) )
162  {
163  // Create partial object.
164  rotationMatrixPartials[ std::make_pair(
165  parameterIterator->second->getParameterName( ).first,
166  parameterIterator->second->getSecondaryIdentifier( ) ) ] =
167  createRotationMatrixPartialsWrtParameter( bodyMap, parameterIterator->second );
168  }
169  }
170 
171  return rotationMatrixPartials;
172 }

206 template< typename ParameterType >
207 std::pair< SingleLinkObservationThreePartialList, boost::shared_ptr< PositionPartialScaling > >
208 createPositionObservablePartials(
209  const observation_models::LinkEnds positionObservableLinkEnds,
210  const simulation_setup::NamedBodyMap& bodyMap,
211  const boost::shared_ptr< estimatable_parameters::EstimatableParameterSet< ParameterType > > parametersToEstimate )
212 
213 {
214  // Create scaling object, to be used for each partial created here (i.e. same scaling for different parameters but same
215  // observable).
216  boost::shared_ptr< PositionObservationScaling > positionObservableScaling =
217  boost::make_shared< PositionObservationScaling >( );
218 
219  SingleLinkObservationThreePartialList positionObservablePartials;
220 
221  // Define start index and size of current parameter under consideration
222  int currentIndex = 0;
223  std::pair< int, int > currentPair = std::pair< int, int >( currentIndex, 1 );
224 
225  std::vector< boost::shared_ptr< estimatable_parameters::EstimatableParameter<
226  Eigen::Matrix< ParameterType, Eigen::Dynamic, 1 > > > > initialDynamicalParameters =
227  parametersToEstimate->getEstimatedInitialStateParameters( );
228 
229  // Iterate over list of bodies of which the partials of the accelerations acting on them are required.
230  for( unsigned int i = 0; i < initialDynamicalParameters.size( ); i++ )
231  {
232  if( boost::dynamic_pointer_cast< estimatable_parameters::InitialTranslationalStateParameter< ParameterType > >(
233  initialDynamicalParameters.at( i ) ) != NULL )
234  {
235 
236  // Retrieve name of body w.r.t. position of which partial is to be taken.
237  std::string acceleratedBody = boost::dynamic_pointer_cast<
238  estimatable_parameters::InitialTranslationalStateParameter< ParameterType > >(
239  initialDynamicalParameters.at( i ) )->getParameterName( ).second.first;
240 
241  // Create partial (if needed)
242  boost::shared_ptr< PositionObervationPartial > currentObservablePartial =
243  createPositionObservablePartialWrtPosition(
244  positionObservableLinkEnds, bodyMap, acceleratedBody, positionObservableScaling );
245 
246  // If partial exists, then dependency exists and parameter must be added.
247  if( currentObservablePartial != NULL )
248  {
249  currentPair = std::pair< int, int >( currentIndex, 6 );
250  positionObservablePartials[ currentPair ] = currentObservablePartial;
251  }
252 
253  currentIndex += 6;
254  }
255  }
256 
257  return std::make_pair( positionObservablePartials, positionObservableScaling );
258 }
