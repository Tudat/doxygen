283 template< int ObservationSize >
284 boost::shared_ptr< ObservationPartial< ObservationSize > > createObservationPartialWrtLinkProperty(
285  const observation_models::LinkEnds& linkEnds,
286  const observation_models::ObservableType observableType,
287  const boost::shared_ptr< estimatable_parameters::EstimatableParameter< Eigen::VectorXd > > parameterToEstimate )
288 {
289  boost::shared_ptr< ObservationPartial< ObservationSize > > observationPartial;
290 
291  // Check parameter type
292  switch( parameterToEstimate->getParameterName( ).first )
293  {
294  case estimatable_parameters::constant_additive_observation_bias:
295  {
296  // Check input consistency
297  boost::shared_ptr< estimatable_parameters::ConstantObservationBiasParameter > constantBias =
298  boost::dynamic_pointer_cast< estimatable_parameters::ConstantObservationBiasParameter >(
299  parameterToEstimate );
300  if( constantBias == NULL )
301  {
302  throw std::runtime_error( "Error when making partial w.r.t. observation bias, type is inconsistent" );
303  }
304  else
305  {
306  // Check dependency between parameter and link properties.
307  if( linkEnds == constantBias->getLinkEnds( ) && observableType == constantBias->getObservableType( ) )
308  {
309  observationPartial = boost::make_shared< ObservationPartialWrtConstantAbsoluteBias< ObservationSize > >(
310  observableType, linkEnds );
311  }
312  }
313  break;
314  }
315  case estimatable_parameters::constant_relative_observation_bias:
316  {
317  // Check input consistency
318  boost::shared_ptr< estimatable_parameters::ConstantRelativeObservationBiasParameter > constantBias =
319  boost::dynamic_pointer_cast< estimatable_parameters::ConstantRelativeObservationBiasParameter >(
320  parameterToEstimate );
321  if( constantBias == NULL )
322  {
323  throw std::runtime_error( "Error when making partial w.r.t. observation bias, type is inconsistent" );
324  }
325  else
326  {
327  // Check dependency between parameter and link properties.
328  if( linkEnds == constantBias->getLinkEnds( ) && observableType == constantBias->getObservableType( ) )
329  {
330  observationPartial = boost::make_shared< ObservationPartialWrtConstantRelativeBias< ObservationSize > >(
331  observableType, linkEnds );
332  }
333  }
334  break;
335  }
336  default:
337  break;
338  }
339  return observationPartial;
340 }
