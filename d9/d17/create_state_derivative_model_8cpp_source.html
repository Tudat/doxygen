38 std::map< double, Eigen::Vector6d > performCR3BPIntegration(
39  const std::shared_ptr< numerical_integrators::IntegratorSettings< double > > integratorSettings,
40  const double massParameter,
41  const Eigen::Vector6d& initialState,
42  const double finalTime )
43 {
44  // Create integrator object
45  std::shared_ptr< numerical_integrators::NumericalIntegrator< double, Eigen::Vector6d > > integrator =
46  createCR3BPIntegrator( integratorSettings, massParameter, initialState );
47 
48  // Initialize return data map
49  std::map< double, Eigen::Vector6d > stateHistory;
50 
51  // Store initial state and time
52  double currentTime = integratorSettings->initialTime_;
53  Eigen::Vector6d currentState = initialState;
54  stateHistory[ currentTime ] = currentState;
55 
56  // Integrate to final time
57  double timeStep = integratorSettings->initialTimeStep_;
58  while( currentTime <= finalTime )
59  {
60  currentState = integrator->performIntegrationStep( timeStep );
61  currentTime = integrator->getCurrentIndependentVariable( );
62  timeStep = integrator->getNextStepSize( );
63  stateHistory[ currentTime ] = currentState;
64  }
65 
66  return stateHistory;
