111 static inline void flipMatrixRows( Eigen::MatrixXd& matrixToFlip )
112 {
113  // Loop through each row.
114  for ( int i = 0; i < static_cast< int >( matrixToFlip.rows( ) / 2 ); i++ )
115  {
116  // Save top row in a temporary matrix.
117  const Eigen::MatrixXd temporaryRow = matrixToFlip.row( i );
118 
119  // Set top row to bottom row.
120  matrixToFlip.row( i ) = matrixToFlip.row( matrixToFlip.rows( ) - 1 - i );
121 
122  // Set bottom row to temporarily stored matrix.
123  matrixToFlip.row( matrixToFlip.rows( ) - 1 - i ) = temporaryRow;
124  }
125 }
