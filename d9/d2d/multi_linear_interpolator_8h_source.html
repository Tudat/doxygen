117  DependentVariableType interpolate(
118  const std::vector< IndependentVariableType >& independentValuesToInterpolate )
119  {
120  // Determine the nearest lower neighbours.
121  std::vector< int > nearestLowerIndices;
122  nearestLowerIndices.resize( NumberOfDimensions );
123  for ( unsigned int i = 0; i < NumberOfDimensions; i++ )
124  {
125  nearestLowerIndices[ i ] = lookUpSchemes_[ i ]->findNearestLowerNeighbour(
126  independentValuesToInterpolate[ i ] );
127  }
128 
129  // Initialize function evaluation indices to -1 for debugging purposes.
130  boost::array< int, NumberOfDimensions > interpolationIndices;
131  for ( int i = 0; i < NumberOfDimensions; i++ )
132  {
133  interpolationIndices[ i ] = -1;
134  }
135 
136  // Call first step of interpolation, this function calls itself at subsequent independent
137  // variable dimensions to evaluate and properly scale dependent variable table values at
138  // all 2^n grid edges.
139  return performRecursiveInterpolationStep( 0, independentValuesToInterpolate,
140  interpolationIndices, nearestLowerIndices );
141  }

165  void makeLookupSchemes( const AvailableLookupScheme selectedScheme )
166  {
167  lookUpSchemes_.resize( NumberOfDimensions );
168  // Find which type of scheme is used.
169  switch( selectedScheme )
170  {
171  case binarySearch:
172 
173  for( int i = 0; i < NumberOfDimensions; i++ )
174  {
175  // Create binary search look up scheme.
176  lookUpSchemes_[ i ] = boost::shared_ptr< LookUpScheme< IndependentVariableType > >
177  ( new BinarySearchLookupScheme< IndependentVariableType >(
178  independentValues_[ i ] ) );
179  }
180 
181  break;
182 
183  case huntingAlgorithm:
184 
185  for( int i = 0; i < NumberOfDimensions; i++ )
186  {
187  // Create hunting scheme, which uses an intial guess from previous look-ups.
188  lookUpSchemes_[ i ] = boost::shared_ptr< LookUpScheme< IndependentVariableType > >
189  ( new HuntingAlgorithmLookupScheme< IndependentVariableType >(
190  independentValues_[ i ] ) );
191  }
192 
193  break;
194 
195  default:
196 
197  throw std::runtime_error( "Warning: lookup scheme not found when making scheme for 1-D interpolator" );
198  }

220  DependentVariableType performRecursiveInterpolationStep(
221  const unsigned int currentVariable,
222  const std::vector< IndependentVariableType >& independentValuesToInterpolate,
223  boost::array< int, NumberOfDimensions > currentArrayIndices,
224  const std::vector< int >& nearestLowerIndices )
225  {
226  IndependentVariableType upperFraction, lowerFraction;
227  DependentVariableType upperContribution, lowerContribution;
228 
229  // Calculate fractions of data points above and below independent
230  // variable value to be added to interpolated value.
231  upperFraction = ( independentValuesToInterpolate[ currentVariable ] -
232  independentValues_[ currentVariable ]
233  [ nearestLowerIndices[ currentVariable ] ] ) /
234  ( independentValues_[ currentVariable ]
235  [ nearestLowerIndices[ currentVariable ] + 1 ] -
236  independentValues_[ currentVariable ]
237  [ nearestLowerIndices[ currentVariable ] ] );
238  lowerFraction = -( independentValuesToInterpolate[ currentVariable ] -
239  independentValues_[ currentVariable ]
240  [ nearestLowerIndices[ currentVariable ] + 1 ] ) /
241  ( independentValues_[ currentVariable ]
242  [ nearestLowerIndices[ currentVariable ] + 1 ] -
243  independentValues_[ currentVariable ]
244  [ nearestLowerIndices[ currentVariable ] ] );
245 
246  // If at top dimension, call dependent variable data.
247  if ( currentVariable == NumberOfDimensions - 1 )
248  {
249  currentArrayIndices[ NumberOfDimensions - 1 ] = nearestLowerIndices[ currentVariable ];
250  lowerContribution = dependentData_( currentArrayIndices );
251  currentArrayIndices[ NumberOfDimensions - 1 ] = nearestLowerIndices[ currentVariable ]
252  + 1;
253  upperContribution = dependentData_( currentArrayIndices );
254  }
255 
256  // If at lower dimension, update currentArrayIndices and call function with
257  // currentVariable++.
258  else
259  {
260  currentArrayIndices[ currentVariable ] = nearestLowerIndices[ currentVariable ];
261  lowerContribution = performRecursiveInterpolationStep(
262  currentVariable + 1, independentValuesToInterpolate,
263  currentArrayIndices, nearestLowerIndices );
264  currentArrayIndices[ currentVariable ] = nearestLowerIndices[ currentVariable ] + 1;
265  upperContribution = performRecursiveInterpolationStep(
266  currentVariable + 1, independentValuesToInterpolate,
267  currentArrayIndices, nearestLowerIndices );
268  }
269 
270  // Return interpolated value.
271  DependentVariableType returnValue = upperFraction * upperContribution +
272  lowerFraction * lowerContribution;
273  return returnValue;
274  }
