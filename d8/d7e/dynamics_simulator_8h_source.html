62  const simulation_setup::NamedBodyMap& bodyMap,
63  const TimeType initialTime,
64  const boost::shared_ptr< ephemerides::ReferenceFrameManager > frameManager )
65 {
66  // Set initial states of bodies to integrate.
67  Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > systemInitialState =
68  Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 >::Zero( bodiesToIntegrate.size( ) * 6, 1 );
69  boost::shared_ptr< ephemerides::Ephemeris > ephemerisOfCurrentBody;
70 
71  // Iterate over all bodies.
72  for( unsigned int i = 0; i < bodiesToIntegrate.size( ) ; i++ )
73  {
74  ephemerisOfCurrentBody = bodyMap.at( bodiesToIntegrate.at( i ) )->getEphemeris( );
75 
76  if ( ! ephemerisOfCurrentBody )
77  {
78  throw std::runtime_error( "Could not determine initial state for body " + bodiesToIntegrate.at( i ) +
79  " because it does not have a valid Ephemeris object." );
80  }
81 
82  // Get body initial state from ephemeris
83  systemInitialState.segment( i * 6 , 6 ) = ephemerisOfCurrentBody->getTemplatedStateFromEphemeris<
84  StateScalarType, TimeType >( initialTime );
85 
86  // Correct initial state if integration origin and ephemeris origin are not equal.
87  if( centralBodies.at( i ) != ephemerisOfCurrentBody->getReferenceFrameOrigin( ) )
88  {
89  boost::shared_ptr< ephemerides::Ephemeris > correctionEphemeris =
90  frameManager->getEphemeris( ephemerisOfCurrentBody->getReferenceFrameOrigin( ), centralBodies.at( i ) );
91  systemInitialState.segment( i * 6 , 6 ) -= correctionEphemeris->getTemplatedStateFromEphemeris<

93  }
