57 template< typename TimeType = double, typename StateScalarType = double >
58 Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > getInitialStatesOfBodies(
59  const std::vector< std::string >& bodiesToIntegrate,
60  const std::vector< std::string >& centralBodies,
61  const simulation_setup::NamedBodyMap& bodyMap,
62  const TimeType initialTime,
63  const boost::shared_ptr< ephemerides::ReferenceFrameManager > frameManager )
64 {
65  // Set initial states of bodies to integrate.
66  Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > systemInitialState =
67  Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 >::Zero( bodiesToIntegrate.size( ) * 6, 1 );
68  boost::shared_ptr< ephemerides::Ephemeris > ephemerisOfCurrentBody;
69 
70  // Iterate over all bodies.
71  for( unsigned int i = 0; i < bodiesToIntegrate.size( ) ; i++ )
72  {
73  ephemerisOfCurrentBody = bodyMap.at( bodiesToIntegrate.at( i ) )->getEphemeris( );
74 
75  // Get body initial state from ephemeris
76  systemInitialState.segment( i * 6 , 6 ) = ephemerisOfCurrentBody->getTemplatedStateFromEphemeris<
77  StateScalarType, TimeType >( initialTime );
78 
79  // Correct initial state if integration origin and ephemeris origin are not equal.
80  if( centralBodies.at( i ) != ephemerisOfCurrentBody->getReferenceFrameOrigin( ) )
81  {
82  boost::shared_ptr< ephemerides::Ephemeris > correctionEphemeris =
83  frameManager->getEphemeris( ephemerisOfCurrentBody->getReferenceFrameOrigin( ), centralBodies.at( i ) );
84  systemInitialState.segment( i * 6 , 6 ) -= correctionEphemeris->getTemplatedStateFromEphemeris<
85  StateScalarType, TimeType >( initialTime );
86  }
87  }
88  return systemInitialState;
89 }

688 template< typename StateScalarType = double, typename TimeType = double >
689 Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > getArcInitialStateFromPreviousArcResult(
690  const std::map< TimeType, Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > >& previousArcDynamisSolution,
691  const double currentArcInitialTime )
692 {
693  Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > currentArcInitialState;
694  {
695  // Check if overlap exists
696  if( previousArcDynamisSolution.rbegin( )->first < currentArcInitialTime )
697  {
698  throw std::runtime_error(
699  "Error when getting initial arc state from previous arc: no arc overlap" );
700  }
701  else
702  {
703  int currentIndex = 0;
704  int initialTimeIndex = -1;
705 
706  std::map< TimeType, Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > > initialStateInterpolationMap;
707 
708  // Set sub-part of previous arc to interpolate for current arc
709  for( typename std::map< TimeType, Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > >::
710  const_reverse_iterator previousArcIterator = previousArcDynamisSolution.rbegin( );
711  previousArcIterator != previousArcDynamisSolution.rend( ); previousArcIterator++ )
712  {
713  initialStateInterpolationMap[ previousArcIterator->first ] = previousArcIterator->second;
714  if( initialTimeIndex < 0 )
715  {
716  if( previousArcIterator->first < currentArcInitialTime )
717  {
718  initialTimeIndex = currentIndex;
719  }
720  }
721  else
722  {
723  if( currentIndex - initialTimeIndex > 5 )
724  {
725  break;
726  }
727  }
728  currentIndex++;
729  }
730 
731  // Interpolate to obtain initial state of current arc
732  currentArcInitialState =
733  boost::make_shared< interpolators::LagrangeInterpolator<
734  TimeType, Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 >, long double > >(
735  initialStateInterpolationMap, 8 )->interpolate( currentArcInitialTime );
736 
737  }
738  }
739  return currentArcInitialState;
740 }

927  {

929  for( unsigned int i = 0; i < equationsOfMotionNumericalSolution_.size( ); i++ )
930  {
931  equationsOfMotionNumericalSolution_.at( i ).clear( );

937  }

939 
940  Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > currentArcInitialState;
941  std::vector< Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > > arcInitialStateList;
942  bool updateInitialStates = false;
943 
944  // Propagate dynamics for each arc
945  for( unsigned int i = 0; i < singleArcDynamicsSimulators_.size( ); i++ )
946  {
947  // Get arc initial state. If initial state is NaN, this signals that the initial state is to be taken from previous
948  // arc
949  if( ( i == 0 ) || ( !linear_algebra::doesMatrixHaveNanEntries( initialStatesList.at( i ) ) ) )
950  {

952  }
953  else
954  {

956  equationsOfMotionNumericalSolution_.at( i - 1 ),
957  singleArcDynamicsSimulators_.at( i )->getInitialPropagationTime( ) );
958 
959  // If arc initial state is taken from previous arc, this indicates that the initial states in propagator settings
960  // need to be updated.
961  updateInitialStates = true;
962  }
963  arcInitialStateList.push_back( currentArcInitialState );
964 
965  singleArcDynamicsSimulators_.at( i )->integrateEquationsOfMotion( currentArcInitialState );
966  equationsOfMotionNumericalSolution_[ i ] =
967  singleArcDynamicsSimulators_.at( i )->getEquationsOfMotionNumericalSolution( );
968  dependentVariableHistory_[ i ] =
969  singleArcDynamicsSimulators_.at( i )->getDependentVariableHistory( );
970  propagationTerminationReasons_[ i ] = singleArcDynamicsSimulators_.at( i )->getPropagationTerminationReason( );

972  }
973

976  {
977  multiArcPropagatorSettings_->resetInitialStatesList(
978  arcInitialStateList );
979  }

982  {
