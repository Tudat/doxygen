21 void AerodynamicAccelerationPartial::update( const double currentTime )
22 {
23  Eigen::Vector6d nominalState = vehicleStateGetFunction_( );
24  Eigen::Vector6d perturbedState;
25 
26  // Compute state partial by numerical difference
27  Eigen::Vector3d upperturbedAcceleration, downperturbedAcceleration;
28  for( unsigned int i = 0; i < 6; i++ )
29  {
30  // Perturb state upwards
31  perturbedState = nominalState;
32  perturbedState( i ) += bodyStatePerturbations_( i );
33 
34  // Update environment/acceleration to perturbed state.
35  flightConditions_->resetCurrentTime( TUDAT_NAN );
36  aerodynamicAcceleration_->resetTime( TUDAT_NAN );
37  vehicleStateSetFunction_( perturbedState );
38  flightConditions_->updateConditions( currentTime );
39  aerodynamicAcceleration_->updateMembers( currentTime );
40 
41  // Retrieve perturbed acceleration.
42  upperturbedAcceleration = aerodynamicAcceleration_->getAcceleration( );
43 
44  // Perturb state downwards
45  perturbedState = nominalState;
46  perturbedState( i ) -= bodyStatePerturbations_( i );
47 
48  // Update environment/acceleration to perturbed state.
49  flightConditions_->resetCurrentTime( TUDAT_NAN );
50  aerodynamicAcceleration_->resetTime( TUDAT_NAN );
51  vehicleStateSetFunction_( perturbedState );
52  flightConditions_->updateConditions( currentTime );
53  aerodynamicAcceleration_->updateMembers( currentTime );
54 
55  // Retrieve perturbed acceleration.
56  downperturbedAcceleration = aerodynamicAcceleration_->getAcceleration( );
57 
58  // Compute partial
59  currentAccelerationStatePartials_.block( 0, i, 3, 1 ) =
60  ( upperturbedAcceleration - downperturbedAcceleration ) / ( 2.0 * bodyStatePerturbations_( i ) );
61  }
62 
63  // Reset environment/acceleration mode to nominal conditions
64  flightConditions_->resetCurrentTime( TUDAT_NAN );
65  aerodynamicAcceleration_->resetTime( TUDAT_NAN );
66 
67  vehicleStateSetFunction_( nominalState );
68  flightConditions_->updateConditions( currentTime );
69  aerodynamicAcceleration_->updateMembers( currentTime );
70 }
