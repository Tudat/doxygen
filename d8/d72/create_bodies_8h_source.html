104 template< typename StateScalarType = double, typename TimeType = double >
105 void setGlobalFrameBodyEphemerides( const NamedBodyMap& bodyMap,
106  const std::string& globalFrameOrigin,
107  const std::string& globalFrameOrientation )
108 {
109  using namespace tudat::simulation_setup;
110  std::string ephemerisFrameOrigin;
111  std::string ephemerisFrameOrientation;
112  std::string rotationModelFrame;
113 
114  // Iterate over all bodies
115  for( NamedBodyMap::const_iterator bodyIterator = bodyMap.begin( );
116  bodyIterator != bodyMap.end( ); bodyIterator++ )
117  {
118  // Check id body contains an ephemeris
119  if( bodyIterator->second->getEphemeris( ) != NULL )
120  {
121  // Retrieve ephemeris origin
122  ephemerisFrameOrigin = bodyIterator->second->getEphemeris( )->getReferenceFrameOrigin( );
123 
124  // Check if ephemeris origin differs from global origin.
125  if( ephemerisFrameOrigin != globalFrameOrigin )
126  {
127  // Check if correction can be made
128  if( bodyMap.count( ephemerisFrameOrigin ) == 0 )
129  {
130  throw std::runtime_error(
131  "Error, body " + bodyIterator->first + " has ephemeris in frame " +
132  ephemerisFrameOrigin + ", but no conversion to frame " + globalFrameOrigin +
133  " can be made" );
134  }
135  else
136  {
137  boost::function< Eigen::Matrix< StateScalarType, 6, 1 >( const TimeType ) > stateFunction =
138  boost::bind( &Body::getStateInBaseFrameFromEphemeris< StateScalarType, TimeType >,
139  bodyMap.at( ephemerisFrameOrigin ), _1 );
140 
141  boost::shared_ptr< BaseStateInterface > baseStateInterface = boost::make_shared< BaseStateInterfaceImplementation< double, double > >(
142  ephemerisFrameOrigin, stateFunction );
143 
144  bodyIterator->second->setEphemerisFrameToBaseFrame( baseStateInterface );
145  }
146  }
147 
148  // Retrieve ephemeris orientation
149  ephemerisFrameOrientation = bodyIterator->second->getEphemeris( )->getReferenceFrameOrientation( );
150 
151  // If two are not equal, throw error.
152  if( ephemerisFrameOrientation != globalFrameOrientation )
153  {
154  throw std::runtime_error(
155  "Error, ephemeris orientation of body " + bodyIterator->first
156  + " is not the same as global orientation " + ephemerisFrameOrientation
157  + ", " + globalFrameOrientation );
158  }
159 
160 
161  }
162 
163  // Check if body has rotational ephemeris.
164  if( bodyIterator->second->getRotationalEphemeris( ) != NULL )
165  {
166  // Check if rotational ephemeris base frame orienatation is equal to to global
167  // orientation.
168  rotationModelFrame = bodyIterator->second->getRotationalEphemeris( )
169  ->getBaseFrameOrientation( );
170 
171  // Throw error if two frames are not equal.
172  if( rotationModelFrame != globalFrameOrientation )
173  {
174  throw std::runtime_error(
175  "Error, rotation base orientation of body " + bodyIterator->first +
176  " is not the same as global orientation " + rotationModelFrame + ", " +
177  globalFrameOrientation );
178  }
179  }
180  }
181 
182 }
