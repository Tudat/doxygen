647 template< typename StateScalarType = double, typename TimeType = double >
648 boost::shared_ptr< ephemerides::Ephemeris > createTabulatedEphemerisFromSpice(
649  const std::string& body,
650  const TimeType initialTime,
651  const TimeType endTime,
652  const TimeType timeStep,
653  const std::string& observerName,
654  const std::string& referenceFrameName,
655  boost::shared_ptr< interpolators::InterpolatorSettings > interpolatorSettings =
656  boost::make_shared< interpolators::LagrangeInterpolatorSettings >( 8 ) )
657 {
658  using namespace interpolators;
659 
660  std::map< TimeType, Eigen::Matrix< StateScalarType, 6, 1 > > timeHistoryOfState;
661 
662  // Calculate state from spice at given time intervals and store in timeHistoryOfState.
663  TimeType currentTime = initialTime;
664  while( currentTime < endTime )
665  {
666  timeHistoryOfState[ currentTime ] = spice_interface::getBodyCartesianStateAtEpoch(
667  body, observerName, referenceFrameName, "none", static_cast< double >( currentTime ) ).
668  template cast< StateScalarType >( );
669  currentTime += timeStep;
670  }
671 
672  // Create interpolator.
673  boost::shared_ptr< OneDimensionalInterpolator< TimeType, Eigen::Matrix< StateScalarType, 6, 1 > > > interpolator =
674  interpolators::createOneDimensionalInterpolator(
675  timeHistoryOfState, interpolatorSettings );
676 
677  // Create ephemeris and return.
678  return boost::make_shared< ephemerides::TabulatedCartesianEphemeris< StateScalarType, TimeType > >(
679  interpolator, observerName, referenceFrameName );
680 }
681 #endif
