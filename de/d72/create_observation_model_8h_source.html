484 template< typename ObservationScalarType = double, typename TimeType = double >
485 std::shared_ptr< DopplerProperTimeRateInterface > createOneWayDopplerProperTimeCalculator(
486  std::shared_ptr< DopplerProperTimeRateSettings > properTimeRateSettings,
487  const LinkEnds& linkEnds,
488  const simulation_setup::NamedBodyMap &bodyMap,
489  const LinkEndType linkEndForCalculator )
490 {
491  std::shared_ptr< DopplerProperTimeRateInterface > properTimeRateInterface;
492 
493  // Check tyope of proper time rate model
494  switch( properTimeRateSettings->dopplerProperTimeRateType_ )
495  {
496  case direct_first_order_doppler_proper_time_rate:
497  {
498  // Check input consistency
499  std::shared_ptr< DirectFirstOrderDopplerProperTimeRateSettings > directFirstOrderDopplerProperTimeRateSettings =
500  std::dynamic_pointer_cast< DirectFirstOrderDopplerProperTimeRateSettings >( properTimeRateSettings );
501  if( directFirstOrderDopplerProperTimeRateSettings == nullptr )
502  {
503  throw std::runtime_error( "Error when making DopplerProperTimeRateInterface, input type (direct_first_order_doppler_proper_time_rate) is inconsistent" );
504  }
505  else if( linkEnds.count( linkEndForCalculator ) == 0 )
506  {
507  std::string errorMessage = "Error when creating one-way Doppler proper time calculator, did not find link end " +
508  std::to_string( linkEndForCalculator );
509  throw std::runtime_error( errorMessage );
510  }
511  else
512  {
513  if( bodyMap.at( directFirstOrderDopplerProperTimeRateSettings->centralBodyName_ )->getGravityFieldModel( ) == nullptr )
514  {
515  throw std::runtime_error( "Error when making DirectFirstOrderDopplerProperTimeRateInterface, no gravity field found for " +
516  directFirstOrderDopplerProperTimeRateSettings->centralBodyName_ );
517  }
518  else
519  {
520  // Retrieve gravitational parameter
521  std::function< double( ) > gravitationalParameterFunction =
522  std::bind( &gravitation::GravityFieldModel::getGravitationalParameter,
523  bodyMap.at( directFirstOrderDopplerProperTimeRateSettings->centralBodyName_ )->
524  getGravityFieldModel( ) );
525 
526  // Create calculation object.
527  LinkEndId referencePointId =
528  std::make_pair( directFirstOrderDopplerProperTimeRateSettings->centralBodyName_, "" );
529  if( ( linkEnds.at( receiver ) != referencePointId ) && ( linkEnds.at( transmitter ) != referencePointId ) )
530  {
531  properTimeRateInterface = std::make_shared<
532  DirectFirstOrderDopplerProperTimeRateInterface >(
533  linkEndForCalculator, gravitationalParameterFunction,
534  directFirstOrderDopplerProperTimeRateSettings->centralBodyName_, unidentified_link_end,
535  getLinkEndCompleteEphemerisFunction< double, double >(
536  std::make_pair( directFirstOrderDopplerProperTimeRateSettings->centralBodyName_, ""), bodyMap ) );
537  }
538  else
539  {
540  throw std::runtime_error(
541  "Error, proper time reference point as link end not yet implemented for DopplerProperTimeRateInterface creation" );
542  }
543  }
544  }
545  break;
546  }
547  default:
548  std::string errorMessage = "Error when creating one-way Doppler proper time calculator, did not recognize type " +
549  std::to_string( properTimeRateSettings->dopplerProperTimeRateType_ );
550  throw std::runtime_error( errorMessage );
551  }
552  return properTimeRateInterface;
553 }

584 template< int ObservationSize = 1 >
585 std::shared_ptr< ObservationBias< ObservationSize > > createObservationBiasCalculator(
586  const LinkEnds linkEnds,
587  const ObservableType observableType,
588  const std::shared_ptr< ObservationBiasSettings > biasSettings,
589  const simulation_setup::NamedBodyMap &bodyMap )
590 {
591  std::shared_ptr< ObservationBias< ObservationSize > > observationBias;
592  switch( biasSettings->observationBiasType_ )
593  {
594  case constant_absolute_bias:
595  {
596  // Check input consistency
597  std::shared_ptr< ConstantObservationBiasSettings > constantBiasSettings = std::dynamic_pointer_cast<
598  ConstantObservationBiasSettings >( biasSettings );
599  if( constantBiasSettings == nullptr )
600  {
601  throw std::runtime_error( "Error when making constant observation bias, settings are inconsistent" );
602  }
603 
604  if( !constantBiasSettings->useAbsoluteBias_ )
605  {
606  throw std::runtime_error( "Error when making constant observation bias, class settings are inconsistent" );
607  }
608 
609  // Check if size of bias is consistent with requested observable size
610  if( constantBiasSettings->observationBias_.rows( ) != ObservationSize )
611  {
612  throw std::runtime_error( "Error when making constant observation bias, bias size is inconsistent" );
613  }
614  observationBias = std::make_shared< ConstantObservationBias< ObservationSize > >(
615  constantBiasSettings->observationBias_ );
616  break;
617  }
618  case arc_wise_constant_absolute_bias:
619  {
620  // Check input consistency
621  std::shared_ptr< ArcWiseConstantObservationBiasSettings > arcwiseBiasSettings = std::dynamic_pointer_cast<
622  ArcWiseConstantObservationBiasSettings >( biasSettings );
623  if( arcwiseBiasSettings == nullptr )
624  {
625  throw std::runtime_error( "Error when making arc-wise observation bias, settings are inconsistent" );
626  }
627  else if( !arcwiseBiasSettings->useAbsoluteBias_ )
628  {
629  throw std::runtime_error( "Error when making arc-wise observation bias, class contents are inconsistent" );
630  }
631 
632  std::vector< Eigen::Matrix< double, ObservationSize, 1 > > observationBiases;
633  for( unsigned int i = 0; i < arcwiseBiasSettings->observationBiases_.size( ); i++ )
634  {
635  // Check if size of bias is consistent with requested observable size
636  if( arcwiseBiasSettings->observationBiases_.at( i ).rows( ) != ObservationSize )
637  {
638  throw std::runtime_error( "Error when making arc-wise observation bias, bias size is inconsistent" );
639  }
640  else
641  {
642  observationBiases.push_back( arcwiseBiasSettings->observationBiases_.at( i ) );
643  }
644  }
645  observationBias = std::make_shared< ConstantArcWiseObservationBias< ObservationSize > >(
646  arcwiseBiasSettings->arcStartTimes_, observationBiases,
647  observation_models::getLinkEndIndicesForLinkEndTypeAtObservable(
648  observableType, arcwiseBiasSettings->linkEndForTime_, linkEnds.size( ) ).at( 0 ) );
649  break;
650  }
651  case constant_relative_bias:
652  {
653  // Check input consistency
654  std::shared_ptr< ConstantObservationBiasSettings > constantBiasSettings = std::dynamic_pointer_cast<
655  ConstantObservationBiasSettings >( biasSettings );
656  if( constantBiasSettings == nullptr )
657  {
658  throw std::runtime_error( "Error when making constant relative observation bias, settings are inconsistent" );
659  }
660 
661  if( constantBiasSettings->useAbsoluteBias_ )
662  {
663  throw std::runtime_error( "Error when making constant relative observation bias, class settings are inconsistent" );
664  }
665 
666  // Check if size of bias is consistent with requested observable size
667  if( constantBiasSettings->observationBias_.rows( ) != ObservationSize )
668  {
669  throw std::runtime_error( "Error when making constant relative observation bias, bias size is inconsistent" );
670  }
671  observationBias = std::make_shared< ConstantRelativeObservationBias< ObservationSize > >(
672  constantBiasSettings->observationBias_ );
673  break;
674  }
675  case arc_wise_constant_relative_bias:
676  {
677  // Check input consistency
678  std::shared_ptr< ArcWiseConstantObservationBiasSettings > arcwiseBiasSettings = std::dynamic_pointer_cast<
679  ArcWiseConstantObservationBiasSettings >( biasSettings );
680  if( arcwiseBiasSettings == nullptr )
681  {
682  throw std::runtime_error( "Error when making arc-wise relative observation bias, settings are inconsistent" );
683  }
684  else if( arcwiseBiasSettings->useAbsoluteBias_ )
685  {
686  throw std::runtime_error( "Error when making arc-wise relative observation bias, class contents are inconsistent" );
687  }
688 
689  std::vector< Eigen::Matrix< double, ObservationSize, 1 > > observationBiases;
690  for( unsigned int i = 0; i < arcwiseBiasSettings->observationBiases_.size( ); i++ )
691  {
692  // Check if size of bias is consistent with requested observable size
693  if( arcwiseBiasSettings->observationBiases_.at( i ).rows( ) != ObservationSize )
694  {
695  throw std::runtime_error( "Error when making arc-wise observation bias, bias size is inconsistent" );
696  }
697  else
698  {
699  observationBiases.push_back( arcwiseBiasSettings->observationBiases_.at( i ) );
700  }
701  }
702  observationBias = std::make_shared< ConstantRelativeArcWiseObservationBias< ObservationSize > >(
703  arcwiseBiasSettings->arcStartTimes_, observationBiases,
704  observation_models::getLinkEndIndicesForLinkEndTypeAtObservable(
705  observableType, arcwiseBiasSettings->linkEndForTime_, linkEnds.size( ) ).at( 0 ) );
706  break;
707  }
708  case multiple_observation_biases:
709  {
710  // Check input consistency
711  std::shared_ptr< MultipleObservationBiasSettings > multiBiasSettings = std::dynamic_pointer_cast<
712  MultipleObservationBiasSettings >( biasSettings );
713  if( multiBiasSettings == nullptr )
714  {
715  throw std::runtime_error( "Error when making multiple observation biases, settings are inconsistent" );
716  }
717 
718  // Create list of biases
719  std::vector< std::shared_ptr< ObservationBias< ObservationSize > > > observationBiasList;
720  for( unsigned int i = 0; i < multiBiasSettings->biasSettingsList_.size( ); i++ )
721  {
722  observationBiasList.push_back( createObservationBiasCalculator< ObservationSize >(
723  linkEnds, observableType, multiBiasSettings->biasSettingsList_.at( i ) , bodyMap ) );
724  }
725 
726  // Create combined bias object
727  observationBias = std::make_shared< MultiTypeObservationBias< ObservationSize > >(
728  observationBiasList );
729  break;
730  }
731  default:
732  {
733  std::string errorMessage = "Error when making observation bias, bias type " +
734  std::to_string( biasSettings->observationBiasType_ ) + " not recognized";
735  throw std::runtime_error( errorMessage );
736  }
737  }
738  return observationBias;
739 }
