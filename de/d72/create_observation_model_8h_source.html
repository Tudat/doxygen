441 template< typename ObservationScalarType = double, typename TimeType = double >
442 boost::shared_ptr< DopplerProperTimeRateInterface > createOneWayDopplerProperTimeCalculator(
443  boost::shared_ptr< DopplerProperTimeRateSettings > properTimeRateSettings,
444  const LinkEnds& linkEnds,
445  const simulation_setup::NamedBodyMap &bodyMap,
446  const LinkEndType linkEndForCalculator )
447 {
448  boost::shared_ptr< DopplerProperTimeRateInterface > properTimeRateInterface;
449 
450  // Check tyope of proper time rate model
451  switch( properTimeRateSettings->dopplerProperTimeRateType_ )
452  {
453  case direct_first_order_doppler_proper_time_rate:
454  {
455  // Check input consistency
456  boost::shared_ptr< DirectFirstOrderDopplerProperTimeRateSettings > directFirstOrderDopplerProperTimeRateSettings =
457  boost::dynamic_pointer_cast< DirectFirstOrderDopplerProperTimeRateSettings >( properTimeRateSettings );
458  if( directFirstOrderDopplerProperTimeRateSettings == NULL )
459  {
460  throw std::runtime_error( "Error when making DopplerProperTimeRateInterface, input type (direct_first_order_doppler_proper_time_rate) is inconsistent" );
461  }
462  else if( linkEnds.count( linkEndForCalculator ) == 0 )
463  {
464  std::string errorMessage = "Error when creating one-way Doppler proper time calculator, did not find link end " +
465  std::to_string( linkEndForCalculator );
466  throw std::runtime_error( errorMessage );
467  }
468  else
469  {
470  if( bodyMap.at( directFirstOrderDopplerProperTimeRateSettings->centralBodyName_ )->getGravityFieldModel( ) == NULL )
471  {
472  throw std::runtime_error( "Error when making DirectFirstOrderDopplerProperTimeRateInterface, no gravity field found for " +
473  directFirstOrderDopplerProperTimeRateSettings->centralBodyName_ );
474  }
475  else
476  {
477  // Retrieve gravitational parameter
478  boost::function< double( ) > gravitationalParameterFunction =
479  boost::bind( &gravitation::GravityFieldModel::getGravitationalParameter,
480  bodyMap.at( directFirstOrderDopplerProperTimeRateSettings->centralBodyName_ )->
481  getGravityFieldModel( ) );
482 
483  // Create calculation object.
484  LinkEndId referencePointId =
485  std::make_pair( directFirstOrderDopplerProperTimeRateSettings->centralBodyName_, "" );
486  if( ( linkEnds.at( receiver ) != referencePointId ) && ( linkEnds.at( transmitter ) != referencePointId ) )
487  {
488  properTimeRateInterface = boost::make_shared<
489  DirectFirstOrderDopplerProperTimeRateInterface >(
490  linkEndForCalculator, gravitationalParameterFunction,
491  directFirstOrderDopplerProperTimeRateSettings->centralBodyName_, unidentified_link_end,
492  getLinkEndCompleteEphemerisFunction< double, double >(
493  std::make_pair( directFirstOrderDopplerProperTimeRateSettings->centralBodyName_, ""), bodyMap ) );
494  }
495  else
496  {
497  throw std::runtime_error(
498  "Error, proper time reference point as link end not yet implemented for DopplerProperTimeRateInterface creation" );
499  }
500  }
501  }
502  break;
503  }
504  default:
505  std::string errorMessage = "Error when creating one-way Doppler proper time calculator, did not recognize type " +
506  std::to_string( properTimeRateSettings->dopplerProperTimeRateType_ );
507  throw std::runtime_error( errorMessage );
508  }
509  return properTimeRateInterface;
510 }

540 template< int ObservationSize = 1 >
541 boost::shared_ptr< ObservationBias< ObservationSize > > createObservationBiasCalculator(
542  const LinkEnds linkEnds,
543  const boost::shared_ptr< ObservationBiasSettings > biasSettings,
544  const simulation_setup::NamedBodyMap &bodyMap )
545 {
546  boost::shared_ptr< ObservationBias< ObservationSize > > observationBias;
547  switch( biasSettings->observationBiasType_ )
548  {
549  case constant_absolute_bias:
550  {
551  // Check input consistency
552  boost::shared_ptr< ConstantObservationBiasSettings > constantBiasSettings = boost::dynamic_pointer_cast<
553  ConstantObservationBiasSettings >( biasSettings );
554  if( constantBiasSettings == NULL )
555  {
556  throw std::runtime_error( "Error when making constant observation bias, settings are inconsistent" );
557  }
558 
559  // Check if size of bias is consistent with requested observable size
560  if( constantBiasSettings->observationBias_.rows( ) != ObservationSize )
561  {
562  throw std::runtime_error( "Error when making constant observation bias, bias size is inconsistent" );
563  }
564  observationBias = boost::make_shared< ConstantObservationBias< ObservationSize > >(
565  constantBiasSettings->observationBias_ );
566  break;
567  }
568  case constant_relative_bias:
569  {
570  // Check input consistency
571  boost::shared_ptr< ConstantRelativeObservationBiasSettings > constantBiasSettings = boost::dynamic_pointer_cast<
572  ConstantRelativeObservationBiasSettings >( biasSettings );
573  if( constantBiasSettings == NULL )
574  {
575  throw std::runtime_error( "Error when making constant relative observation bias, settings are inconsistent" );
576  }
577 
578  // Check if size of bias is consistent with requested observable size
579  if( constantBiasSettings->relativeObservationBias_.rows( ) != ObservationSize )
580  {
581  throw std::runtime_error( "Error when making constant relative observation bias, bias size is inconsistent" );
582  }
583  observationBias = boost::make_shared< ConstantRelativeObservationBias< ObservationSize > >(
584  constantBiasSettings->relativeObservationBias_ );
585  break;
586  }
587  case multiple_observation_biases:
588  {
589  // Check input consistency
590  boost::shared_ptr< MultipleObservationBiasSettings > multiBiasSettings = boost::dynamic_pointer_cast<
591  MultipleObservationBiasSettings >( biasSettings );
592  if( multiBiasSettings == NULL )
593  {
594  throw std::runtime_error( "Error when making multiple observation biases, settings are inconsistent" );
595  }
596 
597  // Create list of biases
598  std::vector< boost::shared_ptr< ObservationBias< ObservationSize > > > observationBiasList;
599  for( unsigned int i = 0; i < multiBiasSettings->biasSettingsList_.size( ); i++ )
600  {
601  observationBiasList.push_back( createObservationBiasCalculator< ObservationSize >(
602  linkEnds, multiBiasSettings->biasSettingsList_.at( i ) , bodyMap ) );
603  }
604 
605  // Create combined bias object
606  observationBias = boost::make_shared< MultiTypeObservationBias< ObservationSize > >(
607  observationBiasList );
608  break;
609  }
610  default:
611  {
612  std::string errorMessage = "Error when making observation bias, bias type " +
613  std::to_string( biasSettings->observationBiasType_ ) + " not recognized";
614  throw std::runtime_error( errorMessage );
615  }
616  }
617  return observationBias;
618 }
