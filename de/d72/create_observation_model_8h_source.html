405 template< typename ObservationScalarType = double, typename TimeType = double >
406 boost::shared_ptr< DopplerProperTimeRateInterface > createOneWayDopplerProperTimeCalculator(
407  boost::shared_ptr< DopplerProperTimeRateSettings > properTimeRateSettings,
408  const LinkEnds& linkEnds,
409  const simulation_setup::NamedBodyMap &bodyMap,
410  const LinkEndType linkEndForCalculator )
411 {
412  boost::shared_ptr< DopplerProperTimeRateInterface > properTimeRateInterface;
413 
414  // Check tyope of proper time rate model
415  switch( properTimeRateSettings->dopplerProperTimeRateType_ )
416  {
417  case direct_first_order_doppler_proper_time_rate:
418  {
419  // Check input consistency
420  boost::shared_ptr< DirectFirstOrderDopplerProperTimeRateSettings > directFirstOrderDopplerProperTimeRateSettings =
421  boost::dynamic_pointer_cast< DirectFirstOrderDopplerProperTimeRateSettings >( properTimeRateSettings );
422  if( directFirstOrderDopplerProperTimeRateSettings == NULL )
423  {
424  throw std::runtime_error( "Error when making DopplerProperTimeRateInterface, input type (direct_first_order_doppler_proper_time_rate) is inconsistent" );
425  }
426  else if( linkEnds.count( linkEndForCalculator ) == 0 )
427  {
428  std::string errorMessage = "Error when creating one-way Doppler proper time calculator, did not find link end " +
429  boost::lexical_cast< std::string >( linkEndForCalculator );
430  throw std::runtime_error( errorMessage );
431  }
432  else
433  {
434  if( bodyMap.at( directFirstOrderDopplerProperTimeRateSettings->centralBodyName_ )->getGravityFieldModel( ) == NULL )
435  {
436  throw std::runtime_error( "Error when making DirectFirstOrderDopplerProperTimeRateInterface, no gravity field found for " +
437  directFirstOrderDopplerProperTimeRateSettings->centralBodyName_ );
438  }
439  else
440  {
441  // Retrieve gravitational parameter
442  boost::function< double( ) > gravitationalParameterFunction =
443  boost::bind( &gravitation::GravityFieldModel::getGravitationalParameter,
444  bodyMap.at( directFirstOrderDopplerProperTimeRateSettings->centralBodyName_ )->
445  getGravityFieldModel( ) );
446 
447  // Create calculation object.
448  LinkEndId referencePointId =
449  std::make_pair( directFirstOrderDopplerProperTimeRateSettings->centralBodyName_, "" );
450  if( ( linkEnds.at( receiver ) != referencePointId ) && ( linkEnds.at( transmitter ) != referencePointId ) )
451  {
452  properTimeRateInterface = boost::make_shared<
453  DirectFirstOrderDopplerProperTimeRateInterface >(
454  linkEndForCalculator, gravitationalParameterFunction,
455  directFirstOrderDopplerProperTimeRateSettings->centralBodyName_, unidentified_link_end,
456  getLinkEndCompleteEphemerisFunction< double, double >(
457  std::make_pair( directFirstOrderDopplerProperTimeRateSettings->centralBodyName_, ""), bodyMap ) );
458  }
459  else
460  {
461  throw std::runtime_error(
462  "Error, proper time reference point as link end not yet implemented for DopplerProperTimeRateInterface creation" );
463  }
464  }
465  }
466  break;
467  }
468  default:
469  std::string errorMessage = "Error when creating one-way Doppler proper time calculator, did not recognize type " +
470  boost::lexical_cast< std::string >( properTimeRateSettings->dopplerProperTimeRateType_ );
471  throw std::runtime_error( errorMessage );
472  }
473  return properTimeRateInterface;
474 }

504 template< int ObservationSize = 1 >
505 boost::shared_ptr< ObservationBias< ObservationSize > > createObservationBiasCalculator(
506  const LinkEnds linkEnds,
507  const boost::shared_ptr< ObservationBiasSettings > biasSettings,
508  const simulation_setup::NamedBodyMap &bodyMap )
509 {
510  boost::shared_ptr< ObservationBias< ObservationSize > > observationBias;
511  switch( biasSettings->observationBiasType_ )
512  {
513  case constant_absolute_bias:
514  {
515  // Check input consistency
516  boost::shared_ptr< ConstantObservationBiasSettings > constantBiasSettings = boost::dynamic_pointer_cast<
517  ConstantObservationBiasSettings >( biasSettings );
518  if( constantBiasSettings == NULL )
519  {
520  throw std::runtime_error( "Error when making constant observation bias, settings are inconsistent" );
521  }
522 
523  // Check if size of bias is consistent with requested observable size
524  if( constantBiasSettings->observationBias_.rows( ) != ObservationSize )
525  {
526  throw std::runtime_error( "Error when making constant observation bias, bias size is inconsistent" );
527  }
528  observationBias = boost::make_shared< ConstantObservationBias< ObservationSize > >(
529  constantBiasSettings->observationBias_ );
530  break;
531  }
532  case constant_relative_bias:
533  {
534  // Check input consistency
535  boost::shared_ptr< ConstantRelativeObservationBiasSettings > constantBiasSettings = boost::dynamic_pointer_cast<
536  ConstantRelativeObservationBiasSettings >( biasSettings );
537  if( constantBiasSettings == NULL )
538  {
539  throw std::runtime_error( "Error when making constant relative observation bias, settings are inconsistent" );
540  }
541 
542  // Check if size of bias is consistent with requested observable size
543  if( constantBiasSettings->relativeObservationBias_.rows( ) != ObservationSize )
544  {
545  throw std::runtime_error( "Error when making constant relative observation bias, bias size is inconsistent" );
546  }
547  observationBias = boost::make_shared< ConstantRelativeObservationBias< ObservationSize > >(
548  constantBiasSettings->relativeObservationBias_ );
549  break;
550  }
551  case multiple_observation_biases:
552  {
553  // Check input consistency
554  boost::shared_ptr< MultipleObservationBiasSettings > multiBiasSettings = boost::dynamic_pointer_cast<
555  MultipleObservationBiasSettings >( biasSettings );
556  if( multiBiasSettings == NULL )
557  {
558  throw std::runtime_error( "Error when making multiple observation biases, settings are inconsistent" );
559  }
560 
561  // Create list of biases
562  std::vector< boost::shared_ptr< ObservationBias< ObservationSize > > > observationBiasList;
563  for( unsigned int i = 0; i < multiBiasSettings->biasSettingsList_.size( ); i++ )
564  {
565  observationBiasList.push_back( createObservationBiasCalculator< ObservationSize >(
566  linkEnds, multiBiasSettings->biasSettingsList_.at( i ) , bodyMap ) );
567  }
568 
569  // Create combined bias object
570  observationBias = boost::make_shared< MultiTypeObservationBias< ObservationSize > >(
571  observationBiasList );
572  break;
573  }
574  default:
575  {
576  std::string errorMessage = "Error when making observation bias, bias type " +
577  boost::lexical_cast< std::string >( biasSettings->observationBiasType_ ) + " not recognized";
578  throw std::runtime_error( errorMessage );
579  }
580  }
581  return observationBias;
582 }
