191 template< typename KeyType, typename ScalarType, int NumberOfRows, int NumberOfColumns = 1 >
192 Eigen::Matrix< ScalarType, Eigen::Dynamic, NumberOfColumns > createConcatenatedEigenMatrixFromMapValues(
193  const std::map< KeyType, Eigen::Matrix< ScalarType, NumberOfRows, NumberOfColumns > >& inputMap )
194 {
195  // Create and size return vector.
196  Eigen::Matrix< ScalarType, Eigen::Dynamic, NumberOfColumns > outputVector;
197 
198  int columns;
199  if( NumberOfColumns != Eigen::Dynamic )
200  {
201  columns = NumberOfColumns;
202  }
203  else
204  {
205  columns = inputMap.begin( )->second.cols( );
206  }
207 
208  if( NumberOfRows != Eigen::Dynamic )
209  {
210  int counter = 0;
211 
212  outputVector.resize( inputMap.size( ) * NumberOfRows, columns );
213  for( typename std::map< KeyType, Eigen::Matrix< ScalarType, NumberOfRows, NumberOfColumns > >::const_iterator mapIterator =
214  inputMap.begin( ); mapIterator != inputMap.end( ); mapIterator++ )
215  {
216  outputVector.block( counter, 0, NumberOfRows, columns ) = mapIterator->second;
217  counter += NumberOfRows;
218  }
219  }
220  else
221  {
222  int concatenatedSize = 0;
223 
224  for( typename std::map< KeyType, Eigen::Matrix< ScalarType, NumberOfRows, NumberOfColumns > >::const_iterator mapIterator =
225  inputMap.begin( ); mapIterator != inputMap.end( ); mapIterator++ )
226  {
227  concatenatedSize += mapIterator->second.rows( );
228  }
229 
230  outputVector.resize( concatenatedSize, columns );
231  int currentSize;
232  int counter = 0;
233 
234  for( typename std::map< KeyType, Eigen::Matrix< ScalarType, NumberOfRows, NumberOfColumns > >::const_iterator mapIterator =
235  inputMap.begin( ); mapIterator != inputMap.end( ); mapIterator++ )
236  {
237  currentSize = mapIterator->second.rows( );
238  outputVector.block( counter, 0, currentSize, columns ) = mapIterator->second;
239  counter += currentSize;
240  }
241  }
242 
243  return outputVector;
244 }
