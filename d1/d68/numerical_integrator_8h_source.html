189 template < typename IndependentVariableType, typename StateType, typename StateDerivativeType, typename TimeStepType >
190 StateType NumericalIntegrator< IndependentVariableType, StateType, StateDerivativeType, TimeStepType >::integrateTo(
191  const IndependentVariableType intervalEnd,
192  const TimeStepType initialStepSize,
193  const TimeStepType finalTimeTolerance )
194 {
195  TimeStepType stepSize = initialStepSize;
196 
197  // Flag to indicate that the integration end value of the independent variable has been
198  // reached.
199  bool atIntegrationIntervalEnd = static_cast< TimeStepType >( intervalEnd - getCurrentIndependentVariable( ) )
200  * stepSize / std::fabs( stepSize )
201  <= finalTimeTolerance;
202 
203  int loopCounter = 0;
204  while ( !atIntegrationIntervalEnd )
205  {
206  // Check if the remaining interval is smaller than the step size.
207  if ( std::fabs( static_cast< TimeStepType >( intervalEnd - getCurrentIndependentVariable( ) ) )
208  <= std::fabs( stepSize ) *
209  ( 1.0 + finalTimeTolerance ) )
210  {
211  // The next step is beyond the end of the integration interval, so adjust the
212  // step size accordingly.
213  stepSize = intervalEnd - getCurrentIndependentVariable( );
214 
215  // Explicitly flag that the integration interval end is reached. Due to rounding
216  // off errors, it may not be possible to use
217  // ( currentIndependentVariable >= independentVariableEnd ) // in the while condition.
218  atIntegrationIntervalEnd = true;
219 
220  }
221 
222  // Perform the step.
223  performIntegrationStep( stepSize );
224  stepSize = getNextStepSize( );
225 
226  // Only applicable to adaptive step size methods:
227  // Perform additional step(s) to reach intervalEnd exactly, in case the last step was rejected by
228  // the variable step size routine and a new step was used that was too small to get to intervalEnd.
229  if ( atIntegrationIntervalEnd )
230  {
231  // As long as intervalEnd is not reached, perform additional steps with the remaining time
232  // as suggested step size for the variable step size routine.
233  if( std::fabs( static_cast< TimeStepType >( intervalEnd - getCurrentIndependentVariable( ) ) ) >
234  finalTimeTolerance )
235  {
236  // Ensure that integrateTo function does not get stuck in a loop.
237  if( loopCounter < 1000 )
238  {
239  atIntegrationIntervalEnd = false;
240  loopCounter++;
241  }
242  else
243  {
244  std::cerr<<"Warning, integrateTo function has failed to converge to final time to within tolerances, difference between true and requested final time is "<<
245  intervalEnd - getCurrentIndependentVariable( ) <<", final time is: "<<
246  getCurrentIndependentVariable( )<<std::endl;
247  }
248  }
249  }
250  }
251 
252  return getCurrentState( );
253 }
