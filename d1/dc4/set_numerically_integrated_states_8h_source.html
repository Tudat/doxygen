158 template< typename TimeType, typename StateScalarType >
159 std::map< TimeType, Eigen::Matrix< StateScalarType, 6, 1 > > convertNumericalSolutionToEphemerisInput(
160  const int bodyIndex,
161  const int startIndex,
162  const std::map< TimeType, Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > >&
163  equationsOfMotionNumericalSolution,
164  const boost::function< Eigen::Matrix< StateScalarType, 6, 1 >( const TimeType ) >
165  integrationToEphemerisFrameFunction = NULL )
166 {
167  std::map< TimeType, Eigen::Matrix< StateScalarType, 6, 1 > > ephemerisTable;
168 
169  // If no integrationToEphemerisFrameFunction is provided, origin is already correct; only
170  // extract required indices.
171  if( integrationToEphemerisFrameFunction == 0 )
172  {
173  for( typename std::map< TimeType, Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > >::const_iterator
174  bodyIterator = equationsOfMotionNumericalSolution.begin( );
175  bodyIterator != equationsOfMotionNumericalSolution.end( ); bodyIterator++ )
176  {
177  ephemerisTable[ bodyIterator->first ] = bodyIterator->second.block( startIndex + 6 * bodyIndex, 0, 6, 1 );
178  }
179  }
180  // Else, extract indices and add required translation from integrationToEphemerisFrameFunction
181  else
182  {
183  for( typename std::map< TimeType, Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > >::const_iterator
184  bodyIterator = equationsOfMotionNumericalSolution.begin( );
185  bodyIterator != equationsOfMotionNumericalSolution.end( ); bodyIterator++ )
186  {
187 
188  ephemerisTable[ bodyIterator->first ] = bodyIterator->second.block( startIndex + 6 * bodyIndex, 0, 6, 1 ) -
189  integrationToEphemerisFrameFunction( bodyIterator->first );
190  }
191  }
192  return ephemerisTable;
193 }

288 template< typename TimeType, typename StateScalarType >
289 void resetIntegratedEphemerides(
290  const simulation_setup::NamedBodyMap& bodyMap,
291  const std::map< TimeType, Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > >& equationsOfMotionNumericalSolution,
292  const std::vector< std::string >& bodiesToIntegrate,
293  const std::pair< unsigned int, unsigned int > startIndexAndSize,
294  std::vector< std::string > ephemerisUpdateOrder = std::vector< std::string >( ),
295  const std::map< std::string, boost::function< Eigen::Matrix< StateScalarType, 6, 1 >( const TimeType ) > >&
296  integrationToEphemerisFrameFunctions =
297  std::map< std::string, boost::function< Eigen::Matrix< StateScalarType, 6, 1 >( const TimeType ) > >( ) )
298 {
299  // Set update order arbitrarily if no order is provided.
300  if( ephemerisUpdateOrder.size( ) == 0 )
301  {
302  ephemerisUpdateOrder = bodiesToIntegrate;
303  }
304  // Check input consistency
305  else if( ephemerisUpdateOrder.size( ) != bodiesToIntegrate.size( ) )
306  {
307  throw std::runtime_error( "Error when resetting ephemerides, input vectors have inconsistent size" );
308  }
309 
310  if( static_cast< unsigned int >( equationsOfMotionNumericalSolution.begin( )->second.rows( ) )
311  < startIndexAndSize.first + startIndexAndSize.second )
312  {
313  throw std::runtime_error( "Error when resetting ephemerides, input solution inconsistent with start index and size." );
314  }
315 
316  if( startIndexAndSize.second != 6 * bodiesToIntegrate.size( ) )
317  {
318  throw std::runtime_error( "Error when resetting ephemerides, number of bodies inconsistent with input size." );
319 
320  }
321 
322  // Create interpolators from numerical integration results (states) at discrete times.
323  createAndSetInterpolatorsForEphemerides(
324  bodyMap, bodiesToIntegrate, startIndexAndSize.first, ephemerisUpdateOrder,
325  equationsOfMotionNumericalSolution, integrationToEphemerisFrameFunctions );
326 }

344 template< typename TimeType, typename StateScalarType >
345 void resetMultiArcIntegratedEphemerides(
346  const simulation_setup::NamedBodyMap& bodyMap,
347  const std::vector< std::map< TimeType, Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > > >&
348  equationsOfMotionNumericalSolution,
349  const std::vector< double > arcStartTimes,
350  const std::vector< std::string >& bodiesToIntegrate,
351  const std::pair< unsigned int, unsigned int > startIndexAndSize,
352  std::vector< std::string > ephemerisUpdateOrder = std::vector< std::string >( ),
353  const std::map< std::string, boost::function< Eigen::Matrix< StateScalarType, 6, 1 >( const TimeType ) > >&
354  integrationToEphemerisFrameFunctions =

356 {
357  using namespace tudat::interpolators;
358  using namespace tudat::ephemerides;
359 
360  // Check input consistency
361  if( ephemerisUpdateOrder.size( ) == 0 )
362  {
363  ephemerisUpdateOrder = bodiesToIntegrate;
364  }
365 
366  // Update ephemerides
367  for( unsigned int i = 0; i < ephemerisUpdateOrder.size( ); i++ )
368  {
369  // Find index of current body in bodiesToIntegrate.
370  std::vector< std::string >::const_iterator bodyFindIterator = std::find(
371  bodiesToIntegrate.begin( ), bodiesToIntegrate.end( ), ephemerisUpdateOrder.at( i ) );
372  int bodyIndex = std::distance( bodiesToIntegrate.begin( ), bodyFindIterator );
373 
374 
375  // Check environment consistency

378  bodyMap.at( bodiesToIntegrate.at( bodyIndex ) )->getEphemeris( ) );
379  if( currentBodyEphemeris == NULL )

381  throw std::runtime_error( "Error when resetting ephemeris of body " + bodiesToIntegrate.at( bodyIndex ) +
382  ", original ephemeris is of incompatible type" );
383  }
384 
385  std::vector< boost::shared_ptr< Ephemeris > > arcEphemerisList;
386  for( unsigned int j = 0; j < arcStartTimes.size( ); j++ )
387  {
388  boost::function< Eigen::Matrix< StateScalarType, 6, 1 >( const TimeType ) >
389  integrationToEphemerisFrameFunction = NULL;
390 
391  // Create transformation function, if needed.
392  if( integrationToEphemerisFrameFunctions.count( bodiesToIntegrate.at( bodyIndex ) ) > 0 )
393  {
394  integrationToEphemerisFrameFunction =
395  integrationToEphemerisFrameFunctions.at( bodiesToIntegrate.at( bodyIndex ) );
396  }
397 
398  // Create interpolator.
399  boost::shared_ptr< OneDimensionalInterpolator< TimeType, Eigen::Matrix< StateScalarType, 6, 1 > > >
400  ephemerisInterpolator = createStateInterpolator(
401  convertNumericalSolutionToEphemerisInput(
402  bodyIndex, startIndexAndSize.first,
403  equationsOfMotionNumericalSolution.at( j ), integrationToEphemerisFrameFunction ) );
404 
405  arcEphemerisList.push_back( boost::make_shared< TabulatedCartesianEphemeris< StateScalarType, TimeType > >(
406  ephemerisInterpolator, currentBodyEphemeris->getReferenceFrameOrigin( ),
407  currentBodyEphemeris->getReferenceFrameOrientation( ) ) );
408 
409  }
410  currentBodyEphemeris->resetSingleArcEphemerides( arcEphemerisList, arcStartTimes );
411  }
412 
413  // Having set new ephemerides, update body properties depending on ephemerides.
414  for( simulation_setup::NamedBodyMap::const_iterator bodyIterator = bodyMap.begin( );
415  bodyIterator != bodyMap.end( ); bodyIterator++ )
416  {
417  bodyIterator->second->updateConstantEphemerisDependentMemberQuantities( );
418  }

901 
912 std::map< IntegratedStateType,
913 std::vector< boost::shared_ptr< IntegratedStateProcessor< TimeType, StateScalarType > > > >
914 createIntegratedStateProcessors(
915  const boost::shared_ptr< SingleArcPropagatorSettings< StateScalarType > > propagatorSettings,
916  const simulation_setup::NamedBodyMap& bodyMap,
917  const boost::shared_ptr< ephemerides::ReferenceFrameManager > frameManager,
918  const int startIndex = 0 )
919 {
920  std::map< IntegratedStateType, std::vector< boost::shared_ptr< IntegratedStateProcessor
921  < TimeType, StateScalarType > > > > integratedStateProcessors;
922 
923  // Check dynamics type.
924  switch( propagatorSettings->getStateType( ) )
925  {
926  case hybrid:
927  {
928  // Check input consistency
929  boost::shared_ptr< MultiTypePropagatorSettings< StateScalarType > > multiTypePropagatorSettings =
930  boost::dynamic_pointer_cast< MultiTypePropagatorSettings< StateScalarType > >( propagatorSettings );
931  if( multiTypePropagatorSettings == NULL )
932  {
933  throw std::runtime_error( "Error, multi-type propagator settings are inconsistent when make state processors" );
934  }
935 
936  // Iterate over each propagated state type
937  std::map< IntegratedStateType, std::vector< boost::shared_ptr< IntegratedStateProcessor< TimeType, StateScalarType > > > >
938  singleTypeIntegratedStateProcessors;
939  int currentStartIndex = 0;
940  for( typename std::map< IntegratedStateType,
941  std::vector< boost::shared_ptr< SingleArcPropagatorSettings< StateScalarType > > > >::const_iterator
942  typeIterator = multiTypePropagatorSettings->propagatorSettingsMap_.begin( );
943  typeIterator != multiTypePropagatorSettings->propagatorSettingsMap_.end( ); typeIterator++ )
944  {
945  // Multi-type in multi-type not allowed (yet)
946  if( typeIterator->first != hybrid )
947  {
948  for( unsigned int i = 0; i < typeIterator->second.size( ); i++ )
949  {
950  if( typeIterator->second.at( i ) == NULL )
951  {
952  std::string errorMessage = "Error in when processing hybrid propagator settings, propagator entry " +
953  boost::lexical_cast< std::string >( i ) + " is not defined.";
954  throw std::runtime_error( errorMessage );
955  }
956 
957  // Create state processor
958  singleTypeIntegratedStateProcessors = createIntegratedStateProcessors< TimeType, StateScalarType >(
959  typeIterator->second.at( i ), bodyMap, frameManager, currentStartIndex );
960 
961  if( singleTypeIntegratedStateProcessors.size( ) > 1 )
962  {
963  throw std::runtime_error( "Error when making hybrid integrated result processors, multiple types found" );
964  }
965  else if( ( singleTypeIntegratedStateProcessors.size( ) == 0 ) && ( typeIterator->first != custom_state ) )
966  {
967  throw std::runtime_error( "Error when making hybrid integrated result processors, no types found" );
968  }
969  else if( ( singleTypeIntegratedStateProcessors.size( ) == 1 ) )
970  {
971  if( singleTypeIntegratedStateProcessors.begin( )->second.size( ) != 1 )
972  {
973  throw std::runtime_error(
974  "Error when making hybrid integrated result processors, multiple processors of single type found" );
975  }
976  else
977  {
978  integratedStateProcessors[ singleTypeIntegratedStateProcessors.begin( )->first ].push_back(
979  singleTypeIntegratedStateProcessors.begin( )->second.at( 0 ) );
980  }
981  }
982 
983  currentStartIndex += typeIterator->second.at( i )->getStateSize( );
984 
985  }
986  }
987  else
988  {
989  throw std::runtime_error(
990  "Error when making integrated state processors, cannot handle hybrid propagator inside hybrid propagator" );
991  }
992 
993  }
994 
995  break;
996  }
997  case transational_state:
998  {
999 
1000  // Check input feasibility
1001  boost::shared_ptr< TranslationalStatePropagatorSettings< StateScalarType > >
1002  translationalPropagatorSettings = boost::dynamic_pointer_cast
1003  < TranslationalStatePropagatorSettings< StateScalarType > >( propagatorSettings );
1004  if( translationalPropagatorSettings == NULL )
1005  {
1006  throw std::runtime_error( "Error, input type is inconsistent in createIntegratedStateProcessors" );
1007  }
1008  checkTranslationalStatesFeasibility(
1009  translationalPropagatorSettings->bodiesToIntegrate_, bodyMap );
1010 
1011  // Create state processors
1012  integratedStateProcessors[ transational_state ].push_back(
1013  boost::make_shared< TranslationalStateIntegratedStateProcessor< TimeType, StateScalarType > >(
1014  startIndex, bodyMap, translationalPropagatorSettings->bodiesToIntegrate_,
1015  translationalPropagatorSettings->centralBodies_, frameManager ) );
1016  break;
1017  }
1018  case rotational_state:
1019  {
1020  boost::shared_ptr< RotationalStatePropagatorSettings< StateScalarType > > rotationalPropagatorSettings =
1021  boost::dynamic_pointer_cast< RotationalStatePropagatorSettings< StateScalarType > >( propagatorSettings );
1022 
1023  integratedStateProcessors[ rotational_state ].push_back(
1024  boost::make_shared< RotationalStateIntegratedStateProcessor< TimeType, StateScalarType > >(
1025  startIndex, bodyMap, rotationalPropagatorSettings->bodiesToIntegrate_ ) );
1026  break;
1027  }
1028  case body_mass_state:
1029  {
1030 
1031  // Check input feasibility
1032  boost::shared_ptr< MassPropagatorSettings< StateScalarType > >
1033  massPropagatorSettings = boost::dynamic_pointer_cast
1034  < MassPropagatorSettings< StateScalarType > >( propagatorSettings );
1035  if( massPropagatorSettings == NULL )
1036  {
1037  throw std::runtime_error( "Error, input type is inconsistent in createIntegratedStateProcessors" );
1038  }
1039 
1040  // Create mass processors.
1041  integratedStateProcessors[ body_mass_state ].push_back(
1042  boost::make_shared< BodyMassIntegratedStateProcessor< TimeType, StateScalarType > >(
1043  startIndex, bodyMap, massPropagatorSettings->bodiesWithMassToPropagate_ ) );
1044  break;
1045  }
1046  case custom_state:
1047  {
1048  break;
1049  }
1050  default:
1051  throw std::runtime_error( "Error, could not process integrated state type " +
1052  boost::lexical_cast< std::string >( propagatorSettings->getStateType( ) ) );
1053  }
1054 
1055  return integratedStateProcessors;
1056 }
