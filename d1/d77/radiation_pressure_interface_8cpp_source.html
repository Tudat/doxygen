32 void RadiationPressureInterface::updateInterface(
33  const double currentTime )
34 {
35  currentTime_ = currentTime;
36 
37  // Calculate current radiation pressure
38  currentSolarVector_ = sourcePositionFunction_( ) - targetPositionFunction_( );
39  double distanceFromSource = currentSolarVector_.norm( );
40  currentRadiationPressure_ = calculateRadiationPressure(
41  sourcePower_( ), distanceFromSource );
42 
43  // Calculate total shadowing due to occulting body; note that multiple concurrent
44  // occultations are not completely correctly (prints warning).
45  double shadowFunction = 1.0;
46  double currentShadowFunction = 1.0;
47  for( unsigned int i = 0; i < occultingBodyPositions_.size( ); i++ )
48  {
49  currentShadowFunction *= mission_geometry::computeShadowFunction(
50  sourcePositionFunction_( ), sourceRadius_, occultingBodyPositions_[ i ]( ),
51  occultingBodyRadii_[ i ], targetPositionFunction_( ) );
52 
53  if( currentShadowFunction != 1.0 && shadowFunction != 1.0 )
54  {
55  std::cerr << "Warning, multiple occultation occured in radiation pressure interface, results may be slightly in error" << std::endl;
56  }
57 
58  shadowFunction *= currentShadowFunction;
59  }
60 
61  currentRadiationPressure_ *= shadowFunction;
62 
63  radiationPressureCoefficient_ = radiationPressureCoefficientFunction_( currentTime );
64 }
