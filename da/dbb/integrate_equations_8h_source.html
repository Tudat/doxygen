52 template< typename StateType = Eigen::MatrixXd, typename TimeType = double, typename TimeStepType = TimeType >
53 PropagationTerminationReason integrateEquationsFromIntegrator(
54  const boost::shared_ptr< numerical_integrators::NumericalIntegrator< TimeType, StateType, StateType, TimeStepType > > integrator,
55  const TimeStepType initialTimeStep,
56  const boost::function< bool( const double ) > stopPropagationFunction,
57  std::map< TimeType, StateType >& solutionHistory,
58  std::map< TimeType, Eigen::VectorXd >& dependentVariableHistory,
59  const boost::function< Eigen::VectorXd( ) > dependentVariableFunction =
60  boost::function< Eigen::VectorXd( ) >( ),
61  const int saveFrequency = TUDAT_NAN,
62  const TimeType printInterval = TUDAT_NAN )
63 {
64  PropagationTerminationReason propagationTerminationReason;
65 
66  // Get Initial state and time.
67  TimeType currentTime = integrator->getCurrentIndependentVariable( );
68  TimeType initialTime = currentTime;
69  StateType newState = integrator->getCurrentState( );
70 
71  // Initialization of numerical solutions for variational equations
72  solutionHistory.clear( );
73  solutionHistory[ currentTime ] = newState;
74  dependentVariableHistory.clear( );
75 
76 
77  if( !dependentVariableFunction.empty( ) )
78  {
79  integrator->getStateDerivativeFunction( )( currentTime, newState );
80  dependentVariableHistory[ currentTime ] = dependentVariableFunction( );
81  }
82 
83  // Set initial time step and total integration time.
84  TimeStepType timeStep = initialTimeStep;
85  TimeType previousTime = currentTime;
86 
87  int saveIndex = 0;
88 
89  propagationTerminationReason = unknown_propagation_termination_reason;
90  bool breakPropagation = 0;
91  // Perform numerical integration steps until end time reached.
92  do
93  {
94  try
95  {
96  previousTime = currentTime;
97 
98  // Perform integration step.
99  newState = integrator->performIntegrationStep( timeStep );
100 
101  // Check if the termination condition was reached during evaluation of integration sub-steps.
102  // If evaluation of the termination condition during integration sub-steps is disabled,
103  // this function returns always `false`.
104  // If the termination condition was reached, the last step could not be computed correctly because some
105  // of the integrator sub-steps were not computed. Thus, return immediately without saving the `newState`.
106  if ( integrator->getPropagationTerminationConditionReached() )
107  {
108  propagationTerminationReason = termination_condition_reached;
109  break;
110  }
111 
112  // Update epoch and step-size
113  currentTime = integrator->getCurrentIndependentVariable( );
114  timeStep = integrator->getNextStepSize( );
115 
116  // Save integration result in map
117  saveIndex++;
118  saveIndex = saveIndex % saveFrequency;
119  if( saveIndex == 0 )
120  {
121  solutionHistory[ currentTime ] = newState;
122 
123  if( !dependentVariableFunction.empty( ) )
124  {
125  integrator->getStateDerivativeFunction( )( currentTime, newState );
126  dependentVariableHistory[ currentTime ] = dependentVariableFunction( );
127  }
128  }
129 
130 
131  // Print solutions
132  if( printInterval == printInterval )
133  {
134  if( ( static_cast<int>( std::fabs( static_cast< double >( currentTime - initialTime ) ) ) %
135  static_cast< int >( printInterval ) ) <=
136  ( static_cast< int >( std::fabs( static_cast< double >( previousTime - initialTime ) ) ) %
137  static_cast<int>( printInterval ) ) )
138  {
139  std::cout<<"Current time and state in integration: "<<std::setprecision( 10 )<<
140  timeStep<<" "<<currentTime<<" "<<newState.transpose( )<<std::endl;
141  }
142  }
143 
144  if( stopPropagationFunction( static_cast< double >( currentTime ) ) )
145  {
146  propagationTerminationReason = termination_condition_reached;
147  breakPropagation = true;
148  }
149 
150  }
151  catch( const std::exception &caughtException )
152  {
153  std::cerr << caughtException.what( )<<std::endl;
154  std::cerr<<"Error, propagation terminated at t=" + boost::lexical_cast< std::string >( currentTime ) +
155  ", returning propagation data up to current time"<<std::endl;
156  breakPropagation = 1;
157  propagationTerminationReason = runtime_error_caught_in_propagation;
158  }
159  }
160  while( !breakPropagation );
161 
162  return propagationTerminationReason;
163 }
