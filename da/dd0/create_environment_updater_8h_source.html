83 template< typename StateScalarType >
84 std::map< propagators::EnvironmentModelsToUpdate,
85  std::vector< std::string > > createEnvironmentUpdaterSettings(
86  const boost::shared_ptr< SingleArcPropagatorSettings< StateScalarType > > propagatorSettings,
87  const simulation_setup::NamedBodyMap& bodyMap )
88 {
89  std::map< propagators::EnvironmentModelsToUpdate,
90  std::vector< std::string > > environmentModelsToUpdate;
91 
92  // Check dynamics type
93  switch( propagatorSettings->getStateType( ) )
94  {
95  case hybrid:
96  {
97  // Cast to derived type
98  boost::shared_ptr< MultiTypePropagatorSettings< StateScalarType > > multiTypePropagatorSettings =
99  boost::dynamic_pointer_cast< MultiTypePropagatorSettings< StateScalarType > >( propagatorSettings );
100 
101  // Iterate over all propagation settings in hybrid model
102  std::map< propagators::EnvironmentModelsToUpdate, std::vector< std::string > > singleAccelerationUpdateNeeds;
103 
104  for( typename std::map< IntegratedStateType,
105  std::vector< boost::shared_ptr< SingleArcPropagatorSettings< StateScalarType > > > >::const_iterator
106  typeIterator = multiTypePropagatorSettings->propagatorSettingsMap_.begin( );
107  typeIterator != multiTypePropagatorSettings->propagatorSettingsMap_.end( ); typeIterator++ )
108  {
109  for( unsigned int i = 0; i < typeIterator->second.size( ); i++ )
110  {
111  // Create current propagation model from settings list (must not be hybrid).
112  if( typeIterator->first != hybrid )
113  {
114  singleAccelerationUpdateNeeds = createEnvironmentUpdaterSettings< StateScalarType >(
115  typeIterator->second.at( i ), bodyMap );
116 
117  // Add single model environment model update settings to full list
118  checkValidityOfRequiredEnvironmentUpdates( singleAccelerationUpdateNeeds, bodyMap );
119  addEnvironmentUpdates( environmentModelsToUpdate, singleAccelerationUpdateNeeds );
120  }
121  else
122  {
123  throw std::runtime_error(
124  "Error when making environment updater type list, cannot handle hybrid propagator inside hybrid propagator" );
125  }
126  }
127  }
128  break;
129  }
130  // Retrieve environment model settings for translational dynamics
131  case transational_state:
132  {
133  environmentModelsToUpdate = createTranslationalEquationsOfMotionEnvironmentUpdaterSettings(
134  boost::dynamic_pointer_cast<
135  TranslationalStatePropagatorSettings< StateScalarType > >(
136  propagatorSettings )->getAccelerationsMap( ),
137  bodyMap );
138  break;
139  }
140  case rotational_state:
141  {
142  environmentModelsToUpdate = createRotationalEquationsOfMotionEnvironmentUpdaterSettings(
143  boost::dynamic_pointer_cast< RotationalStatePropagatorSettings< StateScalarType > >( propagatorSettings )->getTorqueModelsMap( ),
144  bodyMap );
145  break;
146  }
147  // Retrieve environment model settings for mass rate model
148  case body_mass_state:
149  {
150  environmentModelsToUpdate = createMassPropagationEnvironmentUpdaterSettings(
151  boost::dynamic_pointer_cast<
152  MassPropagatorSettings< StateScalarType > >(
153  propagatorSettings )->getMassRateModelsMap( ), bodyMap );
154  break;
155  }
156  case custom_state:
157  {
158  break;
159  }
160  default:
161  {
162  throw std::runtime_error( "Error, cannot create environment updates for type " +
163  std::to_string( propagatorSettings->getStateType( ) ) );
164  }
165  }
166  return environmentModelsToUpdate;
167 
168 }
