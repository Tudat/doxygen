213 template< typename IndependentVariableType, typename DependentVariableType, typename TimeStepType = IndependentVariableType >
214 boost::shared_ptr< numerical_integrators::NumericalIntegrator< IndependentVariableType, DependentVariableType,
215 DependentVariableType, TimeStepType > > createIntegrator(
216  boost::function< DependentVariableType(
217  const IndependentVariableType, const DependentVariableType& ) > stateDerivativeFunction,
218  const DependentVariableType initialState,
219  boost::shared_ptr< IntegratorSettings< IndependentVariableType > > integratorSettings )
220 
221 {
222  boost::shared_ptr< NumericalIntegrator
223  < IndependentVariableType, DependentVariableType, DependentVariableType, TimeStepType > > integrator;
224 
225  // Retrieve requested type of integrator
226  switch( integratorSettings->integratorType_ )
227  {
228  case euler:
229  integrator = boost::make_shared< EulerIntegrator
230  < IndependentVariableType, DependentVariableType, DependentVariableType, TimeStepType > >
231  ( stateDerivativeFunction, integratorSettings->initialTime_, initialState ) ;
232  break;
233  case rungeKutta4:
234 
235  integrator = boost::make_shared< RungeKutta4Integrator
236  < IndependentVariableType, DependentVariableType, DependentVariableType, TimeStepType > >
237  ( stateDerivativeFunction, integratorSettings->initialTime_, initialState ) ;
238  break;
239 
240 
241  case rungeKuttaVariableStepSize:
242  {
243  // Check input consistency
244  boost::shared_ptr< RungeKuttaVariableStepSizeSettings< IndependentVariableType > > variableStepIntegratorSettings =
245  boost::dynamic_pointer_cast< RungeKuttaVariableStepSizeSettings< IndependentVariableType > >(
246  integratorSettings );
247  if( variableStepIntegratorSettings == NULL )
248  {
249  std::runtime_error( "Error, type of integrator settings (rungeKuttaVariableStepSize) not compatible with selected integrator (derived class of IntegratorSettings must be RungeKuttaVariableStepSizeSettings for this type)" );
250  }
251  else
252  {
253  // Get requested RK coefficients and create integrator.
254  RungeKuttaCoefficients coefficients = RungeKuttaCoefficients::get(
255  variableStepIntegratorSettings->coefficientSet_ );
256  integrator = boost::make_shared<
257  RungeKuttaVariableStepSizeIntegrator
258  < IndependentVariableType, DependentVariableType, DependentVariableType, TimeStepType > >
259  ( coefficients,
260  stateDerivativeFunction, integratorSettings->initialTime_, initialState,
261  static_cast< TimeStepType >( variableStepIntegratorSettings->minimumStepSize_ ),
262  static_cast< TimeStepType >( variableStepIntegratorSettings->maximumStepSize_ ),
263  variableStepIntegratorSettings->relativeErrorTolerance_,
264  variableStepIntegratorSettings->absoluteErrorTolerance_ ,
265  static_cast< TimeStepType >( variableStepIntegratorSettings->safetyFactorForNextStepSize_ ),
266  static_cast< TimeStepType >( variableStepIntegratorSettings->maximumFactorIncreaseForNextStepSize_ ),
267  static_cast< TimeStepType >( variableStepIntegratorSettings->minimumFactorDecreaseForNextStepSize_ ) );
268  }
269  break;
270  }
271  default:
272  std::runtime_error(
273  "Error, integrator " + boost::lexical_cast< std::string >( integratorSettings->integratorType_ ) +
274  "not found. " ); }
275  return integrator;
276 }
