59 bool HybridPropagationTerminationCondition::checkStopCondition( const double time, const double cpuTime )
60 {
61  // Check if single condition is fulfilled.
62  if( fulFillSingleCondition_ )
63  {
64  bool stopPropagation = 0;
65  for( unsigned int i = 0; i < propagationTerminationCondition_.size( ); i++ )
66  {
67  if( propagationTerminationCondition_.at( i )->checkStopCondition( time, cpuTime ) )
68  {
69  stopPropagation = 1;

93 boost::shared_ptr< PropagationTerminationCondition > createPropagationTerminationConditions(
94  const boost::shared_ptr< PropagationTerminationSettings > terminationSettings,
95  const simulation_setup::NamedBodyMap& bodyMap,
96  const double initialTimeStep )
97 {
98  boost::shared_ptr< PropagationTerminationCondition > propagationTerminationCondition;
99 
100  // Check termination type.
101  switch( terminationSettings->terminationType_ )
102  {
103  case time_stopping_condition:
104  {
105  // Create stopping time termination condition.
106  boost::shared_ptr< PropagationTimeTerminationSettings > timeTerminationSettings =
107  boost::dynamic_pointer_cast< PropagationTimeTerminationSettings >( terminationSettings );
108  propagationTerminationCondition = boost::make_shared< FixedTimePropagationTerminationCondition >(
109  timeTerminationSettings->terminationTime_, ( initialTimeStep > 0 ) );
110  break;
111  }
112  case cpu_time_stopping_condition:
113  {
114  // Create stopping time termination condition.
115  boost::shared_ptr< PropagationCPUTimeTerminationSettings > cpuTimeTerminationSettings =
116  boost::dynamic_pointer_cast< PropagationCPUTimeTerminationSettings >( terminationSettings );
117  propagationTerminationCondition = boost::make_shared< FixedCPUTimePropagationTerminationCondition >(
118  cpuTimeTerminationSettings->cpuTerminationTime_ );
119  break;
120  }
121  case dependent_variable_stopping_condition:
122  {
123  boost::shared_ptr< PropagationDependentVariableTerminationSettings > dependentVariableTerminationSettings =
124  boost::dynamic_pointer_cast< PropagationDependentVariableTerminationSettings >( terminationSettings );
125 
126  // Get dependent variable function
127  boost::function< double( ) > dependentVariableFunction;
128  if( getDependentVariableSaveSize( dependentVariableTerminationSettings->dependentVariableSettings_ ) == 1 )
129  {
130  dependentVariableFunction =
131  getDoubleDependentVariableFunction(
132  dependentVariableTerminationSettings->dependentVariableSettings_, bodyMap );
133  }
134  else
135  {
136  throw std::runtime_error( "Error, cannot make stopping condition from vector dependent variable" );
137  }
138 
139  // Create dependent variable termination condition.
140  propagationTerminationCondition = boost::make_shared< SingleVariableLimitPropagationTerminationCondition >(
141  dependentVariableTerminationSettings->dependentVariableSettings_,
142  dependentVariableFunction, dependentVariableTerminationSettings->limitValue_,
143  dependentVariableTerminationSettings->useAsLowerLimit_ );
144  break;
145  }
146  case hybrid_stopping_condition:
147  {
148  boost::shared_ptr< PropagationHybridTerminationSettings > hybridTerminationSettings =
149  boost::dynamic_pointer_cast< PropagationHybridTerminationSettings >( terminationSettings );
150 
151  // Recursively create termination condition list.
152  std::vector< boost::shared_ptr< PropagationTerminationCondition > > propagationTerminationConditionList;
153  for( unsigned int i = 0; i < hybridTerminationSettings->terminationSettings_.size( ); i++ )
154  {
155  propagationTerminationConditionList.push_back(
156  createPropagationTerminationConditions(
157  hybridTerminationSettings->terminationSettings_.at( i ),
158  bodyMap, initialTimeStep ) );
159  }
160  propagationTerminationCondition = boost::make_shared< HybridPropagationTerminationCondition >(
161  propagationTerminationConditionList, hybridTerminationSettings->fulFillSingleCondition_ );
162  break;
163  }
164  default:
165  std::string errorMessage = "Error, stopping condition type " + std::to_string(
166  terminationSettings->terminationType_ ) + "not recognized when making stopping conditions object";
167  throw std::runtime_error( errorMessage );
168  break;
169  }
170  return propagationTerminationCondition;
