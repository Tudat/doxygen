149  void setConstantPerObservableAndLinkEndsWeights(
150  const std::map< observation_models::ObservableType,
151  std::map< observation_models::LinkEnds, double > > weightPerObservableAndLinkEnds )
152  {
153  for( typename PodInputDataType::const_iterator observablesIterator = observationsAndTimes_.begin( );
154  observablesIterator != observationsAndTimes_.end( ); observablesIterator++ )
155  {
156  if( weightPerObservableAndLinkEnds.count( observablesIterator->first ) == 0 )
157  {
158  std::string errorMessage =
159  "Error when setting weights per observable, observable " +
160  std::to_string( observablesIterator->first ) + " not found";
161  throw std::runtime_error( errorMessage );
162  }
163  else
164  {
165  for( typename SingleObservablePodInputType::const_iterator dataIterator =
166  observablesIterator->second.begin( ); dataIterator != observablesIterator->second.end( ); dataIterator++ )
167  {
168  if( weightPerObservableAndLinkEnds.at( observablesIterator->first ).count( dataIterator->first ) == 0 )
169  {
170  std::string errorMessage =
171  "Error when setting weights per observable, link ends not found for observable " +
172  std::to_string( observablesIterator->first );
173  throw std::runtime_error( errorMessage );
174  }
175  else
176  {
177 
178  weightsMatrixDiagonals_[ observablesIterator->first ][ dataIterator->first ] =
179  Eigen::VectorXd::Constant(
180  dataIterator->second.first.rows( ),
181  weightPerObservableAndLinkEnds.at( observablesIterator->first ).at( dataIterator->first ) );
182  }
183  }
184  }

404  Eigen::MatrixXd getUnnormalizedInverseCovarianceMatrix( )
405  {
406 
407  Eigen::MatrixXd inverseUnnormalizedCovarianceMatrix = inverseNormalizedCovarianceMatrix_;
408 
409  for( int i = 0; i < informationMatrixTransformationDiagonal_.rows( ); i++ )
410  {
411  for( int j = 0; j < informationMatrixTransformationDiagonal_.rows( ); j++ )
412  {
413  inverseUnnormalizedCovarianceMatrix( i, j ) *=
414  informationMatrixTransformationDiagonal_( i ) * informationMatrixTransformationDiagonal_( j );
415  }
416 
417  }
