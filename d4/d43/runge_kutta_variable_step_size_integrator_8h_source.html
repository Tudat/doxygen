431 StateType
432 RungeKuttaVariableStepSizeIntegrator< IndependentVariableType, StateType, StateDerivativeType, TimeStepType >
433 ::performIntegrationStep( const TimeStepType stepSize )
434 {
435  // Define and allocated vector for the number of stages.
436  currentStateDerivatives_.clear( );
437  currentStateDerivatives_.reserve( this->coefficients_.cCoefficients.rows( ) );
438 
439  // Define lower and higher order estimates.
440  StateType lowerOrderEstimate( this->currentState_ ),
441  higherOrderEstimate( this->currentState_ );
442 
443  // Compute the k_i state derivatives per stage.
444  for ( int stage = 0; stage < this->coefficients_.cCoefficients.rows( ); stage++ )
445  {
446  // Compute the intermediate state to pass to the state derivative for this stage.
447  StateType intermediateState( this->currentState_ );
448 
449  // Compute the intermediate state.
450  for ( int column = 0; column < stage; column++ )
451  {
452  intermediateState += stepSize * this->coefficients_.aCoefficients( stage, column )
453  * currentStateDerivatives_[ column ];
454  }
455 
456  // Compute the state derivative.
457  const IndependentVariableType time = this->currentIndependentVariable_ +
458  this->coefficients_.cCoefficients( stage ) * stepSize;
459  currentStateDerivatives_.push_back( this->stateDerivativeFunction_( time, intermediateState ) );
460 
461  // Check if propagation should terminate because the propagation termination condition has been reached
462  // while computing the intermediate state.
463  // If so, return immediately the current state (not recomputed yet), which will be discarded.
464  if ( this->propagationTerminationFunction_( static_cast< double >( time ), TUDAT_NAN ) )
465  {
466  this->propagationTerminationConditionReachedDuringStep_ = true;
467  return this->currentState_;
468  }
469 
470  // Update the estimate.
471  lowerOrderEstimate += this->coefficients_.bCoefficients( 0, stage ) * stepSize *
472  currentStateDerivatives_[ stage ];
473  higherOrderEstimate += this->coefficients_.bCoefficients( 1, stage ) * stepSize *
474  currentStateDerivatives_[ stage ];
475  }
476 
477  // Determine if the error was within bounds and compute a new step size.
478  if ( computeNextStepSizeAndValidateResult( lowerOrderEstimate,
479  higherOrderEstimate, stepSize ) )
480  {
481  // Accept the current step.
482  this->lastIndependentVariable_ = this->currentIndependentVariable_;
483  this->lastState_ = this->currentState_;
484  this->currentIndependentVariable_ += stepSize;
485 
486  switch ( this->coefficients_.orderEstimateToIntegrate )
487  {
488  case RungeKuttaCoefficients::lower:
489  this->currentState_ = lowerOrderEstimate;
490  return this->currentState_;
491 
492  case RungeKuttaCoefficients::higher:
493  this->currentState_ = higherOrderEstimate;
494  return this->currentState_;
495 
496  default: // The default case will never occur because OrderEstimateToIntegrate is an enum.
497  throw std::runtime_error( "Order estimate to integrate is invalid." );
498  }
499  }
500  else
501  {
502  // Reject current step.
503  return performIntegrationStep( this->stepSize_ );
504  }
505 }

508 template< typename IndependentVariableType, typename StateType, typename StateDerivativeType, typename TimeStepType >
509 bool
510 RungeKuttaVariableStepSizeIntegrator< IndependentVariableType, StateType, StateDerivativeType, TimeStepType >
511 ::computeNextStepSizeAndValidateResult(
512  const StateType& lowerOrderEstimate, const StateType& higherOrderEstimate,
513  const TimeStepType stepSize )
514 {
515  // Compute new step size using new step size function, which also returns whether the
516  // relative error is within bounds or not.
517  std::pair< TimeStepType, bool > newStepSizePair = this->newStepSizeFunction_(
518  stepSize, this->coefficients_.lowerOrder, this->coefficients_.higherOrder,
519  this->safetyFactorForNextStepSize_, this->relativeErrorTolerance_,
520  this->absoluteErrorTolerance_, lowerOrderEstimate,
521  higherOrderEstimate );
522 
523  // Check whether change in stepsize does not exceed bounds.
524  // If the stepsize is reduced to less than the prescibed minimum factor, set to minimum factor.
525  // If the stepsize is increased to more than the prescribed maximum factor, set to maximum
526  // factor. These bounds are necessary to prevent the stepsize changes from aliasing
527  // with the dynamics of the system of ODEs.
528  // Also check if maximum step size is exceeded and step next step size to maximum if necessary.
529  // Typically used bounds can be found in (Burden and Faires, 2001).
530  if ( newStepSizePair.first / stepSize <= this->minimumFactorDecreaseForNextStepSize_ )
531  {
532  this->stepSize_ = stepSize * this->minimumFactorDecreaseForNextStepSize_;
533  }
534 
535  else if ( newStepSizePair.first / stepSize >= maximumFactorIncreaseForNextStepSize_ )
536  {
537  this->stepSize_ = stepSize * this->maximumFactorIncreaseForNextStepSize_;
538  }
539 
540  else
541  {
542  this->stepSize_ = newStepSizePair.first;
543  }
544 
545  // Check if minimum step size is violated and throw exception if necessary.
546  if ( std::fabs( this->stepSize_ ) < this->minimumStepSize_ )
547  {
548  throw MinimumStepSizeExceededError( this->minimumStepSize_,
549  std::fabs( this->stepSize_ ) );
550  }
551 
552  else if ( std::fabs( this->stepSize_ ) > this->maximumStepSize_ )
553  {
554  this->stepSize_ = this->maximumStepSize_;
555  }
556 
557  // Check if computed error in state is too large and reject step if true.
558  return newStepSizePair.second;
559 }

565 template< typename IndependentVariableType, typename StateType, typename StateDerivativeType, typename TimeStepType >
566 std::pair< TimeStepType, bool >
567 RungeKuttaVariableStepSizeIntegrator< IndependentVariableType, StateType, StateDerivativeType, TimeStepType >
568 ::computeNewStepSize(
569  const TimeStepType stepSize,
570  const TimeStepType lowerOrder,
571  const TimeStepType higherOrder,
572  const TimeStepType safetyFactorForNextStepSize,
573  const StateType& relativeErrorTolerance,
574  const StateType& absoluteErrorTolerance,
575  const StateType& lowerOrderEstimate,
576  const StateType& higherOrderEstimate )
577 {
578  TUDAT_UNUSED_PARAMETER( lowerOrder);
579 
580  // Compute the truncation error based on the higher and lower order estimates.
581  const StateType truncationError_ =
582  ( higherOrderEstimate - lowerOrderEstimate ).array( ).abs( );
583 
584  // Compute error tolerance based on relative and absolute error tolerances.
585  const StateType errorTolerance_ =
586  ( higherOrderEstimate.array( ).abs( ) *
587  relativeErrorTolerance.array( ) ).matrix( )
588  + absoluteErrorTolerance;
589 
590  // Compute relative truncation error. This will indicate if the current step satisfies the
591  // required tolerances.
592  const StateType relativeTruncationError_ = truncationError_.array( ) /
593  errorTolerance_.array( );
594 
595  // Compute the maximum error based on the largest coefficient in the relative truncation error
596  // matrix.
597  const typename StateType::Scalar maximumErrorInState_
598  = relativeTruncationError_.array( ).abs( ).maxCoeff( );
599 
600  // Compute the new step size. This is based off of the equation given in
601  // (Montenbruck and Gill, 2005).
602  const TimeStepType newStepSize = safetyFactorForNextStepSize * stepSize
603  * std::pow( 1.0 / maximumErrorInState_, 1.0 / higherOrder );
604 
605  // Check if the current state can be accepted.
606  const bool isIntegrationStepAccepted = maximumErrorInState_ <= 1.0;
607 
608  // Return the computed new step size and whether the current step taken is acceptable. If it
609  // isn't, the step will be recomputed with the computed new step size.
610  return std::make_pair( newStepSize, isIntegrationStepAccepted );
611 }
