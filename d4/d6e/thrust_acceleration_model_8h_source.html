127  void updateMembers( const double currentTime = TUDAT_NAN )
128  {
129  // Check if update is needed
130  if( !( currentTime_ == currentTime ) )
131  {
132  // Update thrust dependencies if needed
133  if( !thrustUpdateFunction_.empty( ) )
134  {
135  thrustUpdateFunction_( currentTime );
136  }
137 
138  // Retrieve thrust direction.
139  currentAccelerationDirection_ = thrustDirectionFunction_( );
140 
141  if( ( std::fabs( currentAccelerationDirection_.norm( ) ) - 1.0 ) >
142  10.0 * std::numeric_limits< double >::epsilon( ) )
143  {
144  throw std::runtime_error( "Error in thrust acceleration, direction is not a unit vector" );
145  }
146 
147  // Retrieve magnitude of thrust and mass rate.
148  currentThrustMagnitude_ = thrustMagnitudeFunction_( );
149  currentMassRate_ = -massRateFunction_( );
150 
151  // Compute acceleration due to thrust
152  currentAcceleration_ = currentAccelerationDirection_ * currentThrustMagnitude_ / bodyMassFunction_( );
153 
154  // Reset current time.
155  currentTime_ = currentTime;
156 
157  }
