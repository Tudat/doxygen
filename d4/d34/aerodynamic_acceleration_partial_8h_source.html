234  void computeAccelerationPartialWrtCurrentDragCoefficient( Eigen::MatrixXd& accelerationPartial )
235  {
236  Eigen::Quaterniond rotationToInertialFrame =
237  flightConditions_->getAerodynamicAngleCalculator( )->getRotationQuaternionBetweenFrames(
238  reference_frames::aerodynamic_frame, reference_frames::inertial_frame );
239 
240  double currentAirspeed = flightConditions_->getCurrentAirspeed( );
241  accelerationPartial =
242  rotationToInertialFrame * Eigen::Vector3d::UnitX( ) * (
243  -0.5 * flightConditions_->getCurrentDensity( ) * currentAirspeed * currentAirspeed *
244  flightConditions_->getAerodynamicCoefficientInterface( )->getReferenceArea( ) ) /
245  aerodynamicAcceleration_->getCurrentMass( );
246 
247  }
