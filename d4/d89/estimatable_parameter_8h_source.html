338  template< typename ParameterScalar >
339  void resetParameterValues( const Eigen::Matrix< ParameterScalar, Eigen::Dynamic, 1 >& newParameterValues )
340  {
341  // Check input consistency
342  if( newParameterValues.rows( ) != totalParameterSetSize_ )
343  {
344  throw std::runtime_error( "Error when resetting parameters of parameter set, given vector has size " +
345  std::to_string( newParameterValues.rows( ) ) +
346  ", while internal size is " + std::to_string( totalParameterSetSize_ ) );
347  }
348  else
349  {
350  int currentStartIndex = 0;
351 
352  for( unsigned int i = 0; i < estimateInitialStateParameters_.size( ); i++ )
353  {
354  estimateInitialStateParameters_[ i ]->setParameterValue(
355  newParameterValues.segment( currentStartIndex, estimateInitialStateParameters_[ i ]->getParameterSize( ) ).
356  template cast< InitialStateParameterType >( ) );
357  currentStartIndex += estimateInitialStateParameters_[ i ]->getParameterSize( );
358  }
359 
360  // Set double parameter values.
361  for( unsigned int i = 0; i < estimatedDoubleParameters_.size( ); i++ )
362  {
363  estimatedDoubleParameters_[ i ]->setParameterValue( static_cast< double >( newParameterValues( currentStartIndex ) ) );
364  currentStartIndex++;
365  }
366 
367  // Set vector parameter values.
368 
369  for( unsigned int i = 0; i < estimatedVectorParameters_.size( ); i++ )
370  {
371  estimatedVectorParameters_[ i ]->setParameterValue(
372  newParameterValues.segment( currentStartIndex, estimatedVectorParameters_[ i ]->getParameterSize( ) ).
373  template cast< double >( ) );
