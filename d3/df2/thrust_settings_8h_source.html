530 class AccelerationLimitedThrottleGuidance: public ThrustInputParameterGuidance
531 {
532 public:
533 
535 
546  AccelerationLimitedThrottleGuidance(
547  const NamedBodyMap& bodyMap,
548  const std::string nameOfBodyWithGuidance,
549  const std::string nameOfCentralBody,
550  const std::vector< propulsion::ThrustIndependentVariables > independentVariables,
551  const boost::shared_ptr< interpolators::Interpolator< double, double > > thrustInterpolator,
552  const double maximumAcceleration ): ThrustInputParameterGuidance( 1, 0, true, 0 ),
553  bodyWithGuidance_( bodyMap.at( nameOfBodyWithGuidance ) ), thrustInterpolator_( thrustInterpolator ),
554  maximumAcceleration_( maximumAcceleration )
555  {
556  // Split independent variables into environmental/guidance.
557  int numberOfThrottles = 0;
558  std::vector< propulsion::ThrustIndependentVariables > guidanceFreeIndependentVariables;
559  for( unsigned int i = 0; i < independentVariables.size( ); i++ )
560  {
561  if( independentVariables.at( i ) == propulsion::throttle_dependent_thrust )
562  {
563  numberOfThrottles ++;
564  }
565  else if( independentVariables.at( i ) == propulsion::guidance_input_dependent_thrust )
566  {
567  throw std::runtime_error( "Error in AccelerationLimitedThrottleGuidance, guidance input not supported" );
568  }
569  else
570  {
571  guidanceFreeIndependentVariables.push_back( independentVariables.at( i ) );
572  }
573  }
574 
575  // Check input consistency.
576  if( numberOfThrottles > 1 )
577  {
578  throw std::runtime_error( "Error in AccelerationLimitedThrottleGuidance, multiple throttles detected" );
579  }
580 
581  // Create thrust input.
582  if( bodyWithGuidance_->getFlightConditions( ) == NULL && nameOfCentralBody != "" )
583  {
584  bodyWithGuidance_->setFlightConditions(
585  createFlightConditions( bodyWithGuidance_,
586  bodyMap.at( nameOfCentralBody ),
587  nameOfBodyWithGuidance,
588  nameOfCentralBody ) );
589  }
590  thrustInputFunctions_ = getPropulsionInputVariables(
591  bodyWithGuidance_, guidanceFreeIndependentVariables );
592 
593  currentThrustInput_.resize( thrustInputFunctions_.size( ) );
594  }
595 
597 
601  void updateGuidanceParameters( )
602  {
603  // Compute environmental input variables for thrust.
604  for( unsigned int i = 0; i < thrustInputFunctions_.size( ); i++ )
605  {
606  currentThrustInput_[ i ] = thrustInputFunctions_.at( i )( );
607  }
608 
609  // Compute maximum thrust
610  double currentThrust = thrustInterpolator_->interpolate( currentThrustInput_ );
611  double currentMass = bodyWithGuidance_->getBodyMass( );
612 
613  // Set throttle (to < 1 if necessary)
614  if( currentThrust / currentMass < maximumAcceleration_ )
615  {
616  currentThrustGuidanceParameters_[ 0 ] = 1.0;
617  }
618  else
619  {
620  currentThrustGuidanceParameters_[ 0 ] = maximumAcceleration_ * currentMass / currentThrust;
621  }
622  }
623

631 
633  double maximumAcceleration_;
634 
636  std::vector< boost::function< double( ) > > thrustInputFunctions_;
637
