38 template< int ObservationSize = 1 >
39 void performObservationParameterEstimationClosureForSingleModelSet(
40  const boost::shared_ptr< estimatable_parameters::EstimatableParameter< Eigen::VectorXd > > parameter,
41  const boost::shared_ptr< ObservationBias< ObservationSize > > observationBias,
42  const LinkEnds linkEnds,
43  const ObservableType observableType )
44 {
45  ObservationBiasTypes biasType = getObservationBiasType( observationBias );
46 
47  // Check if bias type is multi-type
48  if( biasType == multiple_observation_biases )
49  {
50  // Test input consistency
51  boost::shared_ptr< MultiTypeObservationBias< ObservationSize > > multiTypeBias =
52  boost::dynamic_pointer_cast< MultiTypeObservationBias< ObservationSize > >( observationBias );
53  if( multiTypeBias == NULL )
54  {
55  throw std::runtime_error( "Error, cannot perform bias closure for multi-type bias, inconsistent bias types" );
56  }
57 
58  // Perform closure for each constituent bias object
59  for( unsigned int i = 0; i < multiTypeBias->getBiasList( ).size( ); i++ )
60  {
61  performObservationParameterEstimationClosureForSingleModelSet(
62  parameter, multiTypeBias->getBiasList( ).at( i ), linkEnds, observableType );
63  }
64  }
65  else
66  {
67  // Check bias type
68  switch( parameter->getParameterName( ).first )
69  {
70  case estimatable_parameters::constant_additive_observation_bias:
71  {
72  // Test input consistency
73  boost::shared_ptr< estimatable_parameters::ConstantObservationBiasParameter > biasParameter =
74  boost::dynamic_pointer_cast< estimatable_parameters::ConstantObservationBiasParameter >(
75  parameter );
76  if( biasParameter == NULL )
77  {
78  throw std::runtime_error( "Error, cannot perform bias closure for additive bias, inconsistent bias types" );
79  }
80 
81  // Check if bias object is of same type as estimated parameter
82  boost::shared_ptr< ConstantObservationBias< ObservationSize > > constantBiasObject =
83  boost::dynamic_pointer_cast< ConstantObservationBias< ObservationSize > >( observationBias );
84  if( constantBiasObject != NULL )
85  {
86  // Check if bias and parameter link properties are equal
87  if( linkEnds == biasParameter->getLinkEnds( ) &&
88  observableType == biasParameter->getObservableType( ) )
89  {
90  biasParameter->setObservationBiasFunctions(
91  boost::bind( &ConstantObservationBias< ObservationSize >::getTemplateFreeConstantObservationBias,
92  constantBiasObject ),
93  boost::bind( &ConstantObservationBias< ObservationSize >::resetConstantObservationBiasTemplateFree,
94  constantBiasObject, _1 ) );
95  }
96  }
97  break;
98  }
99  case estimatable_parameters::constant_relative_observation_bias:
100  {
101  // Test input consistency
102  boost::shared_ptr< estimatable_parameters::ConstantRelativeObservationBiasParameter > biasParameter =
103  boost::dynamic_pointer_cast< estimatable_parameters::ConstantRelativeObservationBiasParameter >(
104  parameter );
105  if( biasParameter == NULL )
106  {
107  throw std::runtime_error( "Error, cannot perform bias closure for additive bias, inconsistent bias types" );
108  }
109 
110  // Check if bias object is of same type as estimated parameter
111  boost::shared_ptr< ConstantRelativeObservationBias< ObservationSize > > constantBiasObject =
112  boost::dynamic_pointer_cast< ConstantRelativeObservationBias< ObservationSize > >( observationBias );
113  if( constantBiasObject != NULL )
114  {
115  // Check if bias and parameter link properties are equal
116  if( linkEnds == biasParameter->getLinkEnds( ) &&
117  observableType == biasParameter->getObservableType( ) )
118  {
119  biasParameter->setObservationBiasFunctions(
120  boost::bind( &ConstantRelativeObservationBias< ObservationSize >::getTemplateFreeConstantObservationBias,
121  constantBiasObject ),
122  boost::bind( &ConstantRelativeObservationBias< ObservationSize >::resetConstantObservationBiasTemplateFree,
123  constantBiasObject, _1 ) );
124  }
125  }
126  break;
127  }
128  default:
129  std::string errorMessage = "Error when closing observation bias/estimation loop, did not recognize bias type " +
130  std::to_string( parameter->getParameterName( ).first );
131  throw std::runtime_error( errorMessage );
132 
133  }
134  }

143 template< int ObservationSize = 1, typename ObservationScalarType = double, typename TimeType = double >
144 void performObservationParameterEstimationClosure(
145  boost::shared_ptr< ObservationSimulator< ObservationSize, ObservationScalarType, TimeType > > observationSimulator ,
146  const boost::shared_ptr< estimatable_parameters::EstimatableParameterSet< ObservationScalarType > >
147  parametersToEstimate )
148 {
149  // Retrieve observation models and parameter
150  std::map< LinkEnds, boost::shared_ptr< ObservationModel< ObservationSize, ObservationScalarType, TimeType > > >
151  observationModels = observationSimulator->getObservationModels( );
152  std::vector< boost::shared_ptr< estimatable_parameters::EstimatableParameter< Eigen::VectorXd > > > vectorParameters =
153  parametersToEstimate->getEstimatedVectorParameters( );
154 
155  // Retrieve estimated bias parameters.
156  std::vector< boost::shared_ptr< estimatable_parameters::EstimatableParameter< Eigen::VectorXd > > > vectorBiasParameters;
157  for( unsigned int i = 0; i < vectorParameters.size( ); i++ )
158  {
159  if( estimatable_parameters::isParameterObservationLinkProperty( vectorParameters.at( i )->getParameterName( ).first ) )
160  {
161  vectorBiasParameters.push_back( vectorParameters.at( i ) );
162  }
163  }
164 
165  if( vectorBiasParameters.size( ) > 0 )
166  {
167  // Retrieve bias objects
168  std::map< LinkEnds, boost::shared_ptr< ObservationBias< ObservationSize > > > observationBiases =
169  extractObservationBiasList( observationModels );
170 
171  // Iterate over all combinations of parameters and biases and perform closure for each (if needed)
172  for( unsigned int i = 0; i < vectorBiasParameters.size( ); i++ )
173  {
174  for( typename std::map< LinkEnds, boost::shared_ptr< ObservationBias< ObservationSize > > >::const_iterator
175  biasIterator = observationBiases.begin( ); biasIterator != observationBiases.end( ); biasIterator++ )
176  {
177  performObservationParameterEstimationClosureForSingleModelSet(
178  vectorBiasParameters.at( i ), biasIterator->second, biasIterator->first,
179  observationSimulator->getObservableType( ) );
180  }
181  }
182  }

196 template< int ObservationSize = 1, typename ObservationScalarType, typename TimeType >
197 boost::shared_ptr< ObservationManagerBase< ObservationScalarType, TimeType > > createObservationManager(
198  const ObservableType observableType,
199  const std::map< LinkEnds, boost::shared_ptr< ObservationSettings > > settingsPerLinkEnds,
200  const simulation_setup::NamedBodyMap &bodyMap,
201  const boost::shared_ptr< estimatable_parameters::EstimatableParameterSet< ObservationScalarType > >
202  parametersToEstimate,
203  const boost::shared_ptr< propagators::CombinedStateTransitionAndSensitivityMatrixInterface >
204  stateTransitionMatrixInterface )
205 {
206  using namespace observation_models;
207  using namespace observation_partials;
208 
209  // Create observation simulator
210  boost::shared_ptr< ObservationSimulator< ObservationSize, ObservationScalarType, TimeType > > observationSimulator =
211  createObservationSimulator< ObservationSize, ObservationScalarType, TimeType >(
212  observableType, settingsPerLinkEnds, bodyMap );
213 
214 
215  performObservationParameterEstimationClosure(
216  observationSimulator, parametersToEstimate );
217 
218  // Create observation partials for all link ends/parameters
219  boost::shared_ptr< ObservationPartialCreator< ObservationSize, ObservationScalarType, TimeType > > observationPartialCreator =
220  boost::make_shared< ObservationPartialCreator< ObservationSize, ObservationScalarType, TimeType > >( );
221  std::map< LinkEnds, std::pair< std::map< std::pair< int, int >,
222  boost::shared_ptr< ObservationPartial< ObservationSize > > >,
223  boost::shared_ptr< PositionPartialScaling > > > observationPartialsAndScaler;
224  if( parametersToEstimate != NULL )
225  {
226  observationPartialsAndScaler =
227  observationPartialCreator->createObservationPartials(
228  observableType, observationSimulator->getObservationModels( ), bodyMap, parametersToEstimate );
229  }
230 
231  // Split position partial scaling and observation partial objects.
232  std::map< LinkEnds, std::map< std::pair< int, int >,
233  boost::shared_ptr< observation_partials::ObservationPartial< ObservationSize > > > > observationPartials;
234  std::map< LinkEnds, boost::shared_ptr< observation_partials::PositionPartialScaling > > observationPartialScalers;
235  splitObservationPartialsAndScalers( observationPartialsAndScaler, observationPartials, observationPartialScalers );
236 
237 
238  return boost::make_shared< ObservationManager< ObservationSize, ObservationScalarType, TimeType > >(
239  observableType, observationSimulator, observationPartials,
240  observationPartialScalers, stateTransitionMatrixInterface );
241 }

255 template< typename ObservationScalarType, typename TimeType >
256 boost::shared_ptr< ObservationManagerBase< ObservationScalarType, TimeType > > createObservationManagerBase(
257  const ObservableType observableType,
258  const std::map< LinkEnds, boost::shared_ptr< ObservationSettings > > settingsPerLinkEnds,
259  const simulation_setup::NamedBodyMap &bodyMap,
260  const boost::shared_ptr< estimatable_parameters::EstimatableParameterSet< ObservationScalarType > > parametersToEstimate,
261  const boost::shared_ptr< propagators::CombinedStateTransitionAndSensitivityMatrixInterface > stateTransitionMatrixInterface )
262 {
263  boost::shared_ptr< ObservationManagerBase< ObservationScalarType, TimeType > > observationManager;
264  switch( observableType )
265  {
266  case one_way_range:
267  observationManager = createObservationManager< 1, ObservationScalarType, TimeType >(
268  observableType, settingsPerLinkEnds, bodyMap, parametersToEstimate,
269  stateTransitionMatrixInterface );
270  break;
271  case n_way_range:
272  observationManager = createObservationManager< 1, ObservationScalarType, TimeType >(
273  observableType, settingsPerLinkEnds, bodyMap, parametersToEstimate,
274  stateTransitionMatrixInterface );
275  break;
276  case one_way_doppler:
277  observationManager = createObservationManager< 1, ObservationScalarType, TimeType >(
278  observableType, settingsPerLinkEnds, bodyMap, parametersToEstimate,
279  stateTransitionMatrixInterface );
280  break;
281  case two_way_doppler:
282  observationManager = createObservationManager< 1, ObservationScalarType, TimeType >(
283  observableType, settingsPerLinkEnds, bodyMap, parametersToEstimate,
284  stateTransitionMatrixInterface );
285  break;
286  case one_way_differenced_range:
287  observationManager = createObservationManager< 1, ObservationScalarType, TimeType >(
288  observableType, settingsPerLinkEnds, bodyMap, parametersToEstimate,
289  stateTransitionMatrixInterface );
290  break;
291  case angular_position:
292  observationManager = createObservationManager< 2, ObservationScalarType, TimeType >(
293  observableType, settingsPerLinkEnds, bodyMap, parametersToEstimate,
294  stateTransitionMatrixInterface );
295  break;
296  case position_observable:
297  observationManager = createObservationManager< 3, ObservationScalarType, TimeType >(
298  observableType, settingsPerLinkEnds, bodyMap, parametersToEstimate,
299  stateTransitionMatrixInterface );
300  break;
301  default:
302  throw std::runtime_error(
303  "Error when making observation manager, could not identify observable type " +
304  std::to_string( observableType ) );
305  }
306  return observationManager;
307 }
