36 template< typename StateScalarType, typename TimeType >
37 std::map< propagators::IntegratedStateType, orbit_determination::StateDerivativePartialsMap > createStateDerivativePartials(
38  const std::unordered_map< propagators::IntegratedStateType,
39  std::vector< boost::shared_ptr< propagators::SingleStateTypeDerivative< StateScalarType, TimeType > > > >
40  stateDerivativeModels,
41  const simulation_setup::NamedBodyMap& bodyMap,
42  const boost::shared_ptr< estimatable_parameters::EstimatableParameterSet< StateScalarType > >
43  parametersToEstimate )
44 {
45  std::map< propagators::IntegratedStateType, orbit_determination::StateDerivativePartialsMap > stateDerivativePartials;
46 
47  // Iterate over all state types
48  for( typename std::unordered_map< propagators::IntegratedStateType,
49  std::vector< boost::shared_ptr< propagators::SingleStateTypeDerivative< StateScalarType, TimeType > > > >::
50  const_iterator stateDerivativeIterator = stateDerivativeModels.begin( );
51  stateDerivativeIterator != stateDerivativeModels.end( );
52  stateDerivativeIterator++ )
53  {
54  // Identify state types
55  switch( stateDerivativeIterator->first )
56  {
57  case propagators::transational_state:
58  {
59  if( stateDerivativeIterator->second.size( ) > 1 )
60  {
61  throw std::runtime_error(
62  "Error, cannot yet process multiple separate same type propagators when making partial derivatives of translational state." );
63  }
64  else
65  {
66  // Retrieve acceleration models and create partials
67  basic_astrodynamics::AccelerationMap accelerationModelList =
68  boost::dynamic_pointer_cast< propagators::NBodyStateDerivative< StateScalarType, TimeType > >(
69  stateDerivativeIterator->second.at( 0 ) )->getFullAccelerationsMap( );
70  stateDerivativePartials[ propagators::transational_state ] =
71  createAccelerationPartialsMap< StateScalarType >(
72  accelerationModelList, bodyMap, parametersToEstimate );
73  }
74  break;
75  }
76  default:
77  std::string errorMessage = "Cannot yet create state derivative partial models for type " +
78  std::to_string( stateDerivativeIterator->first );
79  throw std::runtime_error( errorMessage );
80  }
81  }
82  return stateDerivativePartials;
83 }
