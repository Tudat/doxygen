42 Eigen::VectorXd solveSystemOfEquationsWithSvd( const Eigen::MatrixXd matrixToInvert,
43  const Eigen::VectorXd rightHandSideVector,
44  const bool checkConditionNumber,
45  const double maximumAllowedConditionNumber )
46 {
47  Eigen::JacobiSVD< Eigen::MatrixXd > svdDecomposition = matrixToInvert.jacobiSvd(
48  Eigen::ComputeThinU | Eigen::ComputeThinV );
49  if( checkConditionNumber )
50  {
51  double conditionNumber = getConditionNumberOfDecomposedMatrix( svdDecomposition );
52 
53  if( conditionNumber > maximumAllowedConditionNumber )
54  {
55  std::cerr << "Warning when performing least squares, condition number is " << conditionNumber << std::endl;
56  }
57  }
58  return svdDecomposition.solve( rightHandSideVector );
59 }

62 Eigen::MatrixXd multiplyInformationMatrixByDiagonalWeightMatrix(
63  const Eigen::MatrixXd& informationMatrix,
64  const Eigen::VectorXd& diagonalOfWeightMatrix )
65 {
66  Eigen::MatrixXd weightedInformationMatrix = Eigen::MatrixXd::Zero( informationMatrix.rows( ), informationMatrix.cols( ) );
67 
68  for( int i = 0; i < informationMatrix.cols( ); i++ )
69  {
70  weightedInformationMatrix.block( 0, i, informationMatrix.rows( ), 1 ) =
71  informationMatrix.block( 0, i, informationMatrix.rows( ), 1 ).cwiseProduct( diagonalOfWeightMatrix );
72  }

99 std::pair< Eigen::VectorXd, Eigen::MatrixXd > performLeastSquaresAdjustmentFromInformationMatrix(
100  const Eigen::MatrixXd& informationMatrix,
101  const Eigen::VectorXd& observationResiduals,
102  const Eigen::VectorXd& diagonalOfWeightMatrix,
103  const Eigen::MatrixXd& inverseOfAPrioriCovarianceMatrix,
104  const bool checkConditionNumber,
105  const double maximumAllowedConditionNumber )
106 {
107  Eigen::VectorXd rightHandSide = informationMatrix.transpose( ) *

143 Eigen::VectorXd getLeastSquaresPolynomialFit(
144  const Eigen::VectorXd& independentValues,
145  const Eigen::VectorXd& dependentValues,
146  const std::vector< double >& polynomialPowers )
147 {
148  if( independentValues.rows( ) != dependentValues.rows( ) )
149  {
150  throw std::runtime_error( "Error when doing least squares polynomial fit, size of dependent and independent variable vectors is not equal" );
151  }
152 
153  Eigen::MatrixXd informationMatrix = Eigen::MatrixXd::Zero( dependentValues.rows( ), polynomialPowers.size( ) );
154 
155  // Compute information matrix
156  for( int i = 0; i < independentValues.rows( ); i++ )
157  {
158  for( unsigned int j = 0; j < polynomialPowers.size( ); j++ )
159  {
160  informationMatrix( i, j ) = std::pow( independentValues( i ), polynomialPowers.at( j ) );
161  }
162  }
163 
164  return performLeastSquaresAdjustmentFromInformationMatrix( informationMatrix, dependentValues ).first;
165 }
