177 template< typename IndependentVariableType, typename DependentVariableType >
178 boost::shared_ptr< OneDimensionalInterpolator< IndependentVariableType, DependentVariableType > >
179 createOneDimensionalInterpolator(
180  const std::map< IndependentVariableType, DependentVariableType > dataToInterpolate,
181  const boost::shared_ptr< InterpolatorSettings > interpolatorSettings,
182  const std::vector< DependentVariableType > firstDerivativeOfDependentVariables =
183  std::vector< DependentVariableType >( ) )
184 {
185  boost::shared_ptr< OneDimensionalInterpolator< IndependentVariableType, DependentVariableType > >
186  createdInterpolator;
187 
188  // Check type of interpolator.
189  switch( interpolatorSettings->getInterpolatorType( ) )
190  {
191  case linear_interpolator:
192  createdInterpolator = boost::make_shared< LinearInterpolator
193  < IndependentVariableType, DependentVariableType > >(
194  dataToInterpolate, interpolatorSettings->getSelectedLookupScheme( ) );
195  break;
196  case cubic_spline_interpolator:
197  {
198  if( !interpolatorSettings->getUseLongDoubleTimeStep( ) )
199  {
200  createdInterpolator = boost::make_shared< CubicSplineInterpolator
201  < IndependentVariableType, DependentVariableType > >(
202  dataToInterpolate, interpolatorSettings->getSelectedLookupScheme( ) );
203  }
204  else
205  {
206  createdInterpolator = boost::make_shared< CubicSplineInterpolator
207  < IndependentVariableType, DependentVariableType, long double > >(
208  dataToInterpolate, interpolatorSettings->getSelectedLookupScheme( ) );
209  }
210  break;
211  }
212  case lagrange_interpolator:
213  {
214  // Check consistency of input
215  boost::shared_ptr< LagrangeInterpolatorSettings > lagrangeInterpolatorSettings =
216  boost::dynamic_pointer_cast< LagrangeInterpolatorSettings >( interpolatorSettings );
217  if( lagrangeInterpolatorSettings != NULL )
218  {
219  // Create Lagrange interpolator with requested time step type
220  if( !lagrangeInterpolatorSettings->getUseLongDoubleTimeStep( ) )
221  {
222  createdInterpolator = boost::make_shared< LagrangeInterpolator
223  < IndependentVariableType, DependentVariableType, double > >(
224  dataToInterpolate, lagrangeInterpolatorSettings->getInterpolatorOrder( ),
225  interpolatorSettings->getSelectedLookupScheme( ),
226  lagrangeInterpolatorSettings->getBoundaryHandling( ) );
227  }
228  else
229  {
230  createdInterpolator = boost::make_shared< LagrangeInterpolator
231  < IndependentVariableType, DependentVariableType, long double > >(
232  dataToInterpolate, lagrangeInterpolatorSettings->getInterpolatorOrder( ),
233  interpolatorSettings->getSelectedLookupScheme( ),
234  lagrangeInterpolatorSettings->getBoundaryHandling( ) );
235  }
236  }
237  else
238  {
239  throw std::runtime_error( "Error, did not recognize lagrange interpolator settings" );
240 
241  }
242  break;
243  }
244  case hermite_spline_interpolator:
245  {
246  if( firstDerivativeOfDependentVariables.size( ) != dataToInterpolate.size( ) )
247  {
248  throw std::runtime_error( "Error when creating hermite spline interpolator, derivative size is inconsistent" );
249  }
250  createdInterpolator = boost::make_shared< HermiteCubicSplineInterpolator
251  < IndependentVariableType, DependentVariableType > >(
252  dataToInterpolate, firstDerivativeOfDependentVariables,
253  interpolatorSettings->getSelectedLookupScheme( ) );
254  break;
255  }
256  case piecewise_constant_interpolator:
257  createdInterpolator = boost::make_shared< PiecewiseConstantInterpolator
258  < IndependentVariableType, DependentVariableType > >(
259  dataToInterpolate, interpolatorSettings->getSelectedLookupScheme( ) );
260  break;
261  default:
262  throw std::runtime_error(
263  "Error when making interpolator, function cannot be used to create interplator of type " +
264  boost::lexical_cast< std::string >(
265  interpolatorSettings->getInterpolatorType( ) ) );
266  }
267  return createdInterpolator;
268 }
