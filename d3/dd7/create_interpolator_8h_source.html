377 template< typename IndependentVariableType, typename DependentVariableType >
378 boost::shared_ptr< OneDimensionalInterpolator< IndependentVariableType, DependentVariableType > >
379 createOneDimensionalInterpolator(
380  const std::map< IndependentVariableType, DependentVariableType > dataToInterpolate,
381  const boost::shared_ptr< InterpolatorSettings > interpolatorSettings,
382  const std::vector< DependentVariableType > firstDerivativeOfDependentVariables =
383  std::vector< DependentVariableType >( ) )
384 {
385  boost::shared_ptr< OneDimensionalInterpolator< IndependentVariableType, DependentVariableType > >
386  createdInterpolator;
387 
388  // Check type of interpolator.
389  switch( interpolatorSettings->getInterpolatorType( ) )
390  {
391  case linear_interpolator:
392  createdInterpolator = boost::make_shared< LinearInterpolator
393  < IndependentVariableType, DependentVariableType > >(
394  dataToInterpolate, interpolatorSettings->getSelectedLookupScheme( ) );
395  break;
396  case cubic_spline_interpolator:
397  {
398  if( !interpolatorSettings->getUseLongDoubleTimeStep( ) )
399  {
400  createdInterpolator = boost::make_shared< CubicSplineInterpolator
401  < IndependentVariableType, DependentVariableType > >(
402  dataToInterpolate, interpolatorSettings->getSelectedLookupScheme( ) );
403  }
404  else
405  {
406  createdInterpolator = boost::make_shared< CubicSplineInterpolator
407  < IndependentVariableType, DependentVariableType, long double > >(
408  dataToInterpolate, interpolatorSettings->getSelectedLookupScheme( ) );
409  }
410  break;
411  }
412  case lagrange_interpolator:
413  {
414  // Check consistency of input
415  boost::shared_ptr< LagrangeInterpolatorSettings > lagrangeInterpolatorSettings =
416  boost::dynamic_pointer_cast< LagrangeInterpolatorSettings >( interpolatorSettings );
417  if( lagrangeInterpolatorSettings != NULL )
418  {
419  // Create Lagrange interpolator with requested time step type
420  if( !lagrangeInterpolatorSettings->getUseLongDoubleTimeStep( ) )
421  {
422  createdInterpolator = boost::make_shared< LagrangeInterpolator
423  < IndependentVariableType, DependentVariableType, double > >(
424  dataToInterpolate, lagrangeInterpolatorSettings->getInterpolatorOrder( ),
425  interpolatorSettings->getSelectedLookupScheme( ),
426  lagrangeInterpolatorSettings->getBoundaryHandling( ) );
427  }
428  else
429  {
430  createdInterpolator = boost::make_shared< LagrangeInterpolator
431  < IndependentVariableType, DependentVariableType, long double > >(
432  dataToInterpolate, lagrangeInterpolatorSettings->getInterpolatorOrder( ),
433  interpolatorSettings->getSelectedLookupScheme( ),
434  lagrangeInterpolatorSettings->getBoundaryHandling( ) );
435  }
436  }
437  else
438  {
439  throw std::runtime_error( "Error, did not recognize lagrange interpolator settings" );
440 
441  }
442  break;
443  }
444  case hermite_spline_interpolator:
445  {
446  if( firstDerivativeOfDependentVariables.size( ) != dataToInterpolate.size( ) )
447  {
448  throw std::runtime_error(
449  "Error when creating hermite spline interpolator, derivative size is inconsistent" );
450  }
451  createdInterpolator = boost::make_shared< HermiteCubicSplineInterpolator
452  < IndependentVariableType, DependentVariableType > >(
453  dataToInterpolate, firstDerivativeOfDependentVariables,
454  interpolatorSettings->getSelectedLookupScheme( ) );
455  break;
456  }
457  case piecewise_constant_interpolator:
458  createdInterpolator = boost::make_shared< PiecewiseConstantInterpolator
459  < IndependentVariableType, DependentVariableType > >(
460  dataToInterpolate, interpolatorSettings->getSelectedLookupScheme( ) );
461  break;
462  default:
463  throw std::runtime_error(
464  "Error when making interpolator, function cannot be used to create interplator of type " +
465  std::to_string(
466  interpolatorSettings->getInterpolatorType( ) ) );
467  }
468  return createdInterpolator;
469 }

478 template< typename IndependentType, typename DependentType >
479 boost::shared_ptr< OneDimensionalInterpolator< IndependentType, DependentType > > createOneDimensionalInterpolator(
480  const boost::shared_ptr< DataInterpolationSettings< IndependentType, DependentType > >
481  dataInterpolationSettings )
482 {
483  std::vector< DependentType > firstDerivativeOfDependentVariables;
484  boost::shared_ptr< HermiteDataSettings< IndependentType, DependentType > > hermiteDataSettings =
485  boost::dynamic_pointer_cast< HermiteDataSettings< IndependentType, DependentType > >(
486  dataInterpolationSettings->dataSettings_ );
487  if ( hermiteDataSettings )
488  {
489  firstDerivativeOfDependentVariables = hermiteDataSettings->firstDerivativeOfDependentVariables_;
490  }
491  return createOneDimensionalInterpolator( dataInterpolationSettings->dataSettings_->getDataMap( ),
492  dataInterpolationSettings->interpolatorSettings_,
493  firstDerivativeOfDependentVariables );
494 }
