299  template< typename StateScalarType = double, typename TimeType = double >
300  void setStateFromEphemeris( const TimeType& time )
301  {
302  if( !( static_cast< Time >( time ) == timeOfCurrentState_ ) )
303  {
304  // If body is not global frame origin, set state.
305  if( bodyIsGlobalFrameOrigin_ == 0 )
306  {
307  if( sizeof( StateScalarType ) == 8 )
308  {
309  currentState_ =
310  ( bodyEphemeris_->getTemplatedStateFromEphemeris< StateScalarType, TimeType >( time ) +
311  ephemerisFrameToBaseFrame_->getBaseFrameState< TimeType, StateScalarType >( time ) ).
312  template cast< double >( );
313  currentLongState_ = currentState_.template cast< long double >( );
314  }
315  else
316  {
317  currentLongState_ =
318  ( bodyEphemeris_->getTemplatedStateFromEphemeris< StateScalarType, TimeType >( time ) +
319  ephemerisFrameToBaseFrame_->getBaseFrameState< TimeType, StateScalarType >( time ) ).
320  template cast< long double >( );
321  currentState_ = currentLongState_.template cast< double >( );
322  }
323  }
324  // If body is global frame origin, set state to zeroes, and barycentric state value.
325  else if( bodyIsGlobalFrameOrigin_ == 1 )
326  {
327  currentState_.setZero( );
328  currentLongState_.setZero( );
329 
330  if( sizeof( StateScalarType ) == 8 )
331  {
332  currentBarycentricState_ =
333  ephemerisFrameToBaseFrame_->getBaseFrameState< TimeType, StateScalarType >( time ).
334  template cast< double >( );
335  currentBarycentricLongState_ = currentBarycentricState_.template cast< long double >( );
336  }
337  else
338  {
339  currentBarycentricLongState_ =
340  ephemerisFrameToBaseFrame_->getBaseFrameState< TimeType, StateScalarType >( time ).
341  template cast< long double >( );
342  currentBarycentricState_ = currentBarycentricLongState_.template cast< double >( );
343  }
344  }
345  else
346  {
347  throw std::runtime_error( "Error when setting body state, global origin not yet defined." );
348  }
349 
350  timeOfCurrentState_ = static_cast< TimeType >( time );
351  }
352  }

570  Eigen::Quaterniond( currentRotationalStateFromLocalToGlobalFrame( 0 ),
