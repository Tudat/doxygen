500  void setCurrentRotationalStateToLocalFrame( const Eigen::Matrix< double, 7, 1 > currentRotationalStateFromLocalToGlobalFrame )
501  {
502  Eigen::Quaterniond currentRotationToGlobalFrame =
503  Eigen::Quaterniond( currentRotationalStateFromLocalToGlobalFrame( 0 ),
504  currentRotationalStateFromLocalToGlobalFrame( 1 ),
505  currentRotationalStateFromLocalToGlobalFrame( 2 ),
506  currentRotationalStateFromLocalToGlobalFrame( 3 ) );
507  currentRotationToGlobalFrame.normalize( );
508 
509  currentRotationToLocalFrame_ = currentRotationToGlobalFrame.inverse( );

511  currentRotationToGlobalFrame * currentRotationalStateFromLocalToGlobalFrame.block( 4, 0, 3, 1 );

513  Eigen::Matrix3d currentRotationMatrixToLocalFrame = ( currentRotationToLocalFrame_ ).toRotationMatrix( );
514  currentRotationToLocalFrameDerivative_ = linear_algebra::getCrossProductMatrix(

643  if( ( boost::dynamic_pointer_cast< reference_frames::AerodynamicAngleCalculator >(
644  dependentOrientationCalculator ) != NULL ) &&

646  dependentOrientationCalculator_ ) == NULL ) )
647  {
648  reference_frames::setAerodynamicDependentOrientationCalculatorClosure(
649  dependentOrientationCalculator_,
650  boost::dynamic_pointer_cast< reference_frames::AerodynamicAngleCalculator >(

652  }
653  else if( ( boost::dynamic_pointer_cast< reference_frames::AerodynamicAngleCalculator >(
654  dependentOrientationCalculator_ ) != NULL ) &&

656  dependentOrientationCalculator ) == NULL ) )
