98 Eigen::VectorXd evaluateListOfVectorFunctions(
99  const std::vector< std::pair< boost::function< Eigen::VectorXd( ) >, int > > vectorFunctionList,
100  const int totalSize )
101 {
102  Eigen::VectorXd variableList = Eigen::VectorXd::Zero( totalSize );
103  int currentIndex = 0;
104 
105  for( std::pair< boost::function< Eigen::VectorXd( ) >, int > vectorFunction: vectorFunctionList )
106  {
107  variableList.segment( currentIndex, vectorFunction.second ) = vectorFunction.first( );
108  currentIndex += vectorFunction.second;
109  }
110 
111  // Check consistency with input
112  if( currentIndex != totalSize )
113  {
114  std::string errorMessage = "Error when evaluating lists of functions, sizes are inconsistent: " +
115  std::to_string( currentIndex ) + " and " +
116  std::to_string( totalSize );
117  throw std::runtime_error( errorMessage );
118  }
119 
120  return variableList;
121 }
