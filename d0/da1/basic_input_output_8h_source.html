227 template< typename InputIterator >
228 void writeDataMapToTextFile(
229  InputIterator iteratorDataMap, InputIterator last,
230  const std::string& outputFilename,
231  const boost::filesystem::path& outputDirectory, const std::string& fileHeader,
232  const int precisionOfKeyType, const int precisionOfValueType,
233  const std::string& delimiter )
234 {
235  // Check if output directory exists; create it if it doesn't.
236  if ( !boost::filesystem::exists( outputDirectory ) )
237  {
238  boost::filesystem::create_directories( outputDirectory );
239  }
240 
241  // Open output file.
242  std::string outputDirectoryAndFilename = outputDirectory.string( ) + "/" + outputFilename;
243  std::ofstream outputFile_( outputDirectoryAndFilename.c_str( ) );
244 
245  // Write file header to file.
246  outputFile_ << fileHeader;
247 
248  // Loop over map of propagation history.
249  for ( ; iteratorDataMap != last; iteratorDataMap++ )
250  {
251  // Print map data to output file.
252  outputFile_ << std::setprecision( precisionOfKeyType )
253  << std::left << std::setw( precisionOfKeyType + 1 )
254  << iteratorDataMap->first;
255  writeValueToStream( outputFile_, iteratorDataMap->second, precisionOfValueType,
256  delimiter );
257  }
258 
259  // Close output file.
260  outputFile_.close( );
261 }
