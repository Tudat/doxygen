193 template< typename InputIterator >
194 void writeDataMapToTextFile(
195  InputIterator iteratorDataMap, InputIterator last,
196  const std::string& outputFilename,
197  const boost::filesystem::path& outputDirectory, const std::string& fileHeader,
198  const int precisionOfKeyType, const int precisionOfValueType,
199  const std::string& delimiter )
200 {
201  // Check if output directory exists; create it if it doesn't.
202  if ( !boost::filesystem::exists( outputDirectory ) )
203  {
204  boost::filesystem::create_directories( outputDirectory );
205  }
206 
207  // Open output file.
208  std::string outputDirectoryAndFilename = outputDirectory.string( ) + "/" + outputFilename;
209  std::ofstream outputFile_( outputDirectoryAndFilename.c_str( ) );
210 
211  // Write file header to file.
212  outputFile_ << fileHeader;
213 
214  // Loop over map of propagation history.
215  for ( ; iteratorDataMap != last; iteratorDataMap++ )
216  {
217  // Print map data to output file.
218  outputFile_ << std::setprecision( precisionOfKeyType )
219  << std::left << std::setw( precisionOfKeyType + 1 )
220  << iteratorDataMap->first;
221  writeValueToStream( outputFile_, iteratorDataMap->second, precisionOfValueType,
222  delimiter );
223  }
224 
225  // Close output file.
226  outputFile_.close( );
227 }
