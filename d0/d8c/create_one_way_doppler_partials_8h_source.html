142 template< typename ParameterType >
143 std::pair< SingleLinkObservationPartialList, boost::shared_ptr< PositionPartialScaling > > createOneWayDopplerPartials(
144  const observation_models::LinkEnds oneWayDopplerLinkEnds,
145  const simulation_setup::NamedBodyMap& bodyMap,
146  const boost::shared_ptr< estimatable_parameters::EstimatableParameterSet< ParameterType > > parametersToEstimate,
147  const boost::shared_ptr< observation_models::DopplerProperTimeRateInterface > transmitterDopplerProperTimeInterface,
148  const boost::shared_ptr< observation_models::DopplerProperTimeRateInterface > receiverDopplerProperTimeInterface,
149  const std::vector< boost::shared_ptr< observation_models::LightTimeCorrection > >& lightTimeCorrections =
150  std::vector< boost::shared_ptr< observation_models::LightTimeCorrection > >( ) )
151 {
152  std::vector< boost::shared_ptr< observation_partials::LightTimeCorrectionPartial > > lightTimeCorrectionPartialObjects;
153  if( lightTimeCorrections.size( ) > 0 )
154  {
155  lightTimeCorrectionPartialObjects = observation_partials::createLightTimeCorrectionPartials( lightTimeCorrections );
156  }
157 
158  boost::function< Eigen::Vector6d( const double )> transmitterNumericalStateDerivativeFunction =
159  boost::bind( &numerical_derivatives::computeCentralDifference< Eigen::Vector6d, double >,
160  observation_models::getLinkEndCompleteEphemerisFunction< double, double >(
161  oneWayDopplerLinkEnds.at( observation_models::transmitter ), bodyMap ), _1, 100.0,
162  numerical_derivatives::order8 );
163  boost::function< Eigen::Vector6d( const double )> receiverNumericalStateDerivativeFunction =
164  boost::bind( numerical_derivatives::computeCentralDifference< Eigen::Vector6d, double >,
165  observation_models::getLinkEndCompleteEphemerisFunction< double, double >(
166  oneWayDopplerLinkEnds.at( observation_models::receiver ), bodyMap ), _1, 100.0,
167  numerical_derivatives::order8 );
168 
169  // Create scaling object, to be used for all one-way doppler partials in current link end.
170  boost::shared_ptr< OneWayDopplerProperTimeComponentScaling > transmitterProperTimePartials =
171  createDopplerProperTimePartials( transmitterDopplerProperTimeInterface, oneWayDopplerLinkEnds,
172  observation_models::transmitter );
173  boost::shared_ptr< OneWayDopplerProperTimeComponentScaling > receiverProperTimePartials =
174  createDopplerProperTimePartials( receiverDopplerProperTimeInterface, oneWayDopplerLinkEnds,
175  observation_models::receiver );
176 
177  boost::shared_ptr< OneWayDopplerScaling > oneWayDopplerScaling = boost::make_shared< OneWayDopplerScaling >(
178  boost::bind( &linear_algebra::evaluateSecondBlockInStateVector, transmitterNumericalStateDerivativeFunction, _1 ),
179  boost::bind( &linear_algebra::evaluateSecondBlockInStateVector, receiverNumericalStateDerivativeFunction, _1 ),
180  transmitterProperTimePartials,
181  receiverProperTimePartials );
182 
183 
184  // Initialize vector index variables.
185  int currentIndex = 0;
186  std::pair< int, int > currentPair = std::pair< int, int >( currentIndex, 1 );
187 
188  SingleLinkObservationPartialList dopplerPartials;
189 
190  std::vector< boost::shared_ptr< estimatable_parameters::EstimatableParameter<
191  Eigen::Matrix< ParameterType, Eigen::Dynamic, 1 > > > > initialDynamicalParameters =
192  parametersToEstimate->getEstimatedInitialStateParameters( );
193 
194  // Iterate over list of bodies of which the partials of the accelerations acting on them are required.
195  for( unsigned int i = 0; i < initialDynamicalParameters.size( ); i++ )
196  {
197  std::string acceleratedBody;
198  if( initialDynamicalParameters.at( i )->getParameterName( ).first == estimatable_parameters::initial_body_state ||
199  ( initialDynamicalParameters.at( i )->getParameterName( ).first == estimatable_parameters::arc_wise_initial_body_state ) )
200  {
201  acceleratedBody = initialDynamicalParameters.at( i )->getParameterName( ).second.first;
202  }
203  else
204  {
205  throw std::runtime_error( "Error when making one way doppler partials, could not identify parameter " +
206  boost::lexical_cast< std::string >(
207  initialDynamicalParameters.at( i )->getParameterName( ).first ) );
208  }
209 
210 
211  // Create position one-way doppler partial for current body
212  boost::shared_ptr< ObservationPartial< 1 > > currentDopplerPartialWrtPosition = createOneWayDopplerPartialWrtBodyState(
213  oneWayDopplerLinkEnds, bodyMap, acceleratedBody, oneWayDopplerScaling, lightTimeCorrectionPartialObjects );
214 
215  // Check if partial is non-null (i.e. whether dependency exists between current doppler and current body)
216  if( currentDopplerPartialWrtPosition != NULL )
217  {
218  // Add partial to the list.
219  currentPair = std::pair< int, int >( currentIndex, 6 );
220  dopplerPartials[ currentPair ] = currentDopplerPartialWrtPosition;
221  }
222 
223  // Increment current index by size of body initial state (6).
224  currentIndex += 6;
225  }
226 
227  // Iterate over all double parameters that are to be estimated.
228  std::map< int, boost::shared_ptr< estimatable_parameters::EstimatableParameter< double > > > doubleParametersToEstimate =
229  parametersToEstimate->getDoubleParameters( );
230  for( std::map< int, boost::shared_ptr< estimatable_parameters::EstimatableParameter< double > > >::iterator
231  parameterIterator =
232  doubleParametersToEstimate.begin( ); parameterIterator != doubleParametersToEstimate.end( ); parameterIterator++ )
233  {
234  // Create position one-way doppler partial for current parameter
235  boost::shared_ptr< ObservationPartial< 1 > > currentDopplerPartial = createOneWayDopplerPartialWrtParameter(
236  oneWayDopplerLinkEnds, bodyMap, parameterIterator->second,
237  oneWayDopplerScaling, lightTimeCorrectionPartialObjects );
238 
239  // Check if partial is non-null (i.e. whether dependency exists between current doppler and current parameter)
240  if( currentDopplerPartial != NULL )
241  {
242  // Add partial to the list.
243  currentPair = std::pair< int, int >( parameterIterator->first, 1 );
244  dopplerPartials[ currentPair ] = currentDopplerPartial;
245  }
246  }
247 
248  // Iterate over all vector parameters that are to be estimated.
249  std::map< int, boost::shared_ptr< estimatable_parameters::EstimatableParameter< Eigen::VectorXd > > >
250  vectorParametersToEstimate =
251  parametersToEstimate->getVectorParameters( );
252  for( std::map< int, boost::shared_ptr< estimatable_parameters::EstimatableParameter< Eigen::VectorXd > > >::iterator
253  parameterIterator =
254  vectorParametersToEstimate.begin( ); parameterIterator != vectorParametersToEstimate.end( ); parameterIterator++ )
255  {
256 
257  boost::shared_ptr< ObservationPartial< 1 > > currentDopplerPartial;
258  if( !isParameterObservationLinkProperty( parameterIterator->second->getParameterName( ).first ) )
259  {
260  currentDopplerPartial = createOneWayDopplerPartialWrtParameter(
261  oneWayDopplerLinkEnds, bodyMap, parameterIterator->second, oneWayDopplerScaling,
262  lightTimeCorrectionPartialObjects );
263  }
264  else
265  {
266  currentDopplerPartial = createObservationPartialWrtLinkProperty< 1 >(
267  oneWayDopplerLinkEnds, observation_models::one_way_doppler, parameterIterator->second );
268  }
269 
270 
271  // Check if partial is non-null (i.e. whether dependency exists between current doppler and current parameter)
272  if( currentDopplerPartial != NULL )
273  {
274  // Add partial to the list.
275  currentPair = std::pair< int, int >( parameterIterator->first,
276  parameterIterator->second->getParameterSize( ) );
277  dopplerPartials[ currentPair ] = currentDopplerPartial;
278  }
279  }
280 
281  // Return complete set of partials and scaling object.
282  return std::make_pair( dopplerPartials, oneWayDopplerScaling );
283 }

299 template< typename ObservationScalarType, typename TimeType >
300 std::map< observation_models::LinkEnds, std::pair< SingleLinkObservationPartialList,
301 boost::shared_ptr< PositionPartialScaling > > > createOneWayDopplerPartials(
302  const std::map< observation_models::LinkEnds,
303  boost::shared_ptr< observation_models::ObservationModel< 1, ObservationScalarType, TimeType > > > observationModelList,
304  const simulation_setup::NamedBodyMap& bodyMap,
305  const boost::shared_ptr< estimatable_parameters::EstimatableParameterSet< ObservationScalarType > > parametersToEstimate )
306 {
307  // Declare return list.
308  std::map< observation_models::LinkEnds, std::pair< SingleLinkObservationPartialList,
309  boost::shared_ptr< PositionPartialScaling > > > dopplerPartials;
310 
311  // Iterate over all link ends.
312  for( auto linkIterator: observationModelList )
313  {
314  // Check if required link end types are present
315  if( ( linkIterator.first.count( observation_models::receiver ) == 0 ) ||
316  ( linkIterator.first.count( observation_models::transmitter ) == 0 ) )
317  {
318  throw std::runtime_error( "Error when making 1-way doppler partials, did not find both receiver and transmitter in link ends" );
319 
320  }
321 
322 
323  boost::shared_ptr< observation_models::OneWayDopplerObservationModel< ObservationScalarType, TimeType > >
324  dopplerObservationModel =
325  boost::dynamic_pointer_cast< observation_models::OneWayDopplerObservationModel
326  < ObservationScalarType, TimeType > >( linkIterator.second );
327 
328  std::vector< boost::shared_ptr< observation_models::LightTimeCorrection > > singleLinkLightTimeCorrections;
329  if( dopplerObservationModel == NULL )
330  {
331  throw std::runtime_error( "Error when making one-way Doppler partials. Type is inconsistent" );
332  }
333  else
334  {
335  singleLinkLightTimeCorrections = dopplerObservationModel->getLightTimeCalculator( )->getLightTimeCorrection( );
336  }
337 
338  // Create doppler partials for current link ends
339  dopplerPartials[ linkIterator.first ] = createOneWayDopplerPartials< ObservationScalarType >(
340  linkIterator.first, bodyMap, parametersToEstimate,
341  dopplerObservationModel->getTransmitterProperTimeRateCalculator( ),
342  dopplerObservationModel->getReceiverProperTimeRateCalculator( ),
343  singleLinkLightTimeCorrections );
344  }
345 
346  // Return complete set of link ends.
347  return dopplerPartials;
348 }
