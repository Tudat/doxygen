109  NBodyStateDerivative( const basic_astrodynamics::AccelerationMap& accelerationModelsPerBody,
110  const boost::shared_ptr< CentralBodyData< StateScalarType, TimeType > > centralBodyData,
111  const TranslationalPropagatorType propagatorType,
112  const std::vector< std::string >& bodiesToIntegrate ):
113  propagators::SingleStateTypeDerivative< StateScalarType, TimeType >(
114  propagators::transational_state ),
115  accelerationModelsPerBody_( accelerationModelsPerBody ),
116  centralBodyData_( centralBodyData ),
117  propagatorType_( propagatorType ),
118  bodiesToBeIntegratedNumerically_( bodiesToIntegrate )
119  {
120  // Add empty acceleration map if body is to be propagated with no accelerations.
121  for( unsigned int i = 0; i < bodiesToBeIntegratedNumerically_.size( ); i++ )
122  {
123  if( accelerationModelsPerBody_.count( bodiesToBeIntegratedNumerically_.at( i ) ) == 0 )
124  {
125  accelerationModelsPerBody_[ bodiesToBeIntegratedNumerically_.at( i ) ] =
126  basic_astrodynamics::SingleBodyAccelerationMap( );
127  }
128  }
129 
130  // Correct order of propagated bodies.
131  for( outerAccelerationIterator = accelerationModelsPerBody_.begin( );
132  outerAccelerationIterator != accelerationModelsPerBody_.end( );
133  outerAccelerationIterator++ )
134  {
135  std::vector< std::string >::iterator findIterator =
136  std::find( bodiesToBeIntegratedNumerically_.begin( ), bodiesToBeIntegratedNumerically_.end( ),
137  outerAccelerationIterator->first );
138  bodyOrder_.push_back( std::distance( bodiesToBeIntegratedNumerically_.begin( ), findIterator ) );
139  }

141  createAccelerationModelList( );
142  }

281  Eigen::Vector3d getTotalAccelerationForBody(
282  const std::string& bodyName )
283  {
284  // Check if body is propagated.
285  Eigen::Vector3d totalAcceleration = Eigen::Vector3d::Zero( );
286  if( std::find( bodiesToBeIntegratedNumerically_.begin( ),
287  bodiesToBeIntegratedNumerically_.end( ),
288  bodyName ) == bodiesToBeIntegratedNumerically_.end( ) )
289  {
290  std::string errorMessage = "Error when getting total acceleration for body " + bodyName +
291  ", no such acceleration is found";
292  throw std::runtime_error( errorMessage );
293  }
294  else
295  {
296  if( accelerationModelsPerBody_.count( bodyName ) != 0 )
297  {
298  basic_astrodynamics::SingleBodyAccelerationMap accelerationsOnBody =
299  accelerationModelsPerBody_.at( bodyName );
300 
301  // Iterate over all accelerations acting on body
302  for( innerAccelerationIterator = accelerationsOnBody.begin( );
303  innerAccelerationIterator != accelerationsOnBody.end( );
304  innerAccelerationIterator++ )
305  {
306  for( unsigned int j = 0; j < innerAccelerationIterator->second.size( ); j++ )
307  {
308  // Calculate acceleration and add to state derivative.
309  totalAcceleration += innerAccelerationIterator->second[ j ]->getAcceleration( );
310  }
311  }
312  }
313  }
314  return totalAcceleration;
315  }

363  void sumStateDerivativeContributions(
364  const Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 >& stateOfSystemToBeIntegrated,
365  Eigen::Block< Eigen::Matrix< StateScalarType, Eigen::Dynamic, Eigen::Dynamic > > stateDerivative,
366  const bool addPositionDerivatives = true )
367  {
368  using namespace basic_astrodynamics;
369 
370  stateDerivative.setZero( );
371 
372  int currentBodyIndex = 0;
373  int currentAccelerationIndex = 0;
374 
375  // Iterate over all bodies with accelerations.
376  for( outerAccelerationIterator = accelerationModelsPerBody_.begin( );
377  outerAccelerationIterator != accelerationModelsPerBody_.end( );
378  outerAccelerationIterator++ )
379  {
380  currentBodyIndex = bodyOrder_[ currentAccelerationIndex ];
381 
382  // Iterate over all accelerations acting on body
383  for( innerAccelerationIterator = outerAccelerationIterator->second.begin( );
384  innerAccelerationIterator != outerAccelerationIterator->second.end( );
385  innerAccelerationIterator++ )
386  {
387  for( unsigned int j = 0; j < innerAccelerationIterator->second.size( ); j++ )
388  {
389  // Calculate acceleration and add to state derivative.
390  stateDerivative.block( currentBodyIndex * 6 + 3, 0, 3, 1 ) += (
391  innerAccelerationIterator->second[ j ]->getAcceleration( ) ).
392  template cast< StateScalarType >( );
393  }
394  }
395 
396  if( addPositionDerivatives )
397  {
398  // Add body velocity as derivative of its position.
399  stateDerivative.block( currentBodyIndex * 6, 0, 3, 1 ) =
400  ( stateOfSystemToBeIntegrated.segment( currentBodyIndex * 6 + 3, 3 ) );
401  }
402  currentAccelerationIndex++;
403  }
404  }
