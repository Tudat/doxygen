181  void updateInterpolator( const double time )
182  {
183  if( ! ( time == currentTime_ ) )
184  {
185  // Retrieve data from interpolator
186  currentRotationalState_ = interpolator_->interpolate( time );
187 
188  // Normalize quaternion and set current rotation quaternion
189  double quaternionNorm = ( currentRotationalState_.block( 0, 0, 4, 1 ) ).norm( );
190  currentRotationalState_.block( 0, 0, 4, 1 ) = currentRotationalState_.block( 0, 0, 4, 1 ) / quaternionNorm;
191  currentRotationToBaseFrame_ = Eigen::Quaternion< StateScalarType >(
192  currentRotationalState_( 0 ), currentRotationalState_( 1 ),
193  currentRotationalState_( 2 ), currentRotationalState_( 3 ) );
194 
195  // Set current angular velocity vector.
196  currentRotationalVelocityVectorInTargetFrame_ = currentRotationalState_.block( 4, 0, 3, 1 );
197 
198  // Set time at which interpolator was evaluated.
199  currentTime_ = time;
200  }
201  }
