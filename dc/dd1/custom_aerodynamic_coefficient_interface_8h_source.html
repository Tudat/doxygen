139  virtual void updateCurrentCoefficients( const std::vector< double >& independentVariables )
140  {
141  // Check if the correct number of aerodynamic coefficients is provided.
142  if( independentVariables.size( ) != numberOfIndependentVariables_ )
143  {
144  throw std::runtime_error(
145  "Error in CustomAerodynamicCoefficientInterface, number of input variables is inconsistent " +
146  boost::lexical_cast< std::string >( independentVariables.size( ) ) + ", " +
147  boost::lexical_cast< std::string >( numberOfIndependentVariables_ ) );
148  }
149 
150  // Update current coefficients.
151  Eigen::Vector6d currentCoefficients = coefficientFunction_(
152  independentVariables );
153  currentForceCoefficients_ = currentCoefficients.segment( 0, 3 );
154  currentMomentCoefficients_ = currentCoefficients.segment( 3, 3 );
155  }
