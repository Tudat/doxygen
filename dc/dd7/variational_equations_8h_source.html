58  template< typename ParameterType >
59  VariationalEquations(
60  const std::map< IntegratedStateType, orbit_determination::StateDerivativePartialsMap >
61  stateDerivativePartialList,
62  const std::shared_ptr< estimatable_parameters::EstimatableParameterSet< ParameterType > > parametersToEstimate,
63  const std::map< IntegratedStateType, int >& stateTypeStartIndices ):
64  stateDerivativePartialList_( stateDerivativePartialList ), stateTypeStartIndices_( stateTypeStartIndices )
65  {
66  dynamicalStatesToEstimate_ =
67  estimatable_parameters::getListOfInitialDynamicalStateParametersEstimate< ParameterType >(
68  parametersToEstimate );
69 
70  // Get size of dynamical state to estimate
71  numberOfParameterValues_ = estimatable_parameters::getSingleArcParameterSetSize( parametersToEstimate );
72  totalDynamicalStateSize_ = 0;
73  for( std::map< IntegratedStateType, orbit_determination::StateDerivativePartialsMap >::iterator
74  partialTypeIterator = stateDerivativePartialList_.begin( );
75  partialTypeIterator != stateDerivativePartialList_.end( ); partialTypeIterator++ )
76  {
77 
78  if( dynamicalStatesToEstimate_.count( partialTypeIterator->first ) == 0 )
79  {
80  std::string errorMessage = "Error when making variational equations object, found no state to estimate of type " +
81  std::to_string( partialTypeIterator->first );
82  throw std::runtime_error( errorMessage );
83  }
84  else if( dynamicalStatesToEstimate_.at( partialTypeIterator->first ).size( ) !=
85  partialTypeIterator->second.size( ) )
86  {
87  throw std::runtime_error( "Error when making variational equations object, input partial list size is inconsistent" );
88  }
89 
90  totalDynamicalStateSize_ +=
91  getSingleIntegrationSize( partialTypeIterator->first ) * partialTypeIterator->second.size( );
92  }
93 
94  // Initialize matrices.
95  variationalMatrix_ = Eigen::MatrixXd::Zero( totalDynamicalStateSize_, totalDynamicalStateSize_ );
96  variationalParameterMatrix_ =
97  Eigen::MatrixXd::Zero( totalDynamicalStateSize_, numberOfParameterValues_ - totalDynamicalStateSize_ );
98 
99  // Set parameter partial functions.
100  setStatePartialFunctionList( );
101  setTranslationalStatePartialFrameScalingFunctions( parametersToEstimate );
102  setParameterPartialFunctionList( parametersToEstimate );
103  }

133  template< typename StateScalarType >
134  void getParameterPartialMatrix(
135  Eigen::Block< Eigen::Matrix< StateScalarType, Eigen::Dynamic, Eigen::Dynamic > > currentMatrixDerivative )
136  {
137  // Initialize matrix to zeros
138  variationalParameterMatrix_.setZero( );
139 
140 
141  // Iterate over all bodies undergoing accelerations for which initial condition is to be estimated.
142  for( std::map< IntegratedStateType, std::vector< std::multimap< std::pair< int, int >,
143  std::function< void( Eigen::Block< Eigen::MatrixXd > ) > > > >::iterator typeIterator =
144  parameterPartialList_.begin( ); typeIterator != parameterPartialList_.end( ); typeIterator++ )
145  {
146  int startIndex = stateTypeStartIndices_.at( typeIterator->first );
147  int currentStateSize = getSingleIntegrationSize( typeIterator->first );
148  int entriesToSkipPerEntry = currentStateSize -
149  currentStateSize / getSingleIntegrationDifferentialEquationOrder( typeIterator->first );
150 
151  // Iterate over all bodies being estimated.
152  for( unsigned int i = 0; i < typeIterator->second.size( ); i++ )
153  {
154  // Iterate over all parameter partial functions determined by setParameterPartialFunctionList( )
155  for( functionIterator = typeIterator->second[ i ].begin( );
156  functionIterator != typeIterator->second[ i ].end( );
157  functionIterator++ )
158  {
159  functionIterator->second(
160  variationalParameterMatrix_.block(
161  startIndex + entriesToSkipPerEntry + currentStateSize * i,
162  functionIterator->first.first - totalDynamicalStateSize_,
163  currentStateSize - entriesToSkipPerEntry,
164  functionIterator->first.second ) );
165  }
166  }
167 
168  }
169 
170  currentMatrixDerivative.block( 0, totalDynamicalStateSize_, totalDynamicalStateSize_,
171  numberOfParameterValues_ - totalDynamicalStateSize_ ) +=
172  variationalParameterMatrix_.template cast< StateScalarType >( );
173 
174  }

247  template< typename CurrentParameterType >
248  void addParameterPartialToList(
249  const std::map< int, std::shared_ptr< estimatable_parameters::EstimatableParameter< CurrentParameterType > > >&
250  parameterList,
251  const std::shared_ptr< orbit_determination::StateDerivativePartial > partialObject,
252  std::multimap< std::pair< int, int >, std::function< void( Eigen::Block< Eigen::MatrixXd > ) > >&
253  functionListOfBody,
254  const int totalParameterVectorIndicesToSubtract = 0 )
255  {
256  using namespace acceleration_partials;
257 
258  // Iterate over all parameters.
259  for( typename std::map< int,
260  std::shared_ptr< estimatable_parameters::EstimatableParameter< CurrentParameterType > > >::const_iterator
261  parameterIterator = parameterList.begin( ); parameterIterator != parameterList.end( ); parameterIterator++ )
262  {
263  // Add current parameter to list of partials to be computed for current acceleration (if dependency exists)
264  int functionToEvaluate =
265  partialObject->setParameterPartialUpdateFunction( parameterIterator->second );
266 
267  // If function is non-nullptr, add to list
268  if( functionToEvaluate != 0 )
269  {
270  // Make pair of indices for generating parameter partial matrix:
271  //first is start column in matrix, second is number of entries (1 for double parameter)
272  std::pair< int, int > indexPair = std::make_pair(
273  parameterIterator->first - totalParameterVectorIndicesToSubtract, functionToEvaluate );
274 
275  // Add to list.
276  functionListOfBody.insert(
277  std::pair< std::pair< int, int >, std::function< void( Eigen::Block< Eigen::MatrixXd > ) > >
278  ( indexPair, std::bind(
279  static_cast< void ( orbit_determination::StateDerivativePartial::* )
280  ( const std::shared_ptr<
281  estimatable_parameters::EstimatableParameter< CurrentParameterType > >,
282  Eigen::Block< Eigen::MatrixXd > )>
283  ( &orbit_determination::StateDerivativePartial::getCurrentParameterPartial ),
284  partialObject, parameterIterator->second, std::placeholders::_1 ) ) );
285  }
286  }

297  template< typename ParameterType >
298  void setParameterPartialFunctionList(
299  const std::shared_ptr< estimatable_parameters::EstimatableParameterSet< ParameterType > >
300  parametersToEstimate )
301  {
302  // Get double parameters.
303  std::map< int, std::shared_ptr< estimatable_parameters::EstimatableParameter< double > > >
304  doubleParametersToEstimate =
305  parametersToEstimate->getDoubleParameters( );
306 
307  // Get vector parameters.
308  std::map< int, std::shared_ptr< estimatable_parameters::EstimatableParameter< Eigen::VectorXd > > >
309  vectorParametersToEstimate =
310  parametersToEstimate->getVectorParameters( );
311 
312  int totalParameterVectorIndicesToSubtract = parametersToEstimate->getInitialDynamicalStateParameterSize( ) -
313  estimatable_parameters::getSingleArcInitialDynamicalStateParameterSetSize( parametersToEstimate );
314 
315  for( std::map< propagators::IntegratedStateType,
316  orbit_determination::StateDerivativePartialsMap >::iterator
317  stateDerivativeTypeIterator = stateDerivativePartialList_.begin( );
318  stateDerivativeTypeIterator != stateDerivativePartialList_.end( );
319  stateDerivativeTypeIterator++ )
320  {
321 
322  // Initialize vector of lists to correct size.
323  parameterPartialList_[ stateDerivativeTypeIterator->first ].resize(
324  stateDerivativeTypeIterator->second.size( ) );
325 
326  // Iterate over all bodies of which initial position is being estimated.
327  for( unsigned int i = 0; i < stateDerivativeTypeIterator->second.size( ); i++ )
328  {
329  // Initialize list of parameter partial functions for single body.
330  std::multimap< std::pair< int, int >, std::function< void( Eigen::Block< Eigen::MatrixXd > ) > >
331  functionListOfBody;
332 
333  // Iterate over all accelerations due to this body on current body.
334  for( unsigned int j = 0; j < stateDerivativeTypeIterator->second.at( i ).size( ); j++ )
335  {
336  addParameterPartialToList< double >(
337  doubleParametersToEstimate, stateDerivativeTypeIterator->second.at( i ).at( j ),
338  functionListOfBody, totalParameterVectorIndicesToSubtract );
339  addParameterPartialToList< Eigen::VectorXd >(
340  vectorParametersToEstimate, stateDerivativeTypeIterator->second.at( i ).at( j ),
341  functionListOfBody, totalParameterVectorIndicesToSubtract );
342  }
343 
344  // Add generated parameter partial list of current body.
345  parameterPartialList_[ stateDerivativeTypeIterator->first ][ i ] = functionListOfBody;
346  }
347  }

357  template< typename ParameterType >
358  void setTranslationalStatePartialFrameScalingFunctions(
359  const std::shared_ptr< estimatable_parameters::EstimatableParameterSet< ParameterType > >
360  parametersToEstimate )
361  {
362  std::vector< std::shared_ptr< estimatable_parameters::EstimatableParameter<
363  Eigen::Matrix< ParameterType, Eigen::Dynamic, 1 > > > > initialDynamicalParameters =
364  parametersToEstimate->getEstimatedInitialStateParameters( );
365 
366  std::vector< std::string > propagatedBodies;
367  std::vector< std::string > centralBodies;
368 
369  // Retrieve propagated bodies and central bodies of estimation.
370  for( unsigned int i = 0; i < initialDynamicalParameters.size( ); i++ )
371  {
372  if( initialDynamicalParameters.at( i )->getParameterName( ).first == estimatable_parameters::initial_body_state )
373  {
374  propagatedBodies.push_back(
375  initialDynamicalParameters.at( i )->getParameterName( ).second.first );
376  centralBodies.push_back( std::dynamic_pointer_cast
377  < estimatable_parameters::InitialTranslationalStateParameter< ParameterType > >(
378  initialDynamicalParameters.at( i ) )->getCentralBody( ) );
379  }
380  else if( initialDynamicalParameters.at( i )->getParameterName( ).first == estimatable_parameters::arc_wise_initial_body_state )
381  {
382  propagatedBodies.push_back(
383  initialDynamicalParameters.at( i )->getParameterName( ).second.first );
384  centralBodies.push_back( std::dynamic_pointer_cast< estimatable_parameters::ArcWiseInitialTranslationalStateParameter< ParameterType > >(
385  initialDynamicalParameters.at( i ) )->getCentralBody( ) );
386  }
387  else
388  {
389  throw std::runtime_error( "Error when settinf up variational equations, did not recognize initial state type" );
390  }
391  }
392 
393  // Get order in which ephemerides were to be updated.
394  std::vector< std::string > updateOrder = determineEphemerisUpdateorder(
395  propagatedBodies, centralBodies, centralBodies );
396 
397  // Iterate over central bodies and propagated bodies and check for dependencies
398  for( int i = updateOrder.size( ) - 1; i >= 0 ; i-- )
399  {
400  int currentBodyIndex = std::distance(
401  propagatedBodies.begin( ),
402  std::find( propagatedBodies.begin( ), propagatedBodies.end( ), updateOrder.at( i ) ) );
403  for( unsigned int j = 0; j < propagatedBodies.size( ); j++ )
404  {
405  if( centralBodies.at( currentBodyIndex ) == propagatedBodies.at( j ) )
406  {
407 
408  statePartialAdditionIndices_.push_back(
409  std::make_pair( stateTypeStartIndices_[ propagators::translational_state ] +
410  currentBodyIndex * propagators::getSingleIntegrationSize( propagators::translational_state ),
411  stateTypeStartIndices_[ propagators::translational_state ] +
412  j * propagators::getSingleIntegrationSize( propagators::translational_state ) ) );
413  }
414  }
415  }
416  }
