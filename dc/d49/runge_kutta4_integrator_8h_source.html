106  virtual StateType performIntegrationStep( const TimeStepType stepSize )
107  {
108  lastIndependentVariable_ = currentIndependentVariable_;
109  lastState_ = currentState_;
110 
111  // Calculate k1-k4.
112  StateDerivativeType k1, k2, k3, k4;
113  for ( unsigned int i = 1; i <= 4; i++ )
114  {
115  IndependentVariableType time;
116  StateType state;
117  switch ( i ) {
118  case 1:
119  time = currentIndependentVariable_;
120  state = currentState_;
121  k1 = stepSize * this->stateDerivativeFunction_( time, state );
122  break;
123  case 2:
124  time = currentIndependentVariable_ + stepSize / 2.0;
125  state = static_cast< StateType >( currentState_ + k1 / 2.0 );
126  k2 = stepSize * this->stateDerivativeFunction_( time, state );
127  break;
128  case 3:
129  time = currentIndependentVariable_ + stepSize / 2.0;
130  state = static_cast< StateType >( currentState_ + k2 / 2.0 );
131  k3 = stepSize * this->stateDerivativeFunction_( time, state );
132  break;
133  case 4:
134  time = currentIndependentVariable_ + stepSize;
135  state = static_cast< StateType >( currentState_ + k3 );
136  k4 = stepSize * this->stateDerivativeFunction_( time, state );
137  break;
138  }
139 
140  // Check if propagation should terminate because the propagation termination condition has been reached
141  // while computing k1, k2, k3 or k4. If so, return immediately the current state (not recomputed yet),
142  // which will be discarded.
143  if ( this->propagationTerminationFunction_( static_cast< double >( time ) ) )
144  {
145  this->propagationTerminationConditionReachedDuringStep_ = true;
146  return currentState_;
147  }
148  }
149 
150  stepSize_ = stepSize;
151  currentIndependentVariable_ += stepSize_;
152  currentState_ += ( k1 + 2.0 * k2 + 2.0 * k3 + k4 ) / 6.0;
153 
154  // Return the integration result.
155  return currentState_;
156  }
