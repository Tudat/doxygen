253 template< typename InitialStateParameterType = double >
254 int getSingleArcParameterSetSize(
255  std::shared_ptr< EstimatableParameterSet< InitialStateParameterType > > estimatableParameterSet )
256 {
257  int totalParameterSetSize = estimatableParameterSet->getEstimatedParameterSetSize( );
258  std::vector< std::shared_ptr< EstimatableParameter< Eigen::Matrix< InitialStateParameterType, Eigen::Dynamic, 1 > > > >
259  initialStateParameters = estimatableParameterSet->getEstimatedInitialStateParameters( );
260 
261  for( unsigned int i = 0; i < initialStateParameters.size( ); i++ )
262  {
263  if( initialStateParameters.at( i )->getParameterName( ).first == arc_wise_initial_body_state )
264  {
265  totalParameterSetSize -=
266  ( std::dynamic_pointer_cast< ArcWiseInitialTranslationalStateParameter< InitialStateParameterType > >(
267  initialStateParameters.at( i ) )->getNumberOfStateArcs( ) - 1 ) * 6;
268  }
269  else if( ( initialStateParameters.at( i )->getParameterName( ).first != initial_body_state ) )
270  {
271  throw std::runtime_error(
272  "Error when getting single arc paramater vector, did not recognize initial state parameter " +
273  std::to_string( initialStateParameters.at( i )->getParameterName( ).first ) );
274  }
275  }
276  return totalParameterSetSize;
277 }

301 template< typename InitialStateParameterType >
302 std::vector< double > getMultiArcStateEstimationArcStartTimes(
303  const std::shared_ptr< EstimatableParameterSet< InitialStateParameterType > > estimatableParameters )
304 {
305  // Retrieve initial dynamical parameters.
306  std::vector< std::shared_ptr< EstimatableParameter<
307  Eigen::Matrix< InitialStateParameterType, Eigen::Dynamic, 1 > > > > initialDynamicalParameters =
308  estimatableParameters->getEstimatedInitialStateParameters( );
309 
310  std::vector< double > arcStartTimes;
311 
312  // Iterate over list of bodies of which the partials of the accelerations acting on them are required.
313  for( unsigned int i = 0; i < initialDynamicalParameters.size( ); i++ )
314  {
315  if( initialDynamicalParameters.at( i )->getParameterName( ).first == arc_wise_initial_body_state )
316  {
317  std::shared_ptr< ArcWiseInitialTranslationalStateParameter< InitialStateParameterType > > arcWiseStateParameter =
318  std::dynamic_pointer_cast< ArcWiseInitialTranslationalStateParameter< InitialStateParameterType > >(
319  initialDynamicalParameters.at( i ) );
320  if( arcWiseStateParameter == nullptr )
321  {
322  throw std::runtime_error( "Error when getting arc times from estimated parameters, parameter is inconsistent" );
323  }
324 
325  if( arcStartTimes.size( ) == 0 )
326  {
327  arcStartTimes = arcWiseStateParameter->getArcStartTimes( );
328  }
329  else
330  {
331  // Check arc consistency
332  std::vector< double > currentArcStartTimes = arcWiseStateParameter->getArcStartTimes( );
333  if( currentArcStartTimes.size( ) != arcStartTimes.size( ) )
334  {
335  throw std::runtime_error(
336  "Error when getting arc start times for parameters, number of arcs is inconsistent" );
337  }
338  else
339  {
340  for( unsigned int j = 0; j < arcStartTimes.size( ); j++ )
341  {
342  if( arcStartTimes.at( j ) != currentArcStartTimes.at( j ) )
343  {
344  throw std::runtime_error(
345  "Error when getting arc start times for parameters, arc times are inconsistent" );
346  }
347  }
348  }
349  }
350 
351  }
352  else
353  {
354  throw std::runtime_error( "Error when getting arc times from estimated parameters, soingle arc dynamics found" );
355  }
356  }
357 
358  return arcStartTimes;
359 }
