340 template< int ObservationSize >
341 ObservationBiasTypes getObservationBiasType(
342  const boost::shared_ptr< ObservationBias< ObservationSize > > biasObject )
343 {
344  ObservationBiasTypes biasType;
345 
346  // Check available bias types
347  if( boost::dynamic_pointer_cast< ConstantObservationBias< ObservationSize > >( biasObject ) != NULL )
348  {
349  biasType = constant_absolute_bias;
350  }
351  else if( boost::dynamic_pointer_cast< ConstantRelativeObservationBias< ObservationSize > >( biasObject ) != NULL )
352  {
353  biasType = constant_relative_bias;
354  }
355  else if( boost::dynamic_pointer_cast< MultiTypeObservationBias< ObservationSize > >( biasObject ) != NULL )
356  {
357  biasType = multiple_observation_biases;
358  }
359  else
360  {
361  std::string errorMessage = "Error, did not recognize observation bias when retrieveing bias type";
362  throw std::runtime_error( errorMessage );
363  }
364  return biasType;
365  }
