641 template< int ObservationSize >
642 ObservationBiasTypes getObservationBiasType(
643  const std::shared_ptr< ObservationBias< ObservationSize > > biasObject )
644 {
645  ObservationBiasTypes biasType;
646 
647  // Check available bias types
648  if( std::dynamic_pointer_cast< ConstantObservationBias< ObservationSize > >( biasObject ) != nullptr )
649  {
650  biasType = constant_absolute_bias;
651  }
652  else if( std::dynamic_pointer_cast< ConstantArcWiseObservationBias< ObservationSize > >( biasObject ) != nullptr )
653  {
654  biasType = arc_wise_constant_absolute_bias;
655  }
656  else if( std::dynamic_pointer_cast< ConstantRelativeObservationBias< ObservationSize > >( biasObject ) != nullptr )
657  {
658  biasType = constant_relative_bias;
659  }
660  else if( std::dynamic_pointer_cast< ConstantRelativeArcWiseObservationBias< ObservationSize > >( biasObject ) != nullptr )
661  {
662  biasType = arc_wise_constant_relative_bias;
663  }
664  else if( std::dynamic_pointer_cast< MultiTypeObservationBias< ObservationSize > >( biasObject ) != nullptr )
665  {
666  biasType = multiple_observation_biases;
667  }
668  else
669  {
670  std::string errorMessage = "Error, did not recognize observation bias when retrieveing bias type";
671  throw std::runtime_error( errorMessage );
672  }
673  return biasType;
674 }
