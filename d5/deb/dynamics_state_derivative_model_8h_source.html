67  const TimeType, const std::unordered_map< IntegratedStateType, Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > >&,

75 
76  // Iterate over vector of state derivative models, check validity, set member variable map
77  // stateDerivativeModels_ and size indices.
78  for( unsigned int i = 0; i < stateDerivativeModels.size( ); i++ )
79  {
80  if( i > 0 )
81  {

83  stateDerivativeModels.at( i )->getIntegratedStateType( ) )

89  std::to_string( stateDerivativeModels.at( i )->getIntegratedStateType( ) )
90  + " entries are non-contiguous" );
91  }
92  }
93 
94  // Check uniqueness of state derivative type calculator in list
95  if( std::find( stateTypeList.begin( ), stateTypeList.end( ),
96  stateDerivativeModels.at( i )->getIntegratedStateType( ) ) == stateTypeList.end( ) )
97  {
98  stateTypeSize_[ stateDerivativeModels.at( i )->getIntegratedStateType( ) ] = 0;
99  stateTypeStartIndex_[ stateDerivativeModels.at( i )->getIntegratedStateType( ) ]
100  = totalStateSize_;
101  }
102  stateTypeList.push_back( stateDerivativeModels.at( i )->getIntegratedStateType( ) );
103 
104  // Set state part sizes
105  stateIndices_[ stateDerivativeModels.at( i )->getIntegratedStateType( ) ].push_back(
106  std::make_pair( totalStateSize_, stateDerivativeModels.at( i )->getStateSize( ) ) );
107  totalStateSize_ += stateDerivativeModels.at( i )->getStateSize( );
108

111 
112  // Set current model in member map.
113  stateDerivativeModels_[ stateDerivativeModels.at( i )->getIntegratedStateType( ) ].push_back(
114  stateDerivativeModels.at( i ) );

135  StateType computeStateDerivative( const TimeType time, const StateType& state )
136  {
137  //std::cout << "Computing state derivative: " << state.transpose( ) << std::endl;
138  // Initialize state derivative
139  if( stateDerivative_.rows( ) != state.rows( ) || stateDerivative_.cols( ) != state.cols( ) )
140  {
141  stateDerivative_.resize( state.rows( ), state.cols( ) );
142  }
143 
144  // If dynamical equations are integrated, update the environment with the current state.
145  if( evaluateDynamicsEquations_ )
146  {
147  // Iterate over all types of equations.
148  for( stateDerivativeModelsIterator_ = stateDerivativeModels_.begin( );
149  stateDerivativeModelsIterator_ != stateDerivativeModels_.end( );
150  stateDerivativeModelsIterator_++ )
151 
152  {
153  for( unsigned int i = 0; i < stateDerivativeModelsIterator_->second.size( ); i++ )
154  {
155  stateDerivativeModelsIterator_->second.at( i )->clearStateDerivativeModel( );
156  }
157  }
158 
159  convertCurrentStateToGlobalRepresentationPerType( state, time, evaluateVariationalEquations_ );
160  environmentUpdateFunction_( time, currentStatesPerTypeInConventionalRepresentation_,
161  integratedStatesFromEnvironment_ );
162  }
163  else
164  {
165  environmentUpdateFunction_(
166  time, std::unordered_map<
167  IntegratedStateType, Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > >( ),
168  integratedStatesFromEnvironment_ );
169  }
170 
171  if( evaluateVariationalEquations_ )
172  {
173  variationalEquations_->clearPartials( );
174  }
175 
176  // If dynamical equations are integrated, evaluate dynamics state derivatives.
177  std::pair< int, int > currentIndices;
178  if( evaluateDynamicsEquations_ )
179  {
180  // Iterate over all types of equations.
181  for( stateDerivativeModelsIterator_ = stateDerivativeModels_.begin( );
182  stateDerivativeModelsIterator_ != stateDerivativeModels_.end( );
183  stateDerivativeModelsIterator_++ )
184 
185  {
186  for( unsigned int i = 0; i < stateDerivativeModelsIterator_->second.size( ); i++ )
187  {
188  // Update state derivative models
189  stateDerivativeModelsIterator_->second.at( i )->updateStateDerivativeModel( time );
190  }
191  }
192 
193  for( stateDerivativeModelsIterator_ = stateDerivativeModels_.begin( );
194  stateDerivativeModelsIterator_ != stateDerivativeModels_.end( );
195  stateDerivativeModelsIterator_++ )
196 
197  {
198  for( unsigned int i = 0; i < stateDerivativeModelsIterator_->second.size( ); i++ )
199  {
200  // Evaluate and set current dynamical state derivative
201  currentIndices = stateIndices_.at( stateDerivativeModelsIterator_->first ).at( i );
202 
203  stateDerivativeModelsIterator_->second.at( i )->calculateSystemStateDerivative(
204  time, state.block( currentIndices.first, dynamicsStartColumn_, currentIndices.second, 1 ),
205  stateDerivative_.block( currentIndices.first, dynamicsStartColumn_, currentIndices.second, 1 ) );
206 
207  }
208  }
209  }
210 
211 
212  // If variational equations are to be integrated: evaluate and set.
213  if( evaluateVariationalEquations_ )
214  {
215  variationalEquations_->updatePartials( time );
216 
217  variationalEquations_->evaluateVariationalEquations< StateScalarType >(
218  time, state.block( 0, 0, totalStateSize_, variationalEquations_->getNumberOfParameterValues( ) ),
219  stateDerivative_.block( 0, 0, totalStateSize_, variationalEquations_->getNumberOfParameterValues( ) ) );
220  }
221 
222  return stateDerivative_;
223  }

389  void updateStateDerivativeModelSettings(
390  const Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > initialBodyStates )
391  {
392  // Iterate over all dynamics types
393  for( stateDerivativeModelsIterator_ = stateDerivativeModels_.begin( );
394  stateDerivativeModelsIterator_ != stateDerivativeModels_.end( );
395  stateDerivativeModelsIterator_++ )
396  {
397  switch( stateDerivativeModelsIterator_->first )
398  {
399  case transational_state:
400  {
401  for( unsigned int i = 0; i < stateDerivativeModelsIterator_->second.size( ); i++ )
402  {
403  boost::shared_ptr< NBodyStateDerivative< StateScalarType, TimeType > > currentTranslationalStateDerivative =
404  boost::dynamic_pointer_cast< NBodyStateDerivative< StateScalarType, TimeType > >(
405  stateDerivativeModelsIterator_->second.at( i ) );
406  switch( currentTranslationalStateDerivative->getPropagatorType( ) )
407  {
408  case cowell:
409  break;
410  case encke:
411  throw std::runtime_error( "Error, reference orbit not reset in Encke propagator" );
412  break;
413  case gauss_keplerian:
414  break;
415  default:
416  throw std::runtime_error( "Error when updating state derivative model settings, did not recognize translational propagator type" );
417  break;
418  }
419  }
420  }
421  case body_mass_state:
422  break;
423  case custom_state:
424  break;
425  default:
426  throw std::runtime_error( "Error when updating state derivative model settings, did not recognize dynamics type" );
427  break;
428  }
429  }
