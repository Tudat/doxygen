225 void setVariationalEquationsSolution(
226  std::map< TimeType, Eigen::Matrix< StateScalarType, Eigen::Dynamic, Eigen::Dynamic > >&
227  numericalIntegrationResult,
228  std::vector< std::map< double, Eigen::MatrixXd > >& variationalEquationsSolution,
229  const std::pair< int, int > stateTransitionStartIndices,
230  const std::pair< int, int > sensitivityStartIndices,
231  const int stateTransitionMatrixSize,
232  const int parameterSetSize )
233 {
234  variationalEquationsSolution.clear( );
235  variationalEquationsSolution.resize( 2 );
236 
237  for( typename std::map< TimeType, Eigen::Matrix< StateScalarType, Eigen::Dynamic, Eigen::Dynamic > >::iterator
238  integrationIterator = numericalIntegrationResult.begin( );
239  integrationIterator != numericalIntegrationResult.end( ); )
240  {
241  // Set result for state transition matrix in each time step.
242  variationalEquationsSolution[ 0 ][ integrationIterator->first ] =
243  ( integrationIterator->second.block( stateTransitionStartIndices.first, stateTransitionStartIndices.second,
244  stateTransitionMatrixSize,
245  stateTransitionMatrixSize ) ).template cast< double >( );
246 
247  // Set result for sensitivity matrix in each time step.
248  variationalEquationsSolution[ 1 ][ integrationIterator->first ] =
249  ( integrationIterator->second.block( sensitivityStartIndices.first, sensitivityStartIndices.second,
250  stateTransitionMatrixSize,
251  parameterSetSize -
252  stateTransitionMatrixSize ) ).template cast< double >( );
253  numericalIntegrationResult.erase( integrationIterator++ );
254  }

287  const std::shared_ptr< estimatable_parameters::EstimatableParameterSet< StateScalarType > > parametersToEstimate )
288 {
289  bool isInputConsistent = 1;
290 
291  // Check type of dynamics
292  switch( propagatorSettings->getStateType( ) )
293  {
294  case translational_state:
295  {
296  std::shared_ptr< TranslationalStatePropagatorSettings< StateScalarType > > translationalPropagatorSettings =
297  std::dynamic_pointer_cast< TranslationalStatePropagatorSettings< StateScalarType > >( propagatorSettings );
298 
299  // Retrieve estimated and propagated translational states, and check equality.
300  std::vector< std::string > propagatedBodies = translationalPropagatorSettings->bodiesToIntegrate_;
301  std::vector< std::string > estimatedBodies = estimatable_parameters::getListOfBodiesWithTranslationalStateToEstimate(
302  parametersToEstimate );
303 
304  if( static_cast< unsigned int >( translationalPropagatorSettings->getPropagatedStateSize( ) ) !=
305  propagatedBodies.size( ) * 6 )
306  {
307  throw std::runtime_error( "Error when propagating variational equations, tranbbslational state vectors not of size 6." );
308  }
309 
310  if( propagatedBodies.size( ) != estimatedBodies.size( ) )
311  {
312  std::string errorMessage = "Error, propagated and estimated body vector sizes are inconsistent " +
313  std::to_string( propagatedBodies.size( ) ) + " " +
314  std::to_string( estimatedBodies.size( ) );
315  throw std::runtime_error( errorMessage );
316  isInputConsistent = 0;
317  }
318  else
319  {
320  for( unsigned int i = 0; i < propagatedBodies.size( ); i++ )
321  {
322  if( propagatedBodies.at( i ) != estimatedBodies.at( i ) )
323  {
324  std::string errorMessage = "Error, propagated and estimated body vectors inconsistent at index" +
325  std::string( propagatedBodies.at( i ) ) + " " +
326  std::string( estimatedBodies.at( i ) );
327  throw std::runtime_error( errorMessage );
328  isInputConsistent = 0;
329  }
330  }
331 
332  }
333  break;
334  }
335  default:
336  std::string errorMessage = "Error, cannot yet check consistency of propagator settings for type " +
337  std::to_string( propagatorSettings->getStateType( ) );
338  throw std::runtime_error( errorMessage );
339  }
340  return isInputConsistent;
341 }

353 template< typename StateScalarType = double, typename TimeType = double >
354 bool checkMultiArcPropagatorSettingsAndParameterEstimationConsistency(
355  const std::shared_ptr< MultiArcPropagatorSettings< StateScalarType > > propagatorSettings,
356  const std::shared_ptr< estimatable_parameters::EstimatableParameterSet< StateScalarType > > parametersToEstimate,
357  const std::vector< double > arcStartTimes )
358 {
359  bool isInputConsistent = 1;
360 
361  // Get list of objets and associated bodies to estimate initial arc-wise translational states
362  typedef std::map< std::string, std::shared_ptr< estimatable_parameters::EstimatableParameter<
363  Eigen::Matrix< StateScalarType, Eigen::Dynamic, 1 > > > > ArcWiseParameterList;
364  ArcWiseParameterList estimatedBodies = estimatable_parameters::getListOfBodiesWithTranslationalMultiArcStateToEstimate(
365  parametersToEstimate );
366 
367  // Iterate over all parameters and check consistency
368  for( typename ArcWiseParameterList::const_iterator parameterIterator = estimatedBodies.begin( ); parameterIterator !=
369  estimatedBodies.end( ); parameterIterator++ )
370  {
371  // Get arc start times of current parameter
372  std::vector< double > parameterArcStartTimes =
373  std::dynamic_pointer_cast< estimatable_parameters::
374  ArcWiseInitialTranslationalStateParameter< StateScalarType > >(
375  parameterIterator->second )->getArcStartTimes( );
376 
377  // Check if arc times are (almost) exactly the same
378  if( arcStartTimes.size( ) != parameterArcStartTimes.size( ) )
379  {
380  isInputConsistent = false;
381  throw std::runtime_error( "Error, arc times for " + parameterIterator->first + " have incompatible size with estimation" );
382  }
383  else
384  {
385  for( unsigned int i = 0; i < arcStartTimes.size( ); i++ )
386  {
387  if( std::fabs( arcStartTimes.at( i ) - parameterArcStartTimes.at( i ) ) >
388  std::max( 4.0 * parameterArcStartTimes.at( i ) * std::numeric_limits< double >::epsilon( ), 1.0E-12 ) )
389  {
390  isInputConsistent = false;
391  throw std::runtime_error( "Error, arc time for " + parameterIterator->first + " is incompatible with estimation" );
392  }
393  }
394  }
395  }
396 
397  std::map< IntegratedStateType, std::vector< std::string > > propagatedStateTypes;
398 
399  // Iterate over each arc in propagator settings and check consistency
400  for( int arc = 0; arc < propagatorSettings->getNmberOfArcs( ); arc++ )
401  {
402  // Check type of dynamics
403  switch( propagatorSettings->getSingleArcSettings( ).at( arc )->getStateType( ) )
404  {
405  case translational_state:
406  {
407  std::shared_ptr< TranslationalStatePropagatorSettings< StateScalarType > > translationalPropagatorSettings =
408  std::dynamic_pointer_cast< TranslationalStatePropagatorSettings< StateScalarType > >(
409  propagatorSettings->getSingleArcSettings( ).at( arc ) );
410 
411  // Retrieve estimated and propagated translational states, and check equality.
412  std::vector< std::string > propagatedBodies = translationalPropagatorSettings->bodiesToIntegrate_;
413  if( arc == 0 )
414  {
415  propagatedStateTypes[ translational_state ] = propagatedBodies;
416  }
417  else
418  {
419  if( propagatedBodies.size( ) != propagatedStateTypes.at( translational_state ).size( ) )
420  {
421  isInputConsistent = false;
422  std::string errorMessage = "Error, propagated body vector sizes are inconsistent between arcs " +
423  std::to_string( propagatedBodies.size( ) ) + " " +
424  std::to_string( propagatedStateTypes[ translational_state ].size( ) ) +
425  " when checking multi-arc estimation/propagation consistency";
426  throw std::runtime_error( errorMessage );
427  }
428  else
429  {
430  for( unsigned int i = 0; i < propagatedBodies.size( ); i++ )
431  {
432  if( propagatedBodies.at( i ) != propagatedStateTypes[ translational_state ].at( i ) )
433  {
434  isInputConsistent = false;
435  std::string errorMessage = "Error, propagated body vector sizes are inconsistent between arcs at index " +
436  std::to_string( i ) + " " +
437  std::string( propagatedBodies.at( i ) ) + " " +
438  std::string( propagatedStateTypes[ translational_state ].at( i ) ) +
439  " when checking multi-arc estimation/propagation consistency";
440  throw std::runtime_error( errorMessage );
441  }
442  }
443 
444  }
445  }
446  break;
447  }
448  default:
449  std::string errorMessage = "Error, cannot yet check consistency of multi-arc propagator settings for type " +
450  std::to_string( propagatorSettings->getSingleArcSettings( ).at( arc )->getStateType( ) );
451  throw std::runtime_error( errorMessage );
452  }
453  }
454 
455  if( estimatedBodies.size( ) != propagatedStateTypes[ translational_state ].size( ) )
456  {
457  isInputConsistent = false;
458  std::string errorMessage = "Error, propagated body vector sizes are inconsistent " +
459  std::to_string( propagatedStateTypes[ translational_state ].size( ) ) + " " +
460  std::to_string( estimatedBodies.size( ) ) +
461  " when checking multi-arc estimation/propagation consistency";
462  throw std::runtime_error( errorMessage );
463 
464  for( unsigned int i = 0; i < propagatedStateTypes[ translational_state ].size( ); i++ )
465  {
466  if( estimatedBodies.count( propagatedStateTypes[ translational_state ].at( i ) ) == 0 )
467  {
468  isInputConsistent = false;
469  std::string errorMessage = "Error, propagated body " +
470  std::string( propagatedStateTypes[ translational_state ].at( i ) ) + " " +
471  " not found in estimated body list when checking multi-arc estimation/propagation consistency";
472  throw std::runtime_error( errorMessage );
473  }
474  }
475  }
476 
477  return isInputConsistent;
478 }
