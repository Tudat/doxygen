314  throw std::runtime_error( errorMessage );
315  isInputConsistent = 0;
316  }
317  }
318 
319  }

321  }

323  std::string errorMessage = "Error, cannot yet check consistency of propagator settings for type " +

365  if( arcStartTimes.size( ) != parameterArcStartTimes.size( ) )

381  }
382  }

385 
386  // Iterate over each arc in propagator settings and check consistency
387  for( int arc = 0; arc < propagatorSettings->getNmberOfArcs( ); arc++ )
388  {
389  // Check type of dynamics
390  switch( propagatorSettings->getSingleArcSettings( ).at( arc )->getStateType( ) )
391  {
392  case transational_state:
393  {
394  boost::shared_ptr< TranslationalStatePropagatorSettings< StateScalarType > > translationalPropagatorSettings =
395  boost::dynamic_pointer_cast< TranslationalStatePropagatorSettings< StateScalarType > >(

403  }
404  else

406  if( propagatedBodies.size( ) != propagatedStateTypes.at( transational_state ).size( ) )
407  {
408  isInputConsistent = false;
409  std::string errorMessage = "Error, propagated body vector sizes are inconsistent between arcs " +
410  boost::lexical_cast< std::string >( propagatedBodies.size( ) ) + " " +
411  boost::lexical_cast< std::string >( propagatedStateTypes[ transational_state ].size( ) ) +

413  throw std::runtime_error( errorMessage );
414  }
415  else
416  {
417  for( unsigned int i = 0; i < propagatedBodies.size( ); i++ )
418  {
419  if( propagatedBodies.at( i ) != propagatedStateTypes[ transational_state ].at( i ) )
420  {
421  isInputConsistent = false;
422  std::string errorMessage = "Error, propagated body vector sizes are inconsistent between arcs at index " +
423  boost::lexical_cast< std::string >( i ) + " " +
424  boost::lexical_cast< std::string >( propagatedBodies.at( i ) ) + " " +
425  boost::lexical_cast< std::string >( propagatedStateTypes[ transational_state ].at( i ) ) +
426  " when checking multi-arc estimation/propagation consistency";
427  throw std::runtime_error( errorMessage );
428  }
429  }
430 
431  }
432  }
433 
434 
435  break;
436  }

438  std::string errorMessage = "Error, cannot yet check consistency of multi-arc propagator settings for type " +
439  boost::lexical_cast< std::string >( propagatorSettings->getSingleArcSettings( ).at( arc )->getStateType( ) );
440  throw std::runtime_error( errorMessage );
441  }
442  }
443 
444  if( estimatedBodies.size( ) != propagatedStateTypes[ transational_state ].size( ) )
445  {
446  isInputConsistent = false;
447  std::string errorMessage = "Error, propagated body vector sizes are inconsistent " +
448  boost::lexical_cast< std::string >( propagatedStateTypes[ transational_state ].size( ) ) + " " +

457  isInputConsistent = false;

459  boost::lexical_cast< std::string >( propagatedStateTypes[ transational_state ].at( i ) ) + " " +
460  " not found in estimated body list when checking multi-arc estimation/propagation consistency";
461  throw std::runtime_error( errorMessage );
462  }
463  }
464  }
