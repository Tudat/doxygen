83 std::string getGlobalFrameOrigin( const NamedBodyMap& bodyMap )
84 {
85  std::string globalFrameOrigin = "SSB";
86 
87  for( NamedBodyMap::const_iterator bodyIterator = bodyMap.begin( ); bodyIterator != bodyMap.end( ); bodyIterator++ )
88  {
89  if( bodyIterator->second->getIsBodyGlobalFrameOrigin( ) == -1 )
90  {
91  throw std::runtime_error( "Error, body " + bodyIterator->first + " does not have global frame origin set" );
92  }
93  else if( bodyIterator->second->getIsBodyGlobalFrameOrigin( ) == 1 )
94  {
95  if( globalFrameOrigin != "SSB" )
96  {
97  throw std::runtime_error( "Error, body " + bodyIterator->first + " found as global frame origin, but body " +
98  globalFrameOrigin + " has already been detected as global frame origin." );
99  }
100  else
101  {
102  globalFrameOrigin = bodyIterator->first;
103  }
104  }
105  }
106  return globalFrameOrigin;
107 }
