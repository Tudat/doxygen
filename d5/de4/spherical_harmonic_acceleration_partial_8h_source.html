189  void wrtGravitationalParameterOfCentralBody( Eigen::MatrixXd& partialMatrix, const int addPartial = 0 )
190  {
191  if( gravitationalParameterFunction_( ) != 0.0 )
192  {
193  if( addPartial == 0 )
194  {
195  partialMatrix = accelerationFunction_( ) / gravitationalParameterFunction_( );
196  }
197  else if( addPartial == 1 )
198  {
199  partialMatrix += accelerationFunction_( ) / gravitationalParameterFunction_( );
200  }
201  else if( addPartial == -1 )
202  {
203  partialMatrix -= accelerationFunction_( ) / gravitationalParameterFunction_( );
204  }
205  else
206  {
207  throw std::runtime_error( "Error when adding partial of spherical harmonic acceleration w.r.t mu, inout is inconsistent" );
208  }
209  }
210  else
211  {
212  throw std::runtime_error( "Error cannot compute partial of spherical harminic gravity w.r.t mu for zero value" );
213  }
214  }
