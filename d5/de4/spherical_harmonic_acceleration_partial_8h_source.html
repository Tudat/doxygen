191  void wrtGravitationalParameterOfCentralBody( Eigen::MatrixXd& partialMatrix, const int addPartial = 0 )
192  {
193  if( gravitationalParameterFunction_( ) != 0.0 )
194  {
195  if( addPartial == 0 )
196  {
197  partialMatrix = accelerationFunction_( ) / gravitationalParameterFunction_( );
198  }
199  else if( addPartial == 1 )
200  {
201  partialMatrix += accelerationFunction_( ) / gravitationalParameterFunction_( );
202  }
203  else if( addPartial == -1 )
204  {
205  partialMatrix -= accelerationFunction_( ) / gravitationalParameterFunction_( );
206  }
207  else
208  {
209  throw std::runtime_error( "Error when adding partial of spherical harmonic acceleration w.r.t mu, inout is inconsistent" );
210  }
211  }
212  else
213  {
214  throw std::runtime_error( "Error cannot compute partial of spherical harminic gravity w.r.t mu for zero value" );
215  }
216  }
