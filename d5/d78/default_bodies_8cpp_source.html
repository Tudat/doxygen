152 boost::shared_ptr< BodySettings > getDefaultSingleBodySettings(
153  const std::string& body,
154  const double initialTime,
155  const double finalTime )
156 {
157  boost::shared_ptr< BodySettings > singleBodySettings = boost::make_shared< BodySettings >( );
158 
159  // Get default settings for each of the environment models in the body.
160  singleBodySettings->atmosphereSettings = getDefaultAtmosphereModelSettings(
161  body, initialTime, finalTime );
162  singleBodySettings->rotationModelSettings = getDefaultRotationModelSettings(
163  body, initialTime, finalTime );
164 
165  if( ( !( initialTime == initialTime ) && ( finalTime == finalTime ) ) ||
166  ( ( initialTime == initialTime ) && !( finalTime == finalTime ) ) )
167  {
168  throw std::runtime_error( "Error when getting default body settings, only one input time is NaN" );
169  }
170  else if( !( initialTime == initialTime ) )
171  {
172  singleBodySettings->ephemerisSettings = getDefaultEphemerisSettings(
173  body );
174  }
175  else
176  {
177  singleBodySettings->ephemerisSettings = getDefaultEphemerisSettings(
178  body, initialTime, finalTime );
179  }
180  singleBodySettings->gravityFieldSettings = getDefaultGravityFieldSettings(
181  body, initialTime, finalTime );
182  singleBodySettings->shapeModelSettings = getDefaultBodyShapeSettings(
183  body, initialTime, finalTime );
184 
185  return singleBodySettings;
186 }
