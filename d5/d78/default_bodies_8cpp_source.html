137 std::shared_ptr< BodySettings > getDefaultSingleBodySettings(
138  const std::string& bodyName,
139  const double initialTime,
140  const double finalTime,
141  const double timeStep )
142 {
143  std::shared_ptr< BodySettings > singleBodySettings = std::make_shared< BodySettings >( );
144 
145  // Get default settings for each of the environment models in the body.
146  singleBodySettings->atmosphereSettings = getDefaultAtmosphereModelSettings(
147  bodyName, initialTime, finalTime );
148  singleBodySettings->rotationModelSettings = getDefaultRotationModelSettings(
149  bodyName, initialTime, finalTime );
150 
151  if( ( !( initialTime == initialTime ) && ( finalTime == finalTime ) ) ||
152  ( ( initialTime == initialTime ) && !( finalTime == finalTime ) ) )
153  {
154  throw std::runtime_error( "Error when getting default body settings, only one input time is NaN" );
155  }
156  else if( !( initialTime == initialTime ) )
157  {
158  singleBodySettings->ephemerisSettings = getDefaultEphemerisSettings(
159  bodyName );
160  }
161  else
162  {
163  singleBodySettings->ephemerisSettings = getDefaultEphemerisSettings(
164  bodyName, initialTime, finalTime, timeStep );
165  }
166  singleBodySettings->gravityFieldSettings = getDefaultGravityFieldSettings(
167  bodyName, initialTime, finalTime );
168  singleBodySettings->shapeModelSettings = getDefaultBodyShapeSettings(
169  bodyName, initialTime, finalTime );
170 
171  return singleBodySettings;
172 }
