51  weightsData.begin( ); observablesIterator != weightsData.end( ); observablesIterator++ )

110  bool isEstimationConverged( const int numberOfIterations, const std::vector< double > rmsResidualHistory )
111  {
112  bool isConverged = 0;
113  if( numberOfIterations >= maximumNumberOfIterations_ )
114  {
115  std::cout << "Maximum number of iterations reached" << std::endl;
116  isConverged = 1;
117  }
118  if( rmsResidualHistory[ rmsResidualHistory.size( ) - 1 ] < minimumResidual_ )
119  {
120  std::cout << "Required residual level achieved" << std::endl;
121  isConverged = 1;
122  }
123  if( ( std::distance( rmsResidualHistory.begin( ), std::max_element(
124  rmsResidualHistory.begin( ), rmsResidualHistory.end( ) ) ) - rmsResidualHistory.size( ) ) <
125  numberOfIterationsWithoutImprovement_ )
126  {
127  std::cout << "Too many iterations without parameter improvement" << std::endl;
128  isConverged = 1;
129  }
130  if( rmsResidualHistory.size( ) > 1 )
131  {
132  if( std::fabs( rmsResidualHistory.at( rmsResidualHistory.size( ) - 1 ) -
133  rmsResidualHistory.at( rmsResidualHistory.size( ) - 2 ) ) < minimumResidualChange_ )
134  {
135  isConverged = 1;
136  }
137  }
138  return isConverged;
139  }

350  residualsAndPartials.first = Eigen::VectorXd::Zero( totalObservationSize );

353  int startIndex = 0;

361  for( typename SingleObservablePodInputType::const_iterator dataIterator = observablesIterator->second.begin( );

461  std::shared_ptr< PodOutput< ObservationScalarType > > estimateParameters(
462  const std::shared_ptr< PodInput< ObservationScalarType, TimeType > >& podInput,
463  const std::shared_ptr< EstimationConvergenceChecker > convergenceChecker =
464  std::make_shared< EstimationConvergenceChecker >( ) )
465  {
466  currentParameterEstimate_ = parametersToEstimate_->template getFullParameterValues< ObservationScalarType >( );
467 
468  // Get size of parameter vector and number of observations (total and per type)
469  int parameterVectorSize = currentParameterEstimate_.size( );
470  std::pair< std::map< observation_models::ObservableType, int >, int > observationNumberPair =
471  getNumberOfObservationsPerObservable( podInput->getObservationsAndTimes( ) );
472  int totalNumberOfObservations = observationNumberPair.second;
473 
474  // Declare variables to be returned (i.e. results from best iteration)
475  double bestResidual = TUDAT_NAN;
476  ParameterVectorType bestParameterEstimate = ParameterVectorType::Constant( parameterVectorSize, TUDAT_NAN );
477  Eigen::VectorXd bestTransformationData = Eigen::VectorXd::Constant( parameterVectorSize, TUDAT_NAN );
478  Eigen::VectorXd bestResiduals = Eigen::VectorXd::Constant( totalNumberOfObservations, TUDAT_NAN );
479  Eigen::MatrixXd bestInformationMatrix = Eigen::MatrixXd::Constant( totalNumberOfObservations, parameterVectorSize, TUDAT_NAN );
480  Eigen::VectorXd bestWeightsMatrixDiagonal = Eigen::VectorXd::Constant( totalNumberOfObservations, TUDAT_NAN );
481  Eigen::MatrixXd bestInverseNormalizedCovarianceMatrix = Eigen::MatrixXd::Constant( parameterVectorSize, parameterVectorSize, TUDAT_NAN );
482 
483  std::vector< Eigen::VectorXd > residualHistory;
484  std::vector< Eigen::VectorXd > parameterHistory;
485  std::vector< std::vector< std::map< TimeType, Eigen::Matrix< ObservationScalarType, Eigen::Dynamic, 1 > > > > dynamicsHistoryPerIteration;
486  std::vector< std::vector< std::map< TimeType, Eigen::VectorXd > > > dependentVariableHistoryPerIteration;
487 
488  // Declare residual bookkeeping variables
489  std::vector< double > rmsResidualHistory;
490  double residualRms;
491 
492  // Declare variables to be used in loop.
493 
494  // Set current parameter estimate as both previous and current estimate
495  ParameterVectorType newParameterEstimate = currentParameterEstimate_ +
496  podInput->getInitialParameterDeviationEstimate( );
497  ParameterVectorType oldParameterEstimate = currentParameterEstimate_;
498 
499  int numberOfEstimatedParameters = parameterVectorSize;
500 
501  bool exceptionDuringPropagation = false, exceptionDuringInversion = false;
502  // Iterate until convergence (at least once)
503  int numberOfIterations = 0;
504  do
505  {
506  try
507  {
508  // Re-integrate equations of motion and variational equations with new parameter estimate.
509  if( ( numberOfIterations > 0 ) ||( podInput->getReintegrateEquationsOnFirstIteration( ) ) )
510  {
511  resetParameterEstimate( newParameterEstimate, podInput->getReintegrateVariationalEquations( ) );
512  }
513 
514  if( podInput->getSaveStateHistoryForEachIteration( ) )
515  {
516  dynamicsHistoryPerIteration.push_back(
517  variationalEquationsSolver_->getDynamicsSimulatorBase( )->getEquationsOfMotionNumericalSolutionBase( ) );
518  dependentVariableHistoryPerIteration.push_back(
519  variationalEquationsSolver_->getDynamicsSimulatorBase( )->getDependentVariableNumericalSolutionBase( ) );
520  }
521  }
522  catch( std::runtime_error )
523  {
524  std::cerr<<"Error when resetting parameters during parameter estimation, terminating estimation"<<std::endl;
525  exceptionDuringPropagation = true;
526  break;
527  }
528 
529  oldParameterEstimate = newParameterEstimate;
530 
531  if( podInput->getPrintOutput( ) )
532  {
533  std::cout << "Calculating residuals and partials " << totalNumberOfObservations << std::endl;
534  }
535  // Calculate residuals and observation matrix for current parameter estimate.
536  std::pair< Eigen::VectorXd, Eigen::MatrixXd > residualsAndPartials;
537  calculateObservationMatrixAndResiduals(
538  podInput->getObservationsAndTimes( ), parameterVectorSize, totalNumberOfObservations, residualsAndPartials );
539 
540  Eigen::VectorXd transformationData = normalizeObservationMatrix( residualsAndPartials.second );
541 
542  Eigen::MatrixXd normalizedInverseAprioriCovarianceMatrix = Eigen::MatrixXd::Zero(
543  numberOfEstimatedParameters, numberOfEstimatedParameters );
544 
545  Eigen::MatrixXd inverseAPrioriCovariance = podInput->getInverseOfAprioriCovariance( );
546  for( int j = 0; j < numberOfEstimatedParameters; j++ )
547  {
548  for( int k = 0; k < numberOfEstimatedParameters; k++ )
549  {
550  normalizedInverseAprioriCovarianceMatrix( j, k ) = inverseAPrioriCovariance( j, k ) /
551  ( transformationData( j ) * transformationData( k ) );
552  }
553  }
554 
555  // Perform least squares calculation for correction to parameter vector.
556  std::pair< Eigen::VectorXd, Eigen::MatrixXd > leastSquaresOutput;
557  try
558  {
559  leastSquaresOutput =
560  std::move( linear_algebra::performLeastSquaresAdjustmentFromInformationMatrix(
561  residualsAndPartials.second.block( 0, 0, residualsAndPartials.second.rows( ), numberOfEstimatedParameters ),
562  residualsAndPartials.first, getConcatenatedWeightsVector( podInput->getWeightsMatrixDiagonals( ) ),
563  normalizedInverseAprioriCovarianceMatrix ) );
564  }
565  catch( std::runtime_error )
566  {
567  std::cerr<<"Error when solving normal equations during parameter estimation, terminating estimation"<<std::endl;
568  exceptionDuringInversion = true;
569  break;
570  }
571 
572 
573  ParameterVectorType parameterAddition =
574  ( leastSquaresOutput.first.cwiseQuotient( transformationData.segment( 0, numberOfEstimatedParameters ) ) ).
575  template cast< ObservationScalarType >( );
576 
577 
578  // Update value of parameter vector
579  newParameterEstimate = oldParameterEstimate + parameterAddition;
580 
581  if( podInput->getSaveResidualsAndParametersFromEachIteration( ) )
582  {
583  residualHistory.push_back( residualsAndPartials.first );
584  if( numberOfIterations == 0 )
585  {
586  parameterHistory.push_back( oldParameterEstimate.template cast< double >( ) );
587  }
588  parameterHistory.push_back( newParameterEstimate.template cast< double >( ) );
589  }
590 
591  oldParameterEstimate = newParameterEstimate;
592 
593  if( podInput->getPrintOutput( ) )
594  {
595  std::cout << "Parameter update" << parameterAddition.transpose( ) << std::endl;
596  }
597 
598  // Calculate mean residual for current iteration.
599  residualRms = linear_algebra::getVectorEntryRootMeanSquare( residualsAndPartials.first );
600 
601  rmsResidualHistory.push_back( residualRms );
602  if( podInput->getPrintOutput( ) )
603  {
604  std::cout << "Current residual: " << residualRms << std::endl;
605  }
606 
607  // If current iteration is better than previous one, update 'best' data.
608  if( residualRms < bestResidual || !( bestResidual == bestResidual ) )
609  {
610  bestResidual = residualRms;
611  bestParameterEstimate = std::move( oldParameterEstimate );
612  bestResiduals = std::move( residualsAndPartials.first );
613  if( podInput->getSaveInformationMatrix( ) )
614  {
615  bestInformationMatrix = std::move( residualsAndPartials.second );
616  }
617  bestWeightsMatrixDiagonal = std::move( getConcatenatedWeightsVector( podInput->getWeightsMatrixDiagonals( ) ) );
618  bestTransformationData = std::move( transformationData );
619  bestInverseNormalizedCovarianceMatrix = std::move( leastSquaresOutput.second );
620  }
621 
622 
623  // Increment number of iterations
624  numberOfIterations++;
625 
626  // Check for convergence
627  } while( convergenceChecker->isEstimationConverged( numberOfIterations, rmsResidualHistory ) == false );
628 
629  if( podInput->getPrintOutput( ) )
630  {
631  std::cout << "Final residual: " << bestResidual << std::endl;
632  }
633 
634  return std::make_shared< PodOutput< ObservationScalarType > >(
635  bestParameterEstimate, bestResiduals, bestInformationMatrix, bestWeightsMatrixDiagonal, bestTransformationData,
636  bestInverseNormalizedCovarianceMatrix, bestResidual, residualHistory, parameterHistory, exceptionDuringInversion,
637  exceptionDuringPropagation );
638  }

680  for( typename std::map< observation_models::LinkEnds, std::pair< std::map< TimeType, ObservationScalarType >, observation_models::LinkEndType > >::const_iterator stationIterator =
681  inputIterator->second.begin( ); stationIterator != inputIterator->second.end( ); stationIterator++ )
