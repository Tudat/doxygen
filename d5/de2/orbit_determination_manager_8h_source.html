110  {

132  rmsResidualHistory.at( rmsResidualHistory.size( ) - 2 ) ) < minimumResidualChange_ )

135  }

455  boost::shared_ptr< PodOutput< ObservationScalarType > > estimateParameters(
456  const boost::shared_ptr< PodInput< ObservationScalarType, TimeType > >& podInput,
457  const boost::shared_ptr< EstimationConvergenceChecker > convergenceChecker =
458  boost::make_shared< EstimationConvergenceChecker >( ) )
459  {
460  currentParameterEstimate_ = parametersToEstimate_->template getFullParameterValues< ObservationScalarType >( );
461 
462  // Get size of parameter vector and number of observations (total and per type)
463  int parameterVectorSize = currentParameterEstimate_.size( );
464  std::pair< std::map< observation_models::ObservableType, int >, int > observationNumberPair =
465  getNumberOfObservationsPerObservable( podInput->getObservationsAndTimes( ) );
466  int totalNumberOfObservations = observationNumberPair.second;
467 
468  // Declare variables to be returned (i.e. results from best iteration)
469  double bestResidual = std::numeric_limits< double >::max( );
470  ParameterVectorType bestParameterEstimate = ParameterVectorType::Zero( parameterVectorSize );
471  Eigen::VectorXd bestTransformationData = Eigen::VectorXd::Zero( parameterVectorSize );
472  Eigen::VectorXd bestResiduals = Eigen::VectorXd::Zero( totalNumberOfObservations );
473  Eigen::MatrixXd bestInformationMatrix = Eigen::MatrixXd::Zero( totalNumberOfObservations, parameterVectorSize );
474  Eigen::VectorXd bestWeightsMatrixDiagonal = Eigen::VectorXd::Zero( totalNumberOfObservations );
475  Eigen::MatrixXd bestInverseNormalizedCovarianceMatrix = Eigen::MatrixXd::Zero( parameterVectorSize, parameterVectorSize );
476 
477  std::vector< Eigen::VectorXd > residualHistory;
478  std::vector< Eigen::VectorXd > parameterHistory;
479  std::vector< std::vector< std::map< TimeType, Eigen::Matrix< ObservationScalarType, Eigen::Dynamic, 1 > > > > dynamicsHistoryPerIteration;
480  std::vector< std::vector< std::map< TimeType, Eigen::VectorXd > > > dependentVariableHistoryPerIteration;
481 
482  // Declare residual bookkeeping variables
483  std::vector< double > rmsResidualHistory;
484  double residualRms;
485 
486  // Declare variables to be used in loop.
487 
488  // Set current parameter estimate as both previous and current estimate
489  ParameterVectorType newParameterEstimate = currentParameterEstimate_ +
490  podInput->getInitialParameterDeviationEstimate( );
491  ParameterVectorType oldParameterEstimate = currentParameterEstimate_;
492 
493  int numberOfEstimatedParameters = parameterVectorSize;
494 
495  // Iterate until convergence (at least once)
496  int numberOfIterations = 0;
497  do
498  {
499  // Re-integrate equations of motion and variational equations with new parameter estimate.
500  if( ( numberOfIterations > 0 ) ||( podInput->getReintegrateEquationsOnFirstIteration( ) ) )
501  {
502  resetParameterEstimate( newParameterEstimate, podInput->getReintegrateVariationalEquations( ) );
503  }
504 
505  if( podInput->getSaveStateHistoryForEachIteration( ) )
506  {
507  dynamicsHistoryPerIteration.push_back(
508  variationalEquationsSolver_->getDynamicsSimulatorBase( )->getEquationsOfMotionNumericalSolutionBase( ) );
509  dependentVariableHistoryPerIteration.push_back(
510  variationalEquationsSolver_->getDynamicsSimulatorBase( )->getDependentVariableNumericalSolutionBase( ) );
511  }
512 
513  oldParameterEstimate = newParameterEstimate;
514 
515  if( podInput->getPrintOutput( ) )
516  {
517  std::cout<<"Calculating residuals and partials "<<totalNumberOfObservations<<std::endl;
518  }
519  // Calculate residuals and observation matrix for current parameter estimate.
520  std::pair< Eigen::VectorXd, Eigen::MatrixXd > residualsAndPartials;
521  calculateObservationMatrixAndResiduals(
522  podInput->getObservationsAndTimes( ), parameterVectorSize, totalNumberOfObservations, residualsAndPartials );
523 
524  Eigen::VectorXd transformationData = normalizeObservationMatrix( residualsAndPartials.second );
525 
526  Eigen::MatrixXd normalizedInverseAprioriCovarianceMatrix = Eigen::MatrixXd::Zero(
527  numberOfEstimatedParameters, numberOfEstimatedParameters );
528 
529  Eigen::MatrixXd inverseAPrioriCovariance = podInput->getInverseOfAprioriCovariance( );
530  for( int j = 0; j < numberOfEstimatedParameters; j++ )
531  {
532  for( int k = 0; k < numberOfEstimatedParameters; k++ )
533  {
534  normalizedInverseAprioriCovarianceMatrix( j, k ) = inverseAPrioriCovariance( j, k ) /
535  ( transformationData( j ) * transformationData( k ) );
536  }
537  }
538 
539  // Perform least squares calculation for correction to parameter vector.
540  std::pair< Eigen::VectorXd, Eigen::MatrixXd > leastSquaresOutput =
541  linear_algebra::performLeastSquaresAdjustmentFromInformationMatrix(
542  residualsAndPartials.second.block( 0, 0, residualsAndPartials.second.rows( ), numberOfEstimatedParameters ),
543  residualsAndPartials.first, getConcatenatedWeightsVector( podInput->getWeightsMatrixDiagonals( ) ),
544  normalizedInverseAprioriCovarianceMatrix );
545  ParameterVectorType parameterAddition =
546  ( leastSquaresOutput.first.cwiseQuotient( transformationData.segment( 0, numberOfEstimatedParameters ) ) ).
547  template cast< ObservationScalarType >( );
548 
549 
550  // Update value of parameter vector
551  newParameterEstimate = oldParameterEstimate + parameterAddition;
552 
553  if( podInput->getSaveResidualsAndParametersFromEachIteration( ) )
554  {
555  residualHistory.push_back( residualsAndPartials.first );
556  if( numberOfIterations == 0 )
557  {
558  parameterHistory.push_back( oldParameterEstimate.template cast< double >( ) );
559  }
560  parameterHistory.push_back( newParameterEstimate.template cast< double >( ) );
561  }
562 
563  oldParameterEstimate = newParameterEstimate;
564 
565 
566  if( podInput->getPrintOutput( ) )
567  {
568  std::cout<<"Parameter update"<<parameterAddition.transpose( )<<std::endl;
569  }
570 
571  // Calculate mean residual for current iteration.
572  residualRms = linear_algebra::getVectorEntryRootMeanSquare( residualsAndPartials.first );
573 
574  rmsResidualHistory.push_back( residualRms );
575  if( podInput->getPrintOutput( ) )
576  {
577  std::cout<<"Current residual: "<<residualRms<<std::endl;
578  }
579 
580  // If current iteration is better than previous one, update 'best' data.
581  if( residualRms < bestResidual )
582  {
583  bestResidual = residualRms;
584  bestParameterEstimate = oldParameterEstimate;
585  bestResiduals = residualsAndPartials.first;
586  if( podInput->getSaveInformationMatrix( ) )
587  {
588  bestInformationMatrix = residualsAndPartials.second;
589  }
590  bestWeightsMatrixDiagonal = getConcatenatedWeightsVector( podInput->getWeightsMatrixDiagonals( ) );
591  bestTransformationData = transformationData;
592  bestInverseNormalizedCovarianceMatrix = leastSquaresOutput.second;
593  }
594 
595 
596  // Increment number of iterations
597  numberOfIterations++;
598 
599  // Check for convergence
600  } while( convergenceChecker->isEstimationConverged( numberOfIterations, rmsResidualHistory ) == false );
601 
602 
603  std::cout<<"Final residual: "<<bestResidual<<std::endl;
604 
605  return boost::make_shared< PodOutput< ObservationScalarType > >(
606  bestParameterEstimate, bestResiduals, bestInformationMatrix, bestWeightsMatrixDiagonal, bestTransformationData,
607  bestInverseNormalizedCovarianceMatrix, bestResidual, residualHistory, parameterHistory );
608  }

784  }

795  }
796  }
797 
798  if( integrateAndEstimateOrbit_ )
799  {

802  }
803  else if( propagatorSettings == NULL )
804  {
805  if( !dynamicsIsMultiArc_ )
806  {
807  stateTransitionAndSensitivityMatrixInterface_ = boost::make_shared<
808  propagators::SingleArcCombinedStateTransitionAndSensitivityMatrixInterface >(

810  boost::shared_ptr< interpolators::OneDimensionalInterpolator< double, Eigen::MatrixXd > >( ),
811  0, parametersToEstimate_->getParameterSetSize( ) );
812  }

815  stateTransitionAndSensitivityMatrixInterface_ = boost::make_shared<
816  propagators::MultiArcCombinedStateTransitionAndSensitivityMatrixInterface >(
817  std::vector< boost::shared_ptr< interpolators::OneDimensionalInterpolator< double, Eigen::MatrixXd > > >( ),
818  std::vector< boost::shared_ptr< interpolators::OneDimensionalInterpolator< double, Eigen::MatrixXd > > >( ),
819  std::vector< double >( ),
820  0, parametersToEstimate_->getParameterSetSize( ) );
821  }
822  }
823  else if( propagatorSettings != NULL )
824  {
825  throw std::runtime_error( "Error, cannot parse propagator settings without estimating dynamics in OrbitDeterminationManager" );
826  }
827 
828  // Iterate over all observables and create observation managers.
829  for( SortedObservationSettingsMap::const_iterator observablesIterator = observationSettingsMap.begin( );
830  observablesIterator != observationSettingsMap.end( ); observablesIterator++ )
831  {
832  // Create observation manager for current observable.
833  observationManagers_[ observablesIterator->first ] =
834  createObservationManagerBase< ObservationScalarType, TimeType >(
835  observablesIterator->first, observablesIterator->second, bodyMap, parametersToEstimate_,
836  stateTransitionAndSensitivityMatrixInterface_ );
837  }
838 
839  // Set current parameter estimate from body initial states and parameter set.
840  currentParameterEstimate_ = parametersToEstimate_->template getFullParameterValues< ObservationScalarType >( );
841

843  parametersToEstimate_->getDoubleParameters( );

845  parameterIterator = doubleParameters.begin( ); parameterIterator != doubleParameters.end( ); parameterIterator++ )

848  parameterIterator->second->getParameterName( ).first ) )

851  }
852  }
853

855  parametersToEstimate_->getVectorParameters( );

857  parameterIterator = vectorParameters.begin( ); parameterIterator != vectorParameters.end( ); parameterIterator++ )
858  {

860  parameterIterator->second->getParameterName( ).first ) )
861  {
862  for( int i = 0; i < parameterIterator->second->getParameterSize( ); i++ )
863  {
864  observationLinkParameterIndices_.push_back( parameterIterator->first + i );
865  }
866  }
867  }
868 
869  }
