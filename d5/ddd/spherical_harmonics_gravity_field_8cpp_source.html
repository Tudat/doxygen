28 double calculateSphericalHarmonicGravitationalPotential(
29  const Eigen::Vector3d& bodyFixedPosition, const double gravitationalParameter,
30  const double referenceRadius,
31  const Eigen::MatrixXd& cosineCoefficients,
32  const Eigen::MatrixXd& sineCoefficients,
33  std::shared_ptr< basic_mathematics::SphericalHarmonicsCache > sphericalHarmonicsCache,
34  const int minimumumDegree,
35  const int minimumumOrder )
36 {
37  // Initialize (distance/reference radius)^n (n=ratioToPowerDegree)
38  double ratioToPowerDegree = 1.0;
39  double radiusRatio = referenceRadius / bodyFixedPosition.norm( );
40 
41  // Declare local variables used in calculation
42  double legendrePolynomial = 0.0;
43  double singleDegreeTerm = 0.0;
44 
45  // Determine body fixed spherical position of body udnergoing acceleration.
46  Eigen::Vector3d sphericalPositon =
47  coordinate_conversions::convertCartesianToSpherical( bodyFixedPosition );
48  double latitude = mathematical_constants::PI / 2.0 - sphericalPositon.y( );
49  double longitude = sphericalPositon.z( );
50 
51  double potential = 0.0;
52  int startDegree = 0;
53 
54  // Initialize value of potential to 1 (C_{0,0})
55  if( minimumumDegree == 0 )
56  {
57  potential = 1.0;
58  startDegree = 1;
59  }
60  else
61  {
62  startDegree = minimumumDegree;
63  ratioToPowerDegree *= basic_mathematics::raiseToIntegerPower< double >( radiusRatio, startDegree - 1 );
64  }
65 
66  basic_mathematics::LegendreCache& legendreCacheReference = *sphericalHarmonicsCache->getLegendreCache( );
67  legendreCacheReference.update( std::sin( latitude ) );
68 
69  // Iterate over all degrees
70  for( int degree = startDegree; degree < cosineCoefficients.rows( ); degree++ )
71  {
72  singleDegreeTerm = 0.0;
73 
74  // Iterate over all orders in current degree for which coefficients are provided.
75  for( int order = minimumumOrder; ( order < cosineCoefficients.cols( ) &&
76  order <= degree ); order++ )
77  {
78  // Calculate legendre polynomial (geodesy-normalized) at current degree and order
79  legendrePolynomial = basic_mathematics::computeGeodesyLegendrePolynomialFromCache(
80  degree, order, legendreCacheReference );
81 
82  // Calculate contribution to potential from current degree and order
83  singleDegreeTerm += legendrePolynomial * ( cosineCoefficients( degree, order ) *
84  std::cos( order * longitude ) +
85  sineCoefficients( degree, order ) *
86  std::sin( order * longitude ) );
87  }
88 
89  // Add potential contributions from current degree to toal value.
90  ratioToPowerDegree *= radiusRatio;
91  potential += singleDegreeTerm * ratioToPowerDegree;
92  }
93 
94  // Multiply by central term and return
95  return potential * gravitationalParameter / bodyFixedPosition.norm( );
96 }
