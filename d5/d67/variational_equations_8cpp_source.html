27 void VariationalEquations::setBodyStatePartialMatrix( )
28 {
29  // Initialize partial matrix
30  variationalMatrix_.setZero( );
31 
32  if( dynamicalStatesToEstimate_.count( propagators::transational_state ) > 0 )
33  {
34  int startIndex = stateTypeStartIndices_.at( propagators::transational_state );
35  for( unsigned int i = 0; i < dynamicalStatesToEstimate_.at( propagators::transational_state ).size( ); i++ )
36  {
37  variationalMatrix_.block( startIndex + i * 6, startIndex + i * 6 + 3, 3, 3 ).setIdentity( );
38  }
39  }
40 
41  // Iterate over all bodies undergoing accelerations for which initial condition is to be estimated.
42  for( std::map< IntegratedStateType, std::vector< std::multimap< std::pair< int, int >,
43  boost::function< void( Eigen::Block< Eigen::MatrixXd > ) > > > >::iterator
44  typeIterator = statePartialList_.begin( ); typeIterator != statePartialList_.end( ); typeIterator++ )
45  {
46  int startIndex = stateTypeStartIndices_.at( typeIterator->first );
47  int currentStateSize = getSingleIntegrationSize( typeIterator->first );
48  int entriesToSkipPerEntry = currentStateSize - currentStateSize /
49  getSingleIntegrationDifferentialEquationOrder( typeIterator->first );
50  for( unsigned int i = 0; i < typeIterator->second.size( ); i++ )
51  {
52  // Iterate over all bodies exerting an acceleration on this body.
53  for( statePartialIterator_ = typeIterator->second.at( i ).begin( );
54  statePartialIterator_ != typeIterator->second.at( i ).end( );
55  statePartialIterator_++ )
56  {
57  statePartialIterator_->second(
58  variationalMatrix_.block(
59  startIndex + entriesToSkipPerEntry + i* currentStateSize, statePartialIterator_->first.first,
60  currentStateSize - entriesToSkipPerEntry, statePartialIterator_->first.second ) );
61 
62  }
63  }
64  }
65 
66  // Correct partials for hierarchical dynamics
67  for( unsigned int i = 0; i < statePartialAdditionIndices_.size( ); i++ )
68  {
69  variationalMatrix_.block( 0, statePartialAdditionIndices_.at( i ).second, totalDynamicalStateSize_, 3 ) +=
70  variationalMatrix_.block( 0, statePartialAdditionIndices_.at( i ).first, totalDynamicalStateSize_, 3 );
71  }
72 }

164  std::make_pair(

166  stateTypeStartIndices_.at( estimatedStateIterator->first ),
